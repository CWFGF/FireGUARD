347 template <typename T, typename V>
348 T binary_find(const T lower,
349  const T upper,
350  const double value,
351  const std::function<V(T)>& fct)
352 {
353  const auto mid = lower + (upper - lower) / 2;
354  if (lower == upper)
355  {
356  return lower;
357  }
358  if (fct(mid) < value)
359  {
360  return binary_find(lower, mid, value, fct);
361  }
362  return binary_find(mid + 1, upper, value, fct);
363 }
374 template <typename T, typename V>
375 T binary_find_checked(const T lower,
376  const T upper,
377  const double value,
378  const std::function<V(T)>& fct)
379 {
380  if (fct(lower) < value)
381  {
382  return lower;
383  }
384  if (fct(upper) >= value)
385  {
386  return upper;
387  }
388  return binary_find(lower, upper, value, fct);
389 }
