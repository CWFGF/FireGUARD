67 def Project(src, outputShapefile, outSpatialRef):
68  """!
69  Project to given spatial reference
70  @param src File to project
71  @param outputShapefile Destination file to save projection to
72  @param outSpatialRef Spatial reference to project into
73  @return None
74  """
75  # https://pcjericks.github.io/py-gdalogr-cookbook/projection.html
76  # input SpatialReference
77  # get the input layer
78  inDataSet = ogr.GetDriverByName('ESRI Shapefile').Open(src)
79  inLayer = inDataSet.GetLayer()
80  # create the CoordinateTransformation
81  inSpatialRef = GetSpatialReference(src)
82  #~ print(inSpatialRef.ExportToWkt())
83  #~ print(outSpatialRef.ExportToWkt())
84  coordTrans = osr.CoordinateTransformation(inSpatialRef, outSpatialRef)
85  # create the output layer
86  Delete(outputShapefile)
87  outDataSet = ogr.GetDriverByName('ESRI Shapefile').CreateDataSource(outputShapefile)
88  outLayer = outDataSet.CreateLayer(inLayer.GetName(), outSpatialRef, geom_type=ogr.wkbMultiPolygon)
89  # add fields
90  inLayerDefn = inLayer.GetLayerDefn()
91  for i in range(0, inLayerDefn.GetFieldCount()):
92  fieldDefn = inLayerDefn.GetFieldDefn(i)
93  if 'shape' not in fieldDefn.GetName().lower():
94  outLayer.CreateField(fieldDefn)
95  # get the output layer's feature definition
96  outLayerDefn = outLayer.GetLayerDefn()
97  # loop through the input features
98  inFeature = inLayer.GetNextFeature()
99  while inFeature:
100  # get the input geometry
101  geom = inFeature.GetGeometryRef()
102  # reproject the geometry
103  geom.Transform(coordTrans)
104  # create a new feature
105  outFeature = ogr.Feature(outLayerDefn)
106  # set the geometry and attribute
107  outFeature.SetGeometry(geom)
108  for i in range(0, outLayerDefn.GetFieldCount()):
109  outFeature.SetField(outLayerDefn.GetFieldDefn(i).GetNameRef(), inFeature.GetField(i))
110  # add the feature to the shapefile
111  outLayer.CreateFeature(outFeature)
112  # dereference the features and get the next input feature
113  del outFeature
114  inFeature = inLayer.GetNextFeature()
115  # Save and close the shapefiles
116  del coordTrans
117  del inSpatialRef
118  del inLayer
119  del inDataSet
120  del inFeature
121  del outLayerDefn
122  del inLayerDefn
123  del outLayer
124  del outDataSet
125  # create projection file for output
126  prj = os.path.splitext(outputShapefile)[0] + ".prj"
127  with open (prj, 'w') as file:
128  file.write(outSpatialRef.ExportToWkt())

172  Convert a shapefile into a raster with the given spatial reference

188  print("Raster must have square pixels")
