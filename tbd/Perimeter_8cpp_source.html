34  BurnedMap(const Grid<unsigned char, unsigned char>& perim_grid, const Environment& env)

36  *env.makeMap<unsigned char>(static_cast<unsigned char>(0)))
37  {
38  // HACK: fix offset if the perimeter raster is different from this one
39  logging::check_fatal(0 != strcmp(perim_grid.proj4().c_str(), this->proj4().c_str()),
40  "Invalid projection for input perimeter raster - %s instead of %s",
41  perim_grid.proj4().c_str(),
42  this->proj4().c_str());
43  logging::check_fatal(perim_grid.cellSize() != this->cellSize(),
44  "Invalid cell size for input perimeter raster - %f instead of %f",
45  perim_grid.cellSize(),
46  this->cellSize());
47  const auto offset_x = static_cast<Idx>((this->xllcorner() - perim_grid.xllcorner()) / this->cellSize());
48  const auto perim_origin = static_cast<Idx>(perim_grid.rows() + perim_grid.yllcorner() / this->cellSize());
49  const auto this_origin = static_cast<Idx>(this->rows() + this->yllcorner() / this->cellSize());
50  const auto offset_y = static_cast<Idx>((perim_origin - this_origin));
51  // make sure we don't go out of bounds on grid
52  const auto min_column = static_cast<Idx>(offset_x < 0 ? abs(offset_x) : 0);
53  const auto max_columns = min(this->columns(), perim_grid.columns());
54  const auto min_row = static_cast<Idx>(offset_y < 0 ? abs(offset_y) : 0);
55  const auto max_rows = min(this->rows(),
56  static_cast<Idx>(perim_grid.rows() - abs(offset_y)));
57  logging::note("Correcting perimeter raster offset by %dx%d cells",
58  offset_x,
59  offset_y);
60  size_t count = 0;
61  // since it was read in as a vector we need to check all the cells
62  for (auto r = min_row; r < max_rows; ++r)
63  {
64  for (auto c = min_column; c < max_columns; ++c)
65  {
66  const auto loc = env.cell(r, c);
67  const Location fixed_loc(static_cast<Idx>(r + offset_y),
68  static_cast<Idx>(c + offset_x));
69  const auto value = perim_grid.at(fixed_loc);
70  if (value != perim_grid.noData() && !fuel::is_null_fuel(loc))
71  {
72  this->GridMap<unsigned char, unsigned char>::set(loc, value);
73  ++count;
74  }
75  }
76  }
77 #ifndef NDEBUG
78  for (auto& kv : data)

102 Perimeter::Perimeter(const Location& location,
103  const size_t size,
104  const wx::FwiWeather&,
105  const Environment& env)
106 {
107  // NOTE: FwiWeather is unused but could change this to try doing length to breadth ratio
108  auto perim_grid = env.makeMap<unsigned char>(0);
109  // want to find cells in the area that fill up the size we're looking for
110  size_t count = 0;
111  // convert into number of cells
112  const auto num_cells = size / (100.0 * 100.0 / (perim_grid->cellSize() * perim_grid->cellSize()));
113  auto max_distance = sqrt(num_cells / M_PI);
114  perim_grid->set(location, 1);
115  ++count;
116  // HACK: assume fuel for origin matches the rest of the fire
117  while (num_cells > count)
118  {
119  const auto range = static_cast<Idx>(ceil(max_distance));
120  for (auto x = -range; x <= range && num_cells > count; ++x)
121  {
122  for (auto y = -range; y <= range && num_cells > count; ++y)
123  {
124  // look at any cell that's within the range
125  if (sqrt(util::pow_int<2>(x) + util::pow_int<2>(y)) < max_distance)
126  {
127  const auto r = static_cast<Idx>(location.row() + x);
128  const auto c = static_cast<Idx>(location.column() + y);
129  const auto loc = env.cell(r, c);
130  if (1 != perim_grid->at(loc) && !fuel::is_null_fuel(loc))
131  {
132  perim_grid->set(loc, 1);
133  ++count;
134  }
135  }
136  }
137  }
138  max_distance += 0.1;
139  }
140  const BurnedMap burned(*perim_grid, env);
141  burned_ = burned.makeList();
142  edge_ = burned.makeEdge();
143 }
