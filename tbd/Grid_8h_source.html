486 template <typename T>
487 [[nodiscard]] GridBase read_header(TIFF* tif, GTIF* gtif)
488 {
489  GTIFDefn definition;
490  if (GTIFGetDefn(gtif, &definition))
491  {
492  void* data;
493  uint32 count;
494  int columns;
495  int rows;
496  TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &columns);
497  // logging::check_fatal(columns > numeric_limits<Idx>::max(),
498  // "Cannot use grids with more than %d columns",
499  // numeric_limits<Idx>::max());
500  TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &rows);
501  // logging::check_fatal(rows > numeric_limits<Idx>::max(),
502  // "Cannot use grids with more than %d rows",
503  // numeric_limits<Idx>::max());
504  TIFFGetField(tif, TIFFTAG_GDAL_NODATA, &count, &data);
505  logging::check_fatal(0 == count, "NODATA value is not set in input");
506  const auto nodata = stoi(string(static_cast<char*>(data)));
507  double x = 0.0;
508  double y = rows;
509  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
510  "Unable to translate image to PCS coordinates.");
511  const auto yllcorner = y;
512  const auto xllcorner = x;
513  logging::debug("Lower left for header is (%f, %f)", xllcorner, yllcorner);
514  double adf_coefficient[6] = {0};
515  x = 0.5;
516  y = 0.5;
517  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
518  "Unable to translate image to PCS coordinates.");
519  adf_coefficient[4] = x;
520  adf_coefficient[5] = y;
521  x = 1.5;
522  y = 0.5;
523  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
524  "Unable to translate image to PCS coordinates.");
525  const auto cell_width = x - adf_coefficient[4];
526  x = 0.5;
527  y = 1.5;
528  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
529  "Unable to translate image to PCS coordinates.");
530  const auto cell_height = y - adf_coefficient[5];
531  logging::check_fatal(cell_width != -cell_height,
532  "Can only use grids with square pixels");
533  logging::debug("Cell size is %f", cell_width);
534  const auto proj4_char = GTIFGetProj4Defn(&definition);
535  auto proj4 = string(proj4_char);
536  free(proj4_char);
537  const auto zone_pos = proj4.find("+zone=");
538  if (string::npos != zone_pos && string::npos != proj4.find("+proj=utm"))
539  {
540  // convert from utm zone to tmerc
541  const auto zone_str = proj4.substr(zone_pos + 6);
542  const auto zone = stoi(zone_str);
543  // zone 15 is -93 and other zones are 6 degrees difference
544  const auto degrees = static_cast<int>(6.0 * (zone - 15.0) - 93);
545  // HACK: assume utm zone is at start
546  proj4 = string(
547  "+proj=tmerc +lat_0=0.000000000 +lon_0=" + to_string(degrees) + ".000000000 +k=0.999600 +x_0=500000.000 +y_0=0.000");
548  }
549  const auto xurcorner = xllcorner + cell_width * columns;
550  const auto yurcorner = yllcorner + cell_width * rows;
551  return {
552  cell_width,
553  nodata,
554  xllcorner,
555  yllcorner,
556  xurcorner,
557  yurcorner,
558  string(proj4)
559  };
560  }
561  throw runtime_error("Cannot read TIFF header");
562 }
