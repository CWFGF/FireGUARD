37 static unique_ptr<data::GridMap<IntensitySize>> acquire_map(const Model& model) noexcept
38 {
39  try
40  {
41  lock_guard<mutex> lock(MUTEX_MAPS);
42  if (!MAPS.empty())

44  auto result = std::move(MAPS.at(MAPS.size() - 1));

98 {
99  // implement here so we can just lock once
100  lock_guard<mutex> lock(mutex_);
101  const auto x = location.column();
102  const auto y = location.row();
103  const auto min_row = static_cast<Idx>(max(y - 1, 0));
104  const auto max_row = min(y + 1, this->rows() - 1);
105  const auto min_column = static_cast<Idx>(max(x - 1, 0));
106  const auto max_column = min(x + 1, this->columns() - 1);
107  for (auto r = min_row; r <= max_row; ++r)
108  {
109  // auto h = static_cast<size_t>(r) * MAX_COLUMNS + min_column;

111  {
112  // actually check x, y too since we care if the cell itself is burned

117  }
