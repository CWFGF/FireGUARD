404  template <class R>
405  void saveToTiffFile(const string& dir,
406  const string& base_name,
407  std::function<R(T)> convert) const
408  {
409  Idx min_row = 0;
410  // Idx num_rows = this->rows();
411  Idx min_column = 0;
412  // Idx num_columns = this->columns();
413  const double xll = this->xllcorner() + min_column * this->cellSize();
414  // offset is different for y since it's flipped
415  const double yll = this->yllcorner() + (min_row) * this->cellSize();
416  logging::extensive("Lower left corner is (%f, %f)", xll, yll);
417  constexpr uint32 tileWidth = 256;
418  constexpr uint32 tileHeight = 256;
419  // ensure this is always divisible by tile size
420  static_assert(0 == MAX_ROWS % tileWidth);
421  static_assert(0 == MAX_COLUMNS % tileHeight);
422  uint32 width = this->columns();
423  uint32 height = this->rows();
424  string filename = dir + base_name + ".tif";
425  TIFF* tif = GeoTiffOpen(filename.c_str(), "w");
426  auto gtif = GTIFNew(tif);
427  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
428  const double xul = this->xllcorner();
429  const double yul = this->yllcorner() + (this->cellSize() * this->rows());
430  double tiePoints[6] = {
431  0.0,
432  0.0,
433  0.0,
434  xul,
435  yul,
436  0.0};
437  double pixelScale[3] = {
438  this->cellSize(),
439  this->cellSize(),
440  0.0};
441  // HACK: why does this have to be +1?
442  uint32 bps = std::numeric_limits<R>::digits + 1;
443  TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, width);
444  TIFFSetField(tif, TIFFTAG_IMAGELENGTH, height);
445  TIFFSetField(tif, TIFFTAG_SAMPLESPERPIXEL, 1);
446  TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, bps);
447  TIFFSetField(tif, TIFFTAG_TILEWIDTH, tileWidth);
448  TIFFSetField(tif, TIFFTAG_TILELENGTH, tileHeight);
449  TIFFSetField(tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
450  TIFFSetField(tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
451  TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_LZW);
452  GTIFSetFromProj4(gtif, this->proj4().c_str());
453  TIFFSetField(tif, TIFFTAG_GEOTIEPOINTS, 6, tiePoints);
454  TIFFSetField(tif, TIFFTAG_GEOPIXELSCALE, 3, pixelScale);
455  size_t tileSize = tileWidth * tileHeight;
456  R* buf = (R*)_TIFFmalloc(tileSize * sizeof(R));
457  for (size_t i = 0; i < width; i += tileWidth)
458  {
459  for (size_t j = 0; j < height; j += tileHeight)
460  {
461  // need to put data from grid into buffer, but flipped vertically
462  for (size_t x = 0; x < tileWidth; ++x)
463  {
464  for (size_t y = 0; y < tileHeight; ++y)
465  {
466  const size_t actual_x = i + x;
467  const size_t actual_y = j + y;
468  const size_t flipped_y = height - actual_y - 1;
469  const size_t actual = actual_x + flipped_y * width;
470  buf[x + y * tileWidth] = convert(this->data[actual]);
471  }
472  }
473  logging::check_fatal(TIFFWriteTile(tif, buf, i, j, 0, 0) < 0, "Cannot write tile to %s", filename.c_str());
474  }
475  }
476  GTIFWriteKeys(gtif);
477  if (gtif)
478  {
479  GTIFFree(gtif);
480  }
481  _TIFFfree(buf);
482  TIFFClose(tif);
483  }
