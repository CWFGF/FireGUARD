447  void saveToTiffFile(const string& dir,
448  const string& base_name,
449  std::function<V(T)> convert) const
450  {
451  Idx min_row = 0;
452  // Idx num_rows = this->rows();
453  Idx min_column = 0;
454  // Idx num_columns = this->columns();
455  const double xll = this->xllcorner() + min_column * this->cellSize();
456  // offset is different for y since it's flipped
457  const double yll = this->yllcorner() + (min_row) * this->cellSize();
458  logging::extensive("Lower left corner is (%f, %f)", xll, yll);
459  constexpr uint32_t tileWidth = 256;
460  constexpr uint32_t tileHeight = 256;
461  // ensure this is always divisible by tile size
462  static_assert(0 == MAX_ROWS % tileWidth);
463  static_assert(0 == MAX_COLUMNS % tileHeight);
464  uint32_t width = this->columns();
465  uint32_t height = this->rows();
466  string filename = dir + base_name + ".tif";
467  TIFF* tif = GeoTiffOpen(filename.c_str(), "w");
468  auto gtif = GTIFNew(tif);
469  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
470  const double xul = this->xllcorner();
471  const double yul = this->yllcorner() + (this->cellSize() * this->rows());
472  double tiePoints[6] = {
473  0.0,
474  0.0,
475  0.0,
476  xul,
477  yul,
478  0.0};
479  double pixelScale[3] = {
480  this->cellSize(),
481  this->cellSize(),
482  0.0};
483  uint32_t bps = std::numeric_limits<V>::digits + (1 * std::numeric_limits<V>::is_signed);
484  TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, width);
485  TIFFSetField(tif, TIFFTAG_IMAGELENGTH, height);
486  TIFFSetField(tif, TIFFTAG_SAMPLESPERPIXEL, 1);
487  TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, bps);
488  TIFFSetField(tif, TIFFTAG_TILEWIDTH, tileWidth);
489  TIFFSetField(tif, TIFFTAG_TILELENGTH, tileHeight);
490  TIFFSetField(tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
491  TIFFSetField(tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
492  TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_LZW);
493  GTIFSetFromProj4(gtif, this->proj4().c_str());
494  TIFFSetField(tif, TIFFTAG_GEOTIEPOINTS, 6, tiePoints);
495  TIFFSetField(tif, TIFFTAG_GEOPIXELSCALE, 3, pixelScale);
496  const auto tile_size = TIFFTileSize(tif);
497  logging::debug("Tile size for writing %s is %ld", base_name.c_str(), tile_size);
498  auto buf = (V*)_TIFFmalloc(tile_size);
499  for (size_t i = 0; i < width; i += tileWidth)
500  {
501  for (size_t j = 0; j < height; j += tileHeight)
502  {
503  // need to put data from grid into buffer, but flipped vertically
504  for (size_t x = 0; x < tileWidth; ++x)
505  {
506  for (size_t y = 0; y < tileHeight; ++y)
507  {
508  const size_t actual_x = i + x;
509  const size_t actual_y = j + y;
510  const size_t flipped_y = height - actual_y - 1;
511  const size_t actual = actual_x + flipped_y * width;
512  buf[x + y * tileWidth] = convert(this->data[actual]);
513  }
514  }
515  logging::check_fatal(TIFFWriteTile(tif, buf, i, j, 0, 0) < 0, "Cannot write tile to %s", filename.c_str());
516  }
517  }
518  GTIFWriteKeys(gtif);
519  if (gtif)
520  {
521  GTIFFree(gtif);
522  }
523  _TIFFfree(buf);
524  TIFFClose(tif);
525  }
