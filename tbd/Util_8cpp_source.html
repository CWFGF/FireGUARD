162 void tbd::month_and_day(const int year, const size_t day_of_year, size_t* month, size_t* day_of_month)
163 {
164  auto days = (is_leap_year(year) ? DAYS_IN_MONTH_LEAP : DAYS_IN_MONTH);
165  *month = 1;
166  int days_left = day_of_year;
167  while (days[*month - 1] <= days_left)
168  {
169  days_left -= days[*month - 1];
170  ++(*month);
171  }
172  *day_of_month = days_left + 1;
173 }

186 string tbd::make_timestamp(const int year, const double time)
187 {
188  size_t day = floor(time);
189  size_t hour = (time - day) * static_cast<double>(DAY_HOURS);
190  size_t minute = round(((time - day) * static_cast<double>(DAY_HOURS) - hour) * HOUR_MINUTES);
191  if (60 == minute)
192  {
193  minute = 0;
194  hour += 1;
195  }
196  if (24 == hour)
197  {
198  day += 1;
199  hour = 0;
200  }
201  size_t month;
202  size_t day_of_month;
203  month_and_day(year, day, &month, &day_of_month);
204  char buffer[128];
205  sprintf(buffer,
206  "%4d-%02ld-%02ld %02ld:%02ld",
207  year,
208  month,
209  day_of_month,
210  hour,
211  minute);
212  return {buffer};
213 }
