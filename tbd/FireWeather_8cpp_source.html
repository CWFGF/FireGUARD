31 static unique_ptr<SurvivalMap> make_survival(
32  const set<const fuel::FuelType*>& used_fuels,
33  const Day min_date,
34  const Day max_date,
35  const vector<const FwiWeather*>& weather_by_hour_by_day)
36 {
37  auto result = make_unique<SurvivalMap>();
38  for (const auto& in_fuel : used_fuels)
39  {
40  if (nullptr != in_fuel && 0 != strcmp("Invalid", fuel::FuelType::safeName(in_fuel)))
41  {
42  // initialize with proper size
43  const auto code = fuel::FuelType::safeCode(in_fuel);
44  auto by_fuel = vector<float>{};
45  by_fuel.resize((static_cast<size_t>(max_date) - min_date + 2) * DAY_HOURS);
46  // calculate the entire stream for this fuel
47  for (auto day = min_date; day <= max_date; ++day)
48  {
49  for (auto h = 0; h < DAY_HOURS; ++h)
50  {
51  const auto wx = weather_by_hour_by_day.at(util::time_index(day, h, min_date));
52  const auto i = util::time_index(day, h, min_date);
53  by_fuel.at(i) = static_cast<float>(nullptr != wx
54  ? in_fuel->survivalProbability(*wx)
55  : 0.0);
56  }
57  }
58  result->at(code) = std::move(by_fuel);
59  }
60  }
61  return result;
62 }
