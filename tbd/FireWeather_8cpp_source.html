443 unique_ptr<vector<const FwiWeather*>> make_vector(map<Day, FwiWeather> data)
444 {
445  const auto min_date = data.begin()->first;
446  const auto max_date = data.rbegin()->first;
447  auto r = make_unique<vector<const FwiWeather*>>((max_date - min_date + 2) * DAY_HOURS);
448  // HACK: just approximate last day
449  for (const auto& kv : data)
450  {
451  const auto day = kv.first;
452  const auto& wx = data.at(day);
453  const auto x = wx.mcFfmcPct();
454  const auto x_sq = x * x;
455  const auto x_cu = x * x * x;
456  const auto rt_x = sqrt(x);
457  const auto ln_x = log(x);
458  const auto exp_x = exp(x);
459  const auto exp_neg_x = exp(-x);
460  const auto add_wx =
461  [&r, &day, &wx, &min_date](const int hour, const Ffmc& ffmc)
462  {
463  r->at(util::time_index(day, hour, min_date)) = make_wx(wx, ffmc, hour);
464  };
465  add_wx(12, ffmc_1200(x, x_sq, x_cu, rt_x, exp_neg_x));
466  add_wx(13, ffmc_1300(x, x_sq, x_cu, rt_x, ln_x));
467  add_wx(14, ffmc_1400(x, x_sq, rt_x, ln_x, exp_x));
468  add_wx(15, ffmc_1500(x, x_sq, x_cu, rt_x, ln_x));
469  add_wx(16, wx.ffmc());
470  add_wx(17, ffmc_1700(x, x_sq, rt_x, ln_x, exp_neg_x));
471  add_wx(18, ffmc_1800(x, x_sq, x_cu, rt_x, ln_x));
472  add_wx(19, ffmc_1900(x, x_sq, rt_x, exp_x, exp_neg_x));
473  add_wx(20, ffmc_2000(x, x_sq, x_cu, rt_x, ln_x, exp_neg_x));
474  }
475  // just use high curve for last day
476  const auto& wx_last = data.at(max_date);
477  const auto x_last = wx_last.mcFfmcPct();
478  const auto add_last =
479  [&r, &max_date, &min_date, &wx_last](const int hour, const Ffmc& ffmc)
480  {
481  r->at(util::time_index(max_date + 1, hour, min_date)) = make_wx(wx_last, ffmc, hour);
482  };
483  add_last(6, ffmc_0600_high(x_last));
484  add_last(7, ffmc_0700_high(x_last));
485  add_last(8, ffmc_0800_high(x_last));
486  add_last(9, ffmc_0900_high(x_last));
487  add_last(10, ffmc_1000_high(x_last));
488  add_last(11, ffmc_1100_high(log(x_last), pow(log(x_last), 2)));
489  // need to look at 1200 for tomorrow to figure out if this matches for today
490  for (auto day = static_cast<Day>(max_date - 1); day >= min_date; --day)
491  {
492  const auto& wx = data.at(day);
493  // make sure we use tomorrow for the wind after midnight
494  const auto& wx_wind = data.at(static_cast<size_t>(day + 1));
495  const auto x = wx.mcFfmcPct();
496  const auto ln_x = log(x);
497  const auto ln_x_sq = ln_x * ln_x;
498  const auto& at_1200 = r->at(util::time_index(day + 1, 12, min_date))->ffmc();
499  // figure out which is the closest match and use that curve
500  const auto at_1100_high = ffmc_1100_high(ln_x, ln_x_sq);
501  const auto at_1100_med = ffmc_1100_med(x);
502  const auto at_1100_low = ffmc_1100_low(ln_x, ln_x_sq);
503  const auto for1200 = at_1200.asDouble();
504  const auto for1100_high = at_1100_high.asDouble();
505  const auto for1100_med = at_1100_med.asDouble();
506  const auto for1100_low = at_1100_low.asDouble();
507  const auto diff_high = abs(for1200 - for1100_high);
508  const auto diff_med = abs(for1200 - for1100_med);
509  const auto diff_low = abs(for1200 - for1100_low);
510  const auto add_wx =
511  [&r, &day, &wx_wind, &wx, &min_date](const int hour, const Ffmc& ffmc)
512  {
513  r->at(util::time_index(day + 1, hour, min_date)) = make_wx(wx_wind, wx, ffmc, hour);
514  };
515  // don't want to have 1100 be higher than 1200 but maybe that can happen
516  if (for1200 >= for1100_low && diff_low <= diff_med && diff_low <= diff_high)
517  {
518  // note("low RH");
519  add_wx(6, ffmc_0600_low(x));
520  add_wx(7, ffmc_0700_low(x));
521  add_wx(8, ffmc_0800_low(x));
522  add_wx(9, ffmc_0900_low(x));
523  add_wx(10, ffmc_1000_low(x));
524  add_wx(11, at_1100_low);
525  }
526  else if (for1200 >= for1100_med && diff_med <= diff_high && diff_med <= diff_low)
527  {
528  // note("med RH");
529  add_wx(6, ffmc_0600_med(x));
530  add_wx(7, ffmc_0700_med(x));
531  add_wx(8, ffmc_0800_med(x));
532  add_wx(9, ffmc_0900_med(x));
533  add_wx(10, ffmc_1000_med(x));
534  add_wx(11, at_1100_med);
535  }
536  else
537  {
538  // note("high RH");
539  add_wx(6, ffmc_0600_high(x));
540  add_wx(7, ffmc_0700_high(x));
541  add_wx(8, ffmc_0800_high(x));

620 static unique_ptr<SurvivalMap> make_survival(
621  const set<const fuel::FuelType*>& used_fuels,
622  const Day min_date,
623  const Day max_date,
624  const vector<const FwiWeather*>& weather_by_hour_by_day)
625 {
626  auto result = make_unique<SurvivalMap>();
627  for (const auto& in_fuel : used_fuels)
628  {
629  if (nullptr != in_fuel && 0 != strcmp("Invalid", fuel::FuelType::safeName(in_fuel)))
630  {
631  // initialize with proper size
632  const auto code = fuel::FuelType::safeCode(in_fuel);
633  auto by_fuel = vector<float>{};
634  by_fuel.resize((static_cast<size_t>(max_date) - min_date + 2) * DAY_HOURS);
635  // calculate the entire stream for this fuel
636  for (auto day = min_date; day <= max_date; ++day)
637  {
638  for (auto h = 0; h < DAY_HOURS; ++h)
639  {
640  const auto wx = weather_by_hour_by_day.at(util::time_index(day, h, min_date));
641  const auto i = util::time_index(day, h, min_date);
642  by_fuel.at(i) = static_cast<float>(nullptr != wx
643  ? in_fuel->survivalProbability(*wx)
644  : 0.0);
645  }
646  }
647  result->at(code) = std::move(by_fuel);
648  }
649  }
650  return result;
651 }
