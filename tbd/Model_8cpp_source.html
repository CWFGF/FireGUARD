90 void Model::readWeather(const string& filename,
91  const wx::FwiWeather& yesterday,
92  const double latitude)
93 {
94  map<size_t, map<Day, wx::FwiWeather>> wx{};
95  map<Day, struct tm> dates{};
96  auto min_date = numeric_limits<Day>::max();
97  ifstream in;
98  in.open(filename);
99  logging::check_fatal(!in.is_open(),
100  "Could not open input weather file %s",
101  filename.c_str());
102  if (in.is_open())
103  {
104  string str;
105  logging::info("Reading scenarios from '%s'", filename.c_str());
106  // read header line
107  getline(in, str);
108  // get rid of whitespace
109  str.erase(std::remove(str.begin(), str.end(), ' '), str.end());
110  str.erase(std::remove(str.begin(), str.end(), '\n'), str.end());
111  str.erase(std::remove(str.begin(), str.end(), '\r'), str.end());
112  constexpr auto expected_header =
113  "Scenario,Date,APCP,TMP,RH,WS,WD";
114  logging::check_fatal(expected_header != str,
115  "Input CSV must have columns in this order:\n'%s'\n but got:\n'%s'",
116  expected_header,
117  str.c_str());
118  auto prev = yesterday;
119  while (getline(in, str))
120  {
121  istringstream iss(str);
122  if (getline(iss, str, ',') && !str.empty())
123  {
124  // HACK: ignore date and just worry about relative order??
125  // Scenario
126  logging::verbose("Scenario is %s", str.c_str());
127  auto cur = 0;
128  try
129  {
130  cur = static_cast<size_t>(-stoi(str));
131  }
132  catch (std::exception&)
133  {
134  // HACK: somehow stoi() is still getting empty strings
135  logging::fatal("Error reading weather file %s: %s is not a valid integer", filename.c_str(), str.c_str());
136  }
137  if (wx.find(cur) == wx.end())
138  {
139  logging::debug("Loading scenario %d...", cur);
140  wx.emplace(cur, map<Day, wx::FwiWeather>());
141  prev = yesterday;
142  }
143  auto& s = wx.at(cur);
144  struct tm t
145  {
146  };
147  util::read_date(&iss, &str, &t);
148  year_ = t.tm_year + 1900;
149  const auto ticks = mktime(&t);
150  if (1 == cur)
151  {
152  logging::debug("Date '%s' is %ld and calculated jd is %d",
153  str.c_str(),
154  ticks,
155  t.tm_yday);
156  if (!s.empty() && t.tm_yday < min_date)
157  {
158  logging::fatal(
159  "Weather input file crosses year boundary or dates are not sequential");
160  }
161  }
162  min_date = min(min_date, static_cast<Day>(t.tm_yday));
163  logging::check_fatal(s.find(static_cast<Day>(t.tm_yday)) != s.end(),
164  "Day already exists");
165  const auto month = t.tm_mon + 1;
166  s.emplace(static_cast<Day>(t.tm_yday),
167  wx::FwiWeather(&iss,
168  &str,
169  prev,
170  month,
171  latitude));
172  prev = s.at(static_cast<Day>(t.tm_yday));
173  if (s.find(static_cast<Day>(t.tm_yday)) == s.end())
174  {
175  dates.emplace(static_cast<Day>(t.tm_yday), t);
176  }
177  }
178  }
179  in.close();
180  }
181  for (auto& kv : wx)
182  {
183  kv.second.emplace(static_cast<Day>(min_date - 1), yesterday);
184  }
185  const auto file_out = string(Settings::outputDirectory()) + "/wx_out.csv";
186  FILE* out = fopen(file_out.c_str(), "w");
187  logging::check_fatal(nullptr == out, "Cannot open file %s for output", file_out.c_str());
188  fprintf(out, "Scenario,Day,APCP,TMP,RH,WS,WD,FFMC,DMC,DC,ISI,BUI,FWI\n");
189  size_t i = 1;
190  for (auto& kv : wx)
191  {
192  auto& s = kv.second;
193  for (auto& kv2 : s)
194  {
195  auto& day = kv2.first;
196  auto& w = kv2.second;
197  fprintf(out,
198  "%ld,%d,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g,%1.6g\n",
199  i,
200  day,
201  w.apcp().asDouble(),
202  w.tmp().asDouble(),
203  w.rh().asDouble(),
204  w.wind().speed().asDouble(),
205  w.wind().direction().asDouble(),
206  w.ffmc().asDouble(),
207  w.dmc().asDouble(),
208  w.dc().asDouble(),
209  w.isi().asDouble(),
210  w.bui().asDouble(),
211  w.fwi().asDouble());
212  }
213  ++i;
214  }
215  logging::check_fatal(0 != fclose(out), "Could not close file %s", file_out.c_str());
216  const auto fuel_lookup = sim::Settings::fuelLookup();
217  // loop through and try to find duplicates
218  for (const auto& kv : wx)
219  {
220  const auto k = kv.first;
221  const auto s = kv.second;
222  if (wx_.find(k) == wx_.end())
223  {
224  const auto w = make_shared<wx::FireWeather>(fuel_lookup.usedFuels(), s);
225  wx_.emplace(k, w);
226  }
227  }

229 void Model::findStarts(const Location location)
230 {
231  logging::error("Trying to start a fire in non-fuel");
232  Idx range = 1;
233  // HACK: should always be centered in the grid
234  while (starts_.empty() && (range < (MAX_COLUMNS / 2)))
235  {
236  for (Idx x = -range; x <= range; ++x)
237  {
238  for (Idx y = -range; y <= range; ++y)
239  {
240  // make sure we only look at the outside of the box
241  if (1 == range || abs(x) == range || abs(y) == range)
242  {
243  // const auto loc = env_->cell(location.hash() + (y * MAX_COLUMNS) + x);
244  const auto loc = env_->cell(Location(location.row() + y, location.column() + x));
245  if (!fuel::is_null_fuel(loc))
246  {
247  starts_.push_back(make_shared<topo::Cell>(cell(loc)));
248  }
249  }
250  }
251  }
252  ++range;
253  }
254  logging::check_fatal(starts_.empty(), "Fuel grid is empty");
255  logging::info("Using %d start locations:", starts_.size());
256  for (const auto& s : starts_)
257  {
258  logging::info("\t%d, %d", s->row(), s->column());
259  }

318 Iteration Model::readScenarios(const topo::StartPoint& start_point,
319  const double start,
320  const bool save_intensity,
321  const Day start_day,
322  const Day last_date)
323 {
324  vector<Scenario*> result{};
325  auto saves = Settings::outputDateOffsets();
326  const auto setup_scenario = [&result, save_intensity, &saves](Scenario* scenario) {
327  if (save_intensity)
328  {
329  scenario->registerObserver(new IntensityObserver(*scenario, ""));
330  scenario->registerObserver(new ArrivalObserver(*scenario));
331  scenario->registerObserver(new SourceObserver(*scenario));
332  }
333  // FIX: this should be relative to the start date, not the weather start date
334  for (const auto& i : saves)
335  {
336  scenario->addSaveByOffset(i);
337  }
338  result.push_back(scenario);
339  };
340  for (const auto& kv : wx_)
341  {
342  const auto id = kv.first;
343  const auto cur_wx = kv.second;
344  if (nullptr != perimeter_)
345  {
346  setup_scenario(new Scenario(this,
347  id,
348  cur_wx.get(),
349  start,
350  perimeter_,
351  start_point,
352  start_day,
353  last_date));
354  }
355  else
356  {
357  for (const auto& cur_start : starts_)
358  {
359  // should always have at least the day before the fire in the weather stream
360  setup_scenario(new Scenario(this,
361  id,
362  cur_wx.get(),
363  start,
364  cur_start,
365  start_point,
366  start_day,
367  last_date));
368  }
369  }
370  }
371  return Iteration(result);
372 }

429 bool add_statistics(const size_t i,
430  vector<double>* means,
431  vector<double>* pct,
432  const Model& model,
433  const util::SafeVector& v)
434 {
435  const auto sizes = v.getValues();
436  logging::check_fatal(sizes.empty(), "No sizes at end of simulation");
437  const util::Statistics s{sizes};
438  static_cast<void>(util::insert_sorted(pct, s.percentile(95)));
439  static_cast<void>(util::insert_sorted(means, s.mean()));
440  if (model.isOutOfTime())
441  {
442  logging::note(
443  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
444  i,
445  Settings::maximumTimeSeconds());
446  return false;
447  }
448  return true;

461 size_t runs_required(const size_t i,
462  const vector<double>* means,
463  const vector<double>* pct,
464  const Model& model)
465 {
466  if (model.isOutOfTime())
467  {
468  logging::note(
469  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
470  i,
471  Settings::maximumTimeSeconds());
472  return 0;
473  }
474  const auto for_means = util::Statistics{*means};
475  const auto for_pct = util::Statistics{*pct};
476  if (!(!for_means.isConfident(Settings::confidenceLevel())
477  || !for_pct.isConfident(Settings::confidenceLevel())))
478  {
479  return 0;
480  }
481  const auto left = max(for_means.runsRequired(i, Settings::confidenceLevel()),
482  for_pct.runsRequired(i, Settings::confidenceLevel()));
483  return left;
484 }
485 map<double, ProbabilityMap*> Model::runIterations(const topo::StartPoint& start_point,
486  const double start,
487  const Day start_day,
488  const bool save_intensity)
489 {
490  auto last_date = start_day;
491  for (const auto i : Settings::outputDateOffsets())
492  {
493  last_date = max(static_cast<Day>(start_day + i), last_date);
494  }
495  // use independent seeds so that if we remove one threshold it doesn't affect the other
496  std::seed_seq seed_spread{0.0, start, start_point.latitude(), start_point.longitude()};
497  std::seed_seq seed_extinction{1.0, start, start_point.latitude(), start_point.longitude()};
498  mt19937 mt_spread(seed_spread);
499  mt19937 mt_extinction(seed_extinction);
500  vector<double> means{};
501  vector<double> pct{};
502  size_t i = 0;
503  auto iterations = readScenarios(start_point,
504  start,
505  save_intensity,
506  start_day,
507  last_date);
508  // put probability maps into map
509  const auto saves = iterations.savePoints();
510  const auto started = iterations.startTime();
511  auto probabilities = make_prob_map(*this,
512  saves,
513  started,
514  0,
515  Settings::intensityMaxLow(),
516  Settings::intensityMaxModerate(),
517  numeric_limits<int>::max());
518  vector<map<double, ProbabilityMap*>> all_probabilities{};
519  all_probabilities.push_back(make_prob_map(*this,
520  saves,
521  started,
522  0,
523  Settings::intensityMaxLow(),
524  Settings::intensityMaxModerate(),
525  numeric_limits<int>::max()));
526  auto runs_left = 1;
527  // HACK: just do this here so that we know it happened
528  //iterations.reset(&mt_extinction, &mt_spread);
529  if (Settings::runAsync())
530  {
531  vector<Iteration> all_iterations{};
532  all_iterations.push_back(std::move(iterations));
533  auto threads = list<std::thread>{};
534  for (size_t x = 1; x < std::thread::hardware_concurrency() / 4; ++x)
535  {
536  all_iterations.push_back(readScenarios(start_point,
537  start,
538  save_intensity,
539  start_day,
540  last_date));
541  all_probabilities.push_back(make_prob_map(*this,
542  saves,
543  started,
544  0,
545  Settings::intensityMaxLow(),
546  Settings::intensityMaxModerate(),
547  numeric_limits<int>::max()));
548  }
549  size_t cur_iter = 0;
550  for (auto& iter : all_iterations)
551  {
552  iter.reset(&mt_extinction, &mt_spread);
553  auto& scenarios = iter.getScenarios();
554  for (auto s : scenarios)
555  {
556  threads.emplace_back(&Scenario::run,
557  s,
558  &all_probabilities[cur_iter]);
559  }
560  ++cur_iter;
561  }
562  cur_iter = 0;
563  while (runs_left > 0)
564  {
565  // should have completed one iteration, so add it
566  auto& iteration = all_iterations[cur_iter];
567  // so now try to loop through and add iterations as they finish
568  size_t k = 0;
569  while (k < iteration.size())
570  {
571  threads.front().join();
572  threads.pop_front();
573  ++k;
574  }
575  auto final_sizes = iteration.finalSizes();
576  ++i;
577  for (auto& kv : all_probabilities[cur_iter])
578  {
579  probabilities[kv.first]->addProbabilities(*kv.second);
580  // clear so we don't double count
581  kv.second->reset();
582  }
583  if (!add_statistics(i, &means, &pct, *this, final_sizes))
584  {
585  // ran out of time
586  for (auto& iter : all_iterations)
587  {
588  iter.cancel();
589  }
590  for (auto& t : threads)
591  {
592  if (t.joinable())
593  {
594  t.join();
595  }
596  }
597  return probabilities;
598  }
599  runs_left = runs_required(i, &means, &pct, *this);
600  logging::note("Need another %d iterations", runs_left);
601  if (runs_left > 0)
602  {
603  iteration.reset(&mt_extinction, &mt_spread);
604  auto& scenarios = iteration.getScenarios();
605  for (auto s : scenarios)
606  {
607  threads.emplace_back(&Scenario::run,
608  s,
609  &all_probabilities[cur_iter]);
610  }
611  ++cur_iter;
612  // loop around to start if required
613  cur_iter %= all_iterations.size();
614  }
615  else
616  {
617  for (auto& iter : all_iterations)
618  {
619  iter.cancel();
620  }
621  for (auto& t : threads)
622  {
623  if (t.joinable())
624  {
625  t.join();
626  }
627  }
628  }
629  }
630  // everything should be done when this section ends
631  }
632  else
633  {
634  logging::note("Running in synchronous mode");
635  while (runs_left > 0)
636  {
637  logging::note("Running iteration %d", i + 1);
638  iterations.reset(&mt_extinction, &mt_spread);
639  for (auto s : iterations.getScenarios())
640  {
641  s->run(&probabilities);
642  }
643  ++i;
644  if (!add_statistics(i, &means, &pct, *this, iterations.finalSizes()))
645  {
646  // ran out of time
647  return probabilities;
648  }
649  runs_left = runs_required(i, &means, &pct, *this);
650  logging::note("Need another %d iterations", runs_left);
651  }
652  }
653  return probabilities;
654 }
655 int Model::runScenarios(const char* const weather_input,
656  const char* const raster_root,
657  const wx::FwiWeather& yesterday,
658  const topo::StartPoint& start_point,
659  const tm& start_time,

663 {

665  start_point,

667  start_time.tm_year);
668  logging::debug("Environment loaded");
669  const auto position = env.findCoordinates(start_point, true);
670 #ifndef NDEBUG

672  std::get<0>(*position) > MAX_ROWS || std::get<1>(*position) > MAX_COLUMNS,
673  "Location loaded outside of grid at position (%d, %d)",

675  std::get<1>(*position));

679  Model model(start_point, &env);
680  auto x = 0.0;
681  auto y = 0.0;
682  const auto zone = lat_lon_to_utm(start_point, &x, &y);
683  logging::note("UTM coordinates are: %d %d %d",
684  zone,
685  static_cast<int>(x),
686  static_cast<int>(y));
687  logging::note("Grid has size (%d, %d)", env.rows(), env.columns());
688  logging::note("Fire start position is cell (%d, %d)",
689  location.row(),
690  location.column());

692  if (model.wx_.empty())
693  {
694  logging::fatal("No weather provided");
695  }
696  const auto w = model.wx_.begin()->second;
697  logging::debug("Have weather from day %d to %d", w->minDate(), w->maxDate());

699  const auto needDays = Settings::maxDateOffset();

701  {
702  logging::fatal("Not enough weather to proceed - have %d days but looking for %d", numDays, needDays);
703  }
704  // want to output internal representation of weather to file
705  model.outputWeather();
706  model.makeStarts(*position, start_point, yesterday, perimeter, size);
707  auto start_hour = ((start_time.tm_hour + (static_cast<double>(start_time.tm_min) / 60))
708  / DAY_HOURS);
709  // HACK: round to 2 digits so that we can keep test output the same
710  start_hour = static_cast<double>(static_cast<int>(start_hour * 100)) / 100;
711  const auto start = start_time.tm_yday + start_hour;
712  const auto start_day = static_cast<Day>(start);
713  auto probabilities =
714  model.runIterations(start_point, start, start_day, save_intensity);
715  logging::note("Ran %d simulations", Scenario::completed());
716  show_probabilities(probabilities);
717  auto final_time = numeric_limits<double>::min();
718  for (const auto by_time : probabilities)
719  {
720  const auto time = by_time.first;
721  final_time = max(final_time, time);
722  const auto prob = by_time.second;
723  prob->saveAll(model, start_time, time, start_day);
724  }
725  for (const auto& kv : probabilities)
726  {
727  delete kv.second;
728  }
729  return 0;
