325 void Model::findStarts(const Location location)
326 {
327  logging::error("Trying to start a fire in non-fuel");
328  Idx range = 1;
329  // HACK: should always be centered in the grid
330  while (starts_.empty() && (range < (MAX_COLUMNS / 2)))
331  {
332  for (Idx x = -range; x <= range; ++x)
333  {
334  for (Idx y = -range; y <= range; ++y)
335  {
336  // make sure we only look at the outside of the box
337  if (1 == range || abs(x) == range || abs(y) == range)
338  {
339  // const auto loc = env_->cell(location.hash() + (y * MAX_COLUMNS) + x);
340  const auto loc = env_->cell(Location(location.row() + y, location.column() + x));
341  if (!fuel::is_null_fuel(loc))
342  {
343  starts_.push_back(make_shared<topo::Cell>(cell(loc)));
344  }
345  }
346  }
347  }
348  ++range;
349  }
350  logging::check_fatal(starts_.empty(), "Fuel grid is empty");
351  logging::info("Using %d start locations:", starts_.size());
352  for (const auto& s : starts_)
353  {
354  logging::info("\t%d, %d", s->row(), s->column());
355  }
