62 void output(const int log_level, const char* format, va_list* args)
63 #ifdef NDEBUG
64 noexcept
65 #endif
66 {
67  if (Log::getLogLevel() > log_level)
68  {
69  return;
70  }
71  try
72  {
73  // NOTE: create string first so that entire line writes
74  // (otherwise threads might mix lines)
75  const string tmp;
76  stringstream iss(tmp);
77 #ifdef NDEBUG
78  const time_t now = time(nullptr);
79  auto buf = localtime(&now);
80  iss << put_time(buf, "[%F %T] ");
81 #endif
82  // try to make output consistent if in debug mode
83  iss << LOG_LABELS[log_level];
84  static char buffer[1024]{0};
85  vsprintf(buffer, format, *args);
86  iss << buffer;
87  {
88  lock_guard<mutex> lock(mutex_);
89  printf("%s\n", iss.str().c_str());
90  if (nullptr != out_)
91  {
92  fprintf(out_, "%s\n", iss.str().c_str());
93  }
94  }
95  }
96  catch (const std::exception& ex)
97  {
98  logging::fatal(ex);
99  std::terminate();
100  }
101 }
