84 static void make_threshold(vector<double>* thresholds,
85  mt19937* mt,
86  const Day start_day,
87  const Day last_date,
88  double (*convert)(double value))
89 {
90  const auto total_weight = Settings::thresholdScenarioWeight() + Settings::thresholdDailyWeight() + Settings::thresholdHourlyWeight();
91  uniform_real_distribution<double> rand(0.0, 1.0);
92  const auto general = rand(*mt);
93  for (size_t i = start_day; i < MAX_DAYS; ++i)
94  {
95  const auto daily = rand(*mt);
96  for (auto h = 0; h < DAY_HOURS; ++h)
97  {
98  // generate no matter what so if we extend the time period the results
99  // for the first days don't change
100  const auto hourly = rand(*mt);
101  // only save if we're going to use it
102  if (i <= last_date)
103  {
104  // subtract from 1.0 because we want weight to make things more likely not less
105  // ensure we stay between 0 and 1
106  thresholds->at((i - start_day) * DAY_HOURS + h) =
107  convert(
108  max(0.0,
109  min(1.0,
110  1.0 - (Settings::thresholdScenarioWeight() * general + Settings::thresholdDailyWeight() * daily + Settings::thresholdHourlyWeight() * hourly) / total_weight)));
111  // thresholds->at((i - start_day) * DAY_HOURS + h) = 0.0;
112  }
113  }
114  }
115 }

218 void Scenario::evaluate(const Event& event)
219 {
220 #ifndef NDEBUG
221  log_check_fatal(event.time() < current_time_,
222  "Expected time to be > %f but got %f",
223  current_time_,
224  event.time());
225 #endif
226  const auto& p = event.cell();
227  switch (event.type())
228  {
229  case Event::FIRE_SPREAD:
230  scheduleFireSpread(event);
231  break;
232  case Event::SAVE:
233  saveObservers(event.time());
234  saveStats(event.time());
235  break;
236  case Event::NEW_FIRE:
237  // HACK: don't do this in constructor because scenario creates this in its constructor
238  points_[p].emplace_back(p.column() + CELL_CENTER, p.row() + CELL_CENTER);
239  if (fuel::is_null_fuel(event.cell()))
240  {
241  log_fatal("Trying to start a fire in non-fuel");
242  }
243  log_verbose("Starting fire at point (%f, %f) in fuel type %s at time %f",
244  p.column() + CELL_CENTER,
245  p.row() + CELL_CENTER,
246  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
247  event.time());
248  if (!survives(event.time(), event.cell(), event.timeAtLocation()))
249  {
250  const auto wx = weather(event.time());
251  log_info("Didn't survive ignition in %s with weather %f, %f",
252  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
253  wx->ffmc(),
254  wx->dmc());
255  // HACK: we still want the fire to have existed, so set the intensity of the origin
256  }
257  // fires start with intensity of 1
258  burn(event, 1);
259  scheduleFireSpread(event);
260  break;
261  case Event::END_SIMULATION:
262  log_verbose("End simulation event reached at %f", event.time());
263  endSimulation();
264  break;
265  default:
266  throw runtime_error("Invalid event type");
267  }
268 }

441 Scenario* Scenario::run(map<double, ProbabilityMap*>* probabilities)
442 {
443 #ifndef NDEBUG
444  log_check_fatal(ran(), "Scenario has already run");
445 #endif
446  log_verbose("Starting");
447  CriticalSection _(Model::task_limiter);
448  unburnable_ = model_->getBurnedVector();
449  probabilities_ = probabilities;
450  for (auto time : save_points_)
451  {
452  // NOTE: these happen in this order because of the way they sort based on type
453  addEvent(Event::makeSave(static_cast<double>(time)));
454  }
455  if (nullptr == perimeter_)
456  {
457  addEvent(Event::makeNewFire(start_time_, cell(*start_cell_)));
458  }
459  else
460  {
461  intensity_->applyPerimeter(*perimeter_);
462  const auto& env = model().environment();
463  for (const auto& location : perimeter_->edge())
464  {
465  // const auto cell = env.cell(location.hash());
466  const auto cell = env.cell(location);
467 #ifndef NDEBUG
468  log_check_fatal(fuel::is_null_fuel(cell), "Null fuel in perimeter");
469 #endif
470  log_verbose("Adding point (%d, %d)",
471  cell.column() + CELL_CENTER,
472  cell.row() + CELL_CENTER);
473  points_[cell].emplace_back(cell.column() + CELL_CENTER, cell.row() + CELL_CENTER);
474  }
475  addEvent(Event::makeFireSpread(start_time_));
476  }
477  // HACK: make a copy of the event so that it still exists after it gets processed
478  // NOTE: sorted so that EventSaveASCII is always just before this
479  // Only run until last time we asked for a save for
480  log_verbose("Creating simulation end event for %f", last_save_);
481  addEvent(Event::makeEnd(last_save_));
482  // mark all original points as burned at start
483  for (auto& kv : points_)
484  {
485  const auto& location = kv.first;
486  // would be burned already if perimeter applied
487  if (canBurn(location))
488  {
489  const auto fake_event = Event::makeFireSpread(
490  start_time_,
491  static_cast<IntensitySize>(1),
492  location);
493  burn(fake_event, static_cast<IntensitySize>(1));
494  }
495  }
496  while (!cancelled_ && !scheduler_.empty())
497  {
498  evaluateNextEvent();
499  }
500  model_->releaseBurnedVector(unburnable_);
501  unburnable_ = nullptr;
502  if (cancelled_)
503  {
504  return nullptr;
505  }
506  ++COMPLETED;
507  // HACK: use + to pull value out of atomic
508 #ifdef NDEBUG
509  log_info("[% d of % d] Completed with final size % 0.1f ha",
510  +COMPLETED,
511  +COUNT,
512  currentFireSize());
513 #else
514  // try to make output consistent if in debug mode
515  log_info("Completed with final size %0.1f ha",
516  currentFireSize());
517 #endif
518  ran_ = true;
519 #ifndef NDEBUG
520  static const size_t BufferSize = 64;
521  char buffer[BufferSize + 1] = {0};
522  sprintf(buffer,
523  "%03zu_%06ld_extinction",
524  id(),
525  simulation());
526  saveProbabilities(Settings::outputDirectory(), string(buffer), extinction_thresholds_);
527  sprintf(buffer,
528  "%03zu_%06ld_spread",
529  id(),
530  simulation());
531  saveProbabilities(Settings::outputDirectory(), string(buffer), spread_thresholds_by_ros_);
532 #endif
533  return this;
534 }

557 CellIndex relativeIndex(const topo::Cell& for_cell, const topo::Cell& from_cell)
558 {
559  const auto r = for_cell.row();
560  const auto r_o = from_cell.row();
561  const auto c = for_cell.column();
562  const auto c_o = from_cell.column();
563  if (r == r_o)
564  {
565  // center row
566  // same cell, so source is 0
567  if (c == c_o)
568  {
569  return DIRECTION_NONE;
570  }
571  if (c < c_o)
572  {
573  //center right
574  return DIRECTION_E;
575  }
576  // else has to be c > c_o
577  // center left
578  return DIRECTION_W;
579  }
580  if (r < r_o)
581  {
582  // came from the row to the north
583  if (c == c_o)
584  {
585  // center top
586  return DIRECTION_N;
587  }
588  if (c < c_o)
589  {
590  // top right
591  return DIRECTION_NE;
592  }
593  // else has to be c > c_o
594  // top left
595  return DIRECTION_NW;
596  }
597  // else r > r_o
598  // came from the row to the south
599  if (c == c_o)
600  {
601  // center bottom
602  return DIRECTION_S;
603  }
604  if (c < c_o)
605  {
606  // bottom right
607  return DIRECTION_SE;
608  }
609  // else has to be c > c_o
610  // bottom left
611  return DIRECTION_SW;

613 void Scenario::scheduleFireSpread(const Event& event)
614 {
615  const auto time = event.time();
616  //note("time is %f", time);
617  current_time_ = time;
618  const auto wx = weather(time);
619  logging::check_fatal(nullptr == wx, "No weather available for time %f", time);
620  // log_note("%d points", points_->size());
621  const auto this_time = util::time_index(time);
622  const auto next_time = static_cast<double>(this_time + 1) / DAY_HOURS;
623  // should be in minutes?
624  const auto max_duration = (next_time - time) * DAY_MINUTES;
625  //note("time is %f, next_time is %f, max_duration is %f",
626  // time,
627  // next_time,
628  // max_duration);
629  const auto max_time = time + max_duration / DAY_MINUTES;
630  if (wx->ffmc().asDouble() < minimumFfmcForSpread(time))
631  {
632  addEvent(Event::makeFireSpread(max_time));
633  log_verbose("Waiting until %f because of FFMC", max_time);
634  return;
635  }
636  if (current_time_index_ != this_time)
637  {
638  current_time_index_ = this_time;
639  // offsets_.clear();
640  // max_intensity_.clear();
641  offsets_ = {};
642  max_intensity_ = {};
643  max_ros_ = 0.0;
644  }
645  auto any_spread = false;
646  for (const auto& kv : points_)
647  {
648  const auto& location = kv.first;
649  // any cell that has the same fuel, slope, and aspect has the same spread
650  const auto key = location.key();
651  const auto seek_spreading = offsets_.find(key);
652  if (seek_spreading == offsets_.end())
653  {
654  // have not calculated spread for this cell yet
655  const SpreadInfo origin(*this, time, location, nd(time), wx);
656  // will be empty if invalid
657  offsets_.emplace(key, origin.offsets());
658  if (!origin.isNotSpreading())
659  {
660  any_spread = true;
661  max_ros_ = max(max_ros_, origin.headRos());
662  max_intensity_[key] = max(max_intensity_[key], origin.maxIntensity());
663  }
664  }
665  else
666  {
667  // already did the lookup so use the result
668  any_spread |= !seek_spreading->second.empty();
669  }
670  }
671  if (!any_spread || max_ros_ < Settings::minimumRos())
672  {
673  log_verbose("Waiting until %f", max_time);
674  addEvent(Event::makeFireSpread(max_time));
675  return;
676  }
677  //note("Max spread is %f, max_ros is %f",
678  // Settings::maximumSpreadDistance() * cellSize(),
679  // max_ros_);
680  const auto duration = ((max_ros_ > 0)
681  ? min(max_duration,
682  Settings::maximumSpreadDistance() * cellSize() / max_ros_)
683  : max_duration);
684  //note("Spreading for %f minutes", duration);
685  map<topo::Cell, CellIndex> sources{};
686  const auto new_time = time + duration / DAY_MINUTES;
687  map<topo::Cell, PointSet> point_map_{};
688  map<topo::Cell, size_t> count{};
689  for (auto& kv : points_)
690  {
691  const auto& location = kv.first;
692  count[location] = kv.second.size();
693  const auto key = location.key();
694  auto& offsets = offsets_.at(key);
695  if (!offsets.empty())
696  {
697  for (auto& o : offsets)
698  {
699  // offsets in meters
700  const auto offset_x = o.x * duration;
701  const auto offset_y = o.y * duration;
702  const Offset offset{offset_x, offset_y};
703  //note("%f, %f", offset_x, offset_y);
704  for (auto& p : kv.second)
705  {
706  const InnerPos pos = p.add(offset);
707  const auto for_cell = cell(pos);
708  const auto source = relativeIndex(for_cell, location);
709  sources[for_cell] |= source;
710  if (!(*unburnable_)[for_cell.hash()])
711  {
712  //log_extensive("Adding point (%f, %f)", pos.x, pos.y);
713  point_map_[for_cell].emplace_back(pos);
714  }
715  }
716  }
717  }
718  else
719  {
720  // can't just keep existing points by swapping because something may have spread into this cell
721  auto& pts = point_map_[location];
722  pts.insert(pts.end(), kv.second.begin(), kv.second.end());
723  }
724  // kv.second.clear();
725  kv.second = {};
726  }
727  vector<topo::Cell> erase_what{};
728  for (auto& kv : point_map_)
729  {
730  auto& for_cell = kv.first;
731  if (!kv.second.empty())
732  {
733  if (canBurn(for_cell) && max_intensity_[for_cell.key()] > 0)
734  {
735  // HACK: make sure it can't round down to 0
736  const auto intensity = static_cast<IntensitySize>(max(
737  1.0,
738  max_intensity_[for_cell.key()]));
739  // HACK: just use the first cell as the source
740  const auto source = sources[for_cell];
741  const auto fake_event = Event::makeFireSpread(
742  new_time,
743  intensity,
744  for_cell,
745  source);
746  burn(fake_event, intensity);
747  }
748  // check if this cell is surrounded by burned cells or non-fuels
749  // if surrounded then just drop all the points inside this cell
750  if (!(*unburnable_)[for_cell.hash()])
751  {
752  // do survival check first since it should be easier
753  if (survives(new_time, for_cell, new_time - arrival_[for_cell]) && !isSurrounded(for_cell))
754  {
755  if (count[for_cell] > 1 && kv.second.size() > 3)
756  {
757  // no point in doing hull if only one point spread
758  // 3 points should just be a triangle usually (could be co-linear, but that's fine
759  hull(kv.second);
760  }
761  std::swap(points_[for_cell], kv.second);
762  }
763  else
764  {
765  // whether it went out or is surrounded just mark it as unburnable
766  (*unburnable_)[for_cell.hash()] = true;
767  erase_what.emplace_back(for_cell);
768  }
769  }
770  // kv.second.clear();
771  kv.second = {};
772  }
773  else
774  {
775  erase_what.emplace_back(for_cell);
776  }
777  }
778  for (auto& c : erase_what)
779  {
780  points_.erase(c);
781  }
782  log_verbose("Spreading %d points until %f", points_.size(), new_time);
783  addEvent(Event::makeFireSpread(new_time));
784 }
