55 double find_meridian(const string& proj4) noexcept
56 {
57  try
58  {
59  int zone{};
60  auto meridian = 0.0;
61  if (find_value("+zone=", proj4, &zone, &str_to_int))
62  {
63  meridian = topo::utm_central_meridian_deg(zone);
64  }
65  else if (!find_value("+lon_0=", proj4, &meridian, &str_to_double))
66  {
67  logging::fatal("Can't find meridian for grid");
68  }
69  return meridian;
70  }
71  catch (...)
72  {
73  return logging::fatal<double>("Unable to parse meridian in string '%s'",
74  proj4.c_str());
75  }
76 }

108 {
109  ofstream out;
110  out.open(dir + base_name + ".prj");
111  logging::extensive(proj4_.c_str());
112  // HACK: use what we know is true for the grids that were generated and
113  // hope it's correct otherwise
114  const auto proj = find_value("+proj=", proj4_);
115  if (!proj.empty() && proj != "tmerc" && proj != "utm")
116  {
117  logging::fatal("Cannot create projection for non-transverse grid");

144 // Use pair instead of Location, so we can go above max columns & rows
145 unique_ptr<FullCoordinates> GridBase::findFullCoordinates(const topo::Point& point,
146  const bool flipped) const
147 {
148  auto x = 0.0;
149  auto y = 0.0;
150  lat_lon_to_utm(point, this->zone(), &x, &y);
151  logging::note("Coordinates (%f, %f) converted to (%0.1f, %f, %f)",
152  point.latitude(),
153  point.longitude(),
154  this->zone(),
155  x,
156  y);
157  logging::debug("Lower left is (%f, %f)", this->xllcorner_, this->yllcorner_);
158  // convert coordinates into cell position
159  const auto actual_x = (x - this->xllcorner_) / this->cell_size_;
160  // these are already flipped across the y-axis on reading, so it's the same as for x now
161  auto actual_y = (!flipped)
162  ? (y - this->yllcorner_) / this->cell_size_
163  : (yurcorner_ - y) / cell_size_;
164  const auto column = static_cast<FullIdx>(actual_x);
165  const auto row = static_cast<FullIdx>(round(actual_y - 0.5));
166  if (0 > column || column >= calculateColumns() || 0 > row || row >= calculateRows())
167  {
168  logging::debug("Returning nullptr from findFullCoordinates() for (%f, %f) => (%d, %d)",
169  actual_x,
170  actual_y,
171  column,
172  row);
173  return nullptr;
174  }
175  const auto sub_x = static_cast<SubSize>((actual_x - column) * 1000);
176  const auto sub_y = static_cast<SubSize>((actual_y - row) * 1000);
177  return make_unique<FullCoordinates>(static_cast<FullIdx>(row),
178  static_cast<FullIdx>(column),
179  sub_x,
180  sub_y);
181 }
