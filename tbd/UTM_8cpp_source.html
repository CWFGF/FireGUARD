43 /* Ellipsoid model constants (actual values here are for WGS84) */
44 constexpr auto SM_A = 6378137.0;
45 constexpr auto SM_B = 6356752.314;
46 constexpr auto UTM_SCALE_FACTOR = 0.9996;
47 double arc_length_of_meridian(const double phi) noexcept
48 {
49  const auto n = (SM_A - SM_B) / (SM_A + SM_B);
50  const auto alpha = ((SM_A + SM_B) / 2.0) * (1.0 + (pow_int<2>(n) / 4.0) + (pow_int<4>(n) / 64.0));
51  const auto beta = (-3.0 * n / 2.0) + (9.0 * pow_int<3>(n) / 16.0) + (-3.0 * pow_int<5>(n) / 32.0);
52  const auto gamma = (15.0 * pow_int<2>(n) / 16.0) + (-15.0 * pow_int<4>(n) / 32.0);
53  const auto delta = (-35.0 * pow_int<3>(n) / 48.0) + (105.0 * pow_int<5>(n) / 256.0);
54  const auto epsilon = (315.0 * pow_int<4>(n) / 512.0);
55  /* Now calculate the sum of the series and return */
56  return alpha * (phi + (beta * _sin(2.0 * phi)) + (gamma * _sin(4.0 * phi)) + (delta * _sin(6.0 * phi)) + (epsilon * _sin(8.0 * phi)));
57 }

66 double footpoint_latitude(const double y) noexcept
67 {
68  /* Precalculate n (Eq. 10.18) */
69  const auto n = (SM_A - SM_B) / (SM_A + SM_B);
70  /* Precalculate alpha_ (Eq. 10.22) */
71  /* (Same as alpha in Eq. 10.17) */
72  const auto alpha = ((SM_A + SM_B) / 2.0) * (1 + (pow_int<2>(n) / 4) + (pow_int<4>(n) / 64));
73  /* Precalculate row_ (Eq. 10.23) */
74  const auto y_alpha = y / alpha;
75  /* Precalculate beta_ (Eq. 10.22) */
76  const auto beta = (3.0 * n / 2.0) + (-27.0 * pow_int<3>(n) / 32.0) + (269.0 * pow_int<5>(n) / 512.0);
77  /* Precalculate gamma_ (Eq. 10.22) */
78  const auto gamma = (21.0 * pow_int<2>(n) / 16.0) + (-55.0 * pow_int<4>(n) / 32.0);
79  /* Precalculate delta_ (Eq. 10.22) */
80  const auto delta = (151.0 * pow_int<3>(n) / 96.0) + (-417.0 * pow_int<5>(n) / 128.0);
81  /* Precalculate epsilon_ (Eq. 10.22) */
82  const auto epsilon = (1097.0 * pow_int<4>(n) / 512.0);
83  /* Now calculate the sum of the series (Eq. 10.21) */
84  return y_alpha + (beta * _sin(2.0 * y_alpha))
85  + (gamma * _sin(4.0 * y_alpha))
86  + (delta * _sin(6.0 * y_alpha))
87  + (epsilon * _sin(8.0 * y_alpha));
88 }
89 void map_lat_lon_to_xy(const double phi,
90  const double lambda,
91  const double lambda0,
92  double* x,
93  double* y) noexcept
94 {
95  const auto ep2 = (pow_int<2>(SM_A) - pow_int<2>(SM_B)) / pow_int<2>(SM_B);
96  const auto c = _cos(phi);
97  const auto nu2 = ep2 * pow_int<2>(c);
98  const auto n = pow_int<2>(SM_A) / (SM_B * sqrt(1 + nu2));
99  const auto t = tan(phi);
100  const auto t2 = t * t;
101  const auto l = lambda - lambda0;
102  /* Precalculate coefficients for l**n in the equations below
103  so a normal human being can read the expressions for easting
104  and northing
105  -- l**1 and l**2 have coefficients of 1.0 */
106  const auto l3_coefficient = 1.0 - t2 + nu2;
107  const auto l4_coefficient = 5.0 - t2 + 9 * nu2 + 4.0 * (nu2 * nu2);
108  const auto l5_coefficient = 5.0 - 18.0 * t2 + (t2 * t2) + 14.0 * nu2 - 58.0 * t2 * nu2;
109  const auto l6_coefficient = 61.0 - 58.0 * t2 + (t2 * t2) + 270.0 * nu2 - 330.0 * t2 * nu2;
110  const auto l7_coefficient = 61.0 - 479.0 * t2 + 179.0 * (t2 * t2) - (t2 * t2 * t2);
111  const auto l8_coefficient = 1385.0 - 3111.0 * t2 + 543.0 * (t2 * t2) - (t2 * t2 * t2);
112  /* Calculate easting (x) */
113  *x = n * c * l
114  + (n / 6.0 * pow_int<3>(c) * l3_coefficient * pow_int<3>(l))
115  + (n / 120.0 * pow_int<5>(c) * l5_coefficient * pow_int<5>(l))
116  + (n / 5040.0 * pow_int<7>(c) * l7_coefficient * pow_int<7>(l));
117  /* Calculate northing (y) */
118  *y = arc_length_of_meridian(phi)
119  + (t / 2.0 * n * pow_int<2>(c) * pow_int<2>(l))
120  + (t / 24.0 * n * pow_int<4>(c) * l4_coefficient * pow_int<4>(l))
121  + (t / 720.0 * n * pow_int<6>(c) * l6_coefficient * pow_int<6>(l))
122  + (t / 40320.0 * n * pow_int<8>(c) * l8_coefficient * pow_int<8>(l));
123 }
124 // The local variables Nf, nuf2, tf, and tf2 serve the same purpose as
125 // N, nu2, t, and t2 in MapLatLonToXY, but they are computed with respect
126 // to the footpoint latitude phi_f.
127 //
128 // x1_fraction, x2_fraction, x2_polynomial, x3_polynomial, etc. are to enhance
129 // readability and to optimize computations.
130 void map_xy_to_lat_lon(const double x,
131  const double y,
132  const double lambda0,
133  double* phi,
134  double* lambda) noexcept
135 {
136  /* Get the value of phi_f, the footpoint latitude. */
137  const auto phi_f = footpoint_latitude(y);
138  const auto ep2 = (pow_int<2>(SM_A) - pow_int<2>(SM_B)) / pow_int<2>(SM_B);
139  const auto cf = _cos(phi_f);
140  const auto nuf2 = ep2 * pow_int<2>(cf);
141  /* Precalculate Nf and initialize nf_power */
142  const auto nf = pow_int<2>(SM_A) / (SM_B * sqrt(1 + nuf2));
143  auto nf_power = nf;
144  const auto tf = tan(phi_f);
145  const auto tf2 = tf * tf;
146  const auto tf4 = tf2 * tf2;
147  /* Precalculate fractional coefficients for x**n in the equations
148  below to simplify the expressions for latitude and longitude. */
149  const auto x1_fraction = 1.0 / (nf_power * cf);
150  nf_power *= nf; /* now equals Nf**2) */
151  const auto x2_fraction = tf / (2.0 * nf_power);
152  nf_power *= nf; /* now equals Nf**3) */
153  const auto x3_fraction = 1.0 / (6.0 * nf_power * cf);
154  nf_power *= nf; /* now equals Nf**4) */
155  const auto x4_fraction = tf / (24.0 * nf_power);
156  nf_power *= nf; /* now equals Nf**5) */
157  const auto x5_fraction = 1.0 / (120.0 * nf_power * cf);
158  nf_power *= nf; /* now equals Nf**6) */
159  const auto x6_fraction = tf / (720.0 * nf_power);
160  nf_power *= nf; /* now equals Nf**7) */
161  const auto x7_fraction = 1.0 / (5040.0 * nf_power * cf);
162  nf_power *= nf; /* now equals Nf**8) */
163  const auto x8_fraction = tf / (40320.0 * nf_power);
164  /* Precalculate polynomial coefficients for x**n.
165  -- x**1 does not have a polynomial coefficient. */
166  const auto x2_polynomial = -1.0 - nuf2;
167  const auto x3_polynomial = -1.0 - 2 * tf2 - nuf2;
168  const auto x4_polynomial = 5.0 + 3.0 * tf2 + 6.0 * nuf2 - 6.0 * tf2 * nuf2 - 3.0 * (nuf2 * nuf2)
169  - 9.0 * tf2 * (nuf2 * nuf2);
170  const auto x5_polynomial = 5.0 + 28.0 * tf2 + 24.0 * tf4 + 6.0 * nuf2 + 8.0 * tf2 * nuf2;
171  const auto x6_polynomial = -61.0 - 90.0 * tf2 - 45.0 * tf4 - 107.0 * nuf2 + 162.0 * tf2 * nuf2;
172  const auto x7_polynomial = -61.0 - 662.0 * tf2 - 1320.0 * tf4 - 720.0 * (tf4 * tf2);
173  const auto x8_polynomial = 1385.0 + 3633.0 * tf2 + 4095.0 * tf4 + 1575 * (tf4 * tf2);
174  /* Calculate latitude */
175  *phi = phi_f + x2_fraction * x2_polynomial * (x * x)
176  + x4_fraction * x4_polynomial * pow_int<4>(x)
177  + x6_fraction * x6_polynomial * pow_int<6>(x)
178  + x8_fraction * x8_polynomial * pow_int<8>(x);
179  /* Calculate longitude */
180  *lambda = lambda0 + x1_fraction * x
181  + x3_fraction * x3_polynomial * pow_int<3>(x)
182  + x5_fraction * x5_polynomial * pow_int<5>(x)
183  + x7_fraction * x7_polynomial * pow_int<7>(x);
184 }
185 int lat_lon_to_utm(const Point& point, double* x, double* y) noexcept
186 {
187  const auto zone = static_cast<int>((point.longitude() + 180.0) / 6) + 1;
188  map_lat_lon_to_xy(to_radians(point.latitude()),
189  to_radians(point.longitude()),
190  utm_central_meridian(zone),
191  x,
192  y);
193  /* Adjust easting and northing for UTM system. */
194  *x = (*x) * UTM_SCALE_FACTOR + 500000.0;
195  *y = (*y) * UTM_SCALE_FACTOR;
196  if (*y < 0.0)
197  *y += 10000000.0;
198  return zone;
199 }

213 void utm_to_lat_lon(double x,
214  double y,
215  const int zone,
216  const bool is_southern_hemisphere,
217  double* lat,
218  double* lon) noexcept
219 {
220  x -= 500000.0;
221  x /= UTM_SCALE_FACTOR;
222  /* If in southern hemisphere, adjust y accordingly. */
223  if (is_southern_hemisphere)
224  y -= 10000000.0;
225  y /= UTM_SCALE_FACTOR;
226  const auto central_meridian = utm_central_meridian(zone);
227  map_xy_to_lat_lon(x, y, central_meridian, lat, lon);
228 }
