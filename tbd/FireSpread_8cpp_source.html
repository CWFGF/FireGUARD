25 SlopeTableArray make_slope_table() noexcept
26 {
27  // HACK: slope can be infinite, but anything > max is the same as max
28  SlopeTableArray result{};
29  for (size_t i = 0; i <= MAX_SLOPE_FOR_FACTOR; ++i)
30  {
31  result.at(i) = exp(3.533 * pow(i / 100.0, 1.2));
32  }
33  static_assert(result.size() == MAX_SLOPE_FOR_DISTANCE + 1);
34  for (size_t i = MAX_SLOPE_FOR_FACTOR + 1; i < result.size(); ++i)
35  {
36  result.at(i) = result.at(MAX_SLOPE_FOR_FACTOR);
37  }
38  return result;
39 }

69 double SpreadInfo::initial(SpreadInfo& spread,
70  const wx::FwiWeather& weather,
71  double& ffmc_effect,
72  double& wsv,
73  bool& is_crown,
74  double& sfc,
75  double& rso,
76  double& raz,
77  const fuel::FuelType* const fuel,
78  bool has_no_slope,
79  double heading_sin,
80  double heading_cos,
81  double bui_eff,
82  double min_ros,
83  double critical_surface_intensity)
84 {
85  ffmc_effect = spread.ffmcEffect();
86  // needs to be non-const so that we can update if slopeEffect changes direction
87  raz = spread.wind().heading();
88  const auto isz = 0.208 * ffmc_effect;
89  wsv = spread.wind().speed().asDouble();
90  if (!has_no_slope)
91  {
92  const auto isf1 = fuel->calculateIsf(spread, isz);
93  // const auto isf = (0.0 == isf1) ? isz : isf1;
94  // we know const auto isz = 0.208 * ffmc_effect;
95  auto wse = 0.0 == isf1 ? 0 : log(isf1 / isz) / 0.05039;
96  if (wse > 40)
97  {
98  wse = 28.0 - log(1.0 - min(0.999 * 2.496 * ffmc_effect, isf1) / (2.496 * ffmc_effect)) / 0.0818;
99  }
100  // we know that at->raz is already set to be the wind heading
101  const auto wsv_x = spread.wind().wsvX() + wse * heading_sin;
102  const auto wsv_y = spread.wind().wsvY() + wse * heading_cos;
103  wsv = sqrt(wsv_x * wsv_x + wsv_y * wsv_y);
104  raz = (0 == wsv) ? 0 : acos(wsv_y / wsv);
105  if (wsv_x < 0)
106  {
107  raz = util::RAD_360 - raz;
108  }
109  }
110  const auto isi = isz * STANDARD_WSV(wsv);
111  // FIX: make this a member function so we don't need to preface head_ros_
112  spread.head_ros_ = fuel->calculateRos(spread.nd(),
113  weather,
114  isi)
115  * bui_eff;
116  if (min_ros > spread.head_ros_)
117  {
118  spread.head_ros_ = -1;
119  }
120  else
121  {
122  sfc = fuel->surfaceFuelConsumption(spread);
123  rso = fuel::FuelType::criticalRos(sfc, critical_surface_intensity);
124  is_crown = fuel::FuelType::isCrown(critical_surface_intensity,
125  fuel::fire_intensity(sfc, spread.head_ros_));
126  if (is_crown)
127  {
128  spread.head_ros_ = fuel->finalRos(spread,
129  isi,
130  fuel->crownFractionBurned(spread.head_ros_, rso),
131  spread.head_ros_);
132  }
133  }
134  return spread.head_ros_;
135 }
136 SpreadInfo::SpreadInfo(const Scenario& scenario,
137  const double time,
138  const topo::Cell& cell,
139  const int nd,
140  const wx::FwiWeather* weather,
141  const wx::FwiWeather* weather_daily)
142  : cell_(cell),
143  weather_(weather),
144  time_(time),
145  nd_(nd)
146 {
147  // HACK: use weather_daily to figure out probability of spread but hourly for ROS
148  max_intensity_ = -1;
149  const auto slope_azimuth = cell_.aspect();
150  const auto fuel = fuel::check_fuel(cell);
151  const auto has_no_slope = 0 == percentSlope();
152  double heading_sin = 0;
153  double heading_cos = 0;
154  if (!has_no_slope)
155  {
156  const auto heading = util::to_heading(
157  util::to_radians(static_cast<double>(slope_azimuth)));
158  heading_sin = _sin(heading);
159  heading_cos = _cos(heading);
160  }
161  // HACK: only use BUI from hourly weather for both calculations
162  const auto bui_eff = fuel->buiEffect(bui().asDouble());
163  const auto min_ros = std::max(scenario.spreadThresholdByRos(time_),
164  Settings::minimumRos());
165  // FIX: gets calculated when not necessary sometimes
166  const auto critical_surface_intensity = fuel->criticalSurfaceIntensity(*this);
167  double ffmc_effect;
168  double wsv;
169  bool is_crown;
170  double sfc;
171  double rso;
172  double raz;
173  if (min_ros > SpreadInfo::initial(
174  *this,
175  *weather_daily,
176  ffmc_effect,
177  wsv,
178  is_crown,
179  sfc,
180  rso,
181  raz,
182  fuel,
183  has_no_slope,
184  heading_sin,
185  heading_cos,
186  bui_eff,
187  min_ros,
188  critical_surface_intensity)
189  || sfc < COMPARE_LIMIT)
190  {
191  return;
192  }
193  // Now use hourly weather for actual spread calculations
194  if (min_ros > SpreadInfo::initial(*this,
195  *weather,
196  ffmc_effect,
197  wsv,
198  is_crown,
199  sfc,
200  rso,
201  raz,
202  fuel,
203  has_no_slope,
204  heading_sin,
205  heading_cos,
206  bui_eff,
207  min_ros,
208  critical_surface_intensity)
209  || sfc < COMPARE_LIMIT)
210  {
211  // no spread with hourly weather
212  // NOTE: only would happen if FFMC hourly is lower than FFMC daily?
213  return;
214  }
215  const auto back_isi = ffmc_effect * STANDARD_BACK_ISI_WSV(wsv);
216  auto back_ros = fuel->calculateRos(nd,
217  *weather,
218  back_isi)
219  * bui_eff;
220  if (is_crown)
221  {
222  back_ros = fuel->finalRos(*this,
223  back_isi,
224  fuel->crownFractionBurned(back_ros, rso),
225  back_ros);
226  }
227  // do everything we can to avoid calling trig functions unnecessarily
228  const auto b_semi = has_no_slope ? 0 : _cos(atan(percentSlope() / 100.0));
229  const auto slope_radians = util::to_radians(slope_azimuth);
230  // do check once and make function just return 1.0 if no slope
231  const auto no_correction = [](const double) noexcept { return 1.0; };
232  const auto do_correction = [b_semi, slope_radians](const double theta) noexcept {
233  // never gets called if isInvalid() so don't check
234  // figure out how far the ground distance is in map distance horizontally
235  auto angle_unrotated = theta - slope_radians;
236  if (util::to_degrees(angle_unrotated) == 270 || util::to_degrees(angle_unrotated) == 90)
237  {
238  // CHECK: if we're going directly across the slope then horizontal distance is same as spread distance
239  return 1.0;
240  }
241  const auto tan_u = tan(angle_unrotated);
242  const auto y = b_semi / sqrt(b_semi * tan_u * (b_semi * tan_u) + 1.0);
243  const auto x = y * tan_u;
244  // CHECK: Pretty sure you can't spread farther horizontally than the spread distance, regardless of angle?
245  return min(1.0, sqrt(x * x + y * y));
246  };
247  const auto correction_factor = has_no_slope
248  ? std::function<double(double)>(no_correction)
249  : std::function<double(double)>(do_correction);
250  const auto cell_size = scenario.cellSize();
251  const auto add_offset = [this, cell_size, min_ros](const double direction,
252  const double ros) {
253  if (ros < min_ros)
254  {
255  return false;
256  }
257  // spreading, so figure out offset from current point
258  const auto ros_cell = ros / cell_size;
259  offsets_.emplace_back(ros_cell * _sin(direction), ros_cell * _cos(direction));
260  return true;
261  };
262  // if not over spread threshold then don't spread
263  // HACK: assume there is no fuel where a crown fire's sfc is < COMPARE_LIMIT and its fc is >
264  double ros{};
265  // HACK: set ros in boolean if we get that far so that we don't have to repeat the if body
266  if (!add_offset(raz, ros = (head_ros_ * correction_factor(raz))))
267  {
268  // mark as invalid
269  head_ros_ = -1;
270  return;
271  }
272  auto fc = sfc;
273  // don't need to re-evaluate if crown with new head_ros_ because it would only go up if is_crown
274  if (fuel->canCrown() && is_crown)
275  {
276  // wouldn't be crowning if ros is 0 so that's why this is in an else
277  fc += fuel->crownConsumption(fuel->crownFractionBurned(head_ros_, rso));
278  }
279  // max intensity should always be at the head
280  max_intensity_ = fuel::fire_intensity(fc, ros);
281  const auto a = (head_ros_ + back_ros) / 2.0;
282  const auto c = a - back_ros;
283  const auto flank_ros = a / fuel->lengthToBreadth(wsv);
284  const auto a_sq = a * a;
285  const auto flank_ros_sq = flank_ros * flank_ros;
286  const auto a_sq_sub_c_sq = a_sq - (c * c);
287  const auto ac = a * c;
288  const auto calculate_ros =
289  [a, c, ac, flank_ros, a_sq, flank_ros_sq, a_sq_sub_c_sq](const double theta) noexcept {
290  const auto cos_t = _cos(theta);
291  const auto cos_t_sq = cos_t * cos_t;
292  const auto f_sq_cos_t_sq = flank_ros_sq * cos_t_sq;
293  // 1.0 = cos^2 + sin^2
294  // const auto sin_t_sq = 1.0 - cos_t_sq;
295  const auto sin_t = _sin(theta);
296  const auto sin_t_sq = sin_t * sin_t;
297  return abs((a * ((flank_ros * cos_t * sqrt(f_sq_cos_t_sq + a_sq_sub_c_sq * sin_t_sq) - ac * sin_t_sq) / (f_sq_cos_t_sq + a_sq * sin_t_sq)) + c) / cos_t);
298  };
299  const auto add_offsets =
300  [&correction_factor, &add_offset, raz, min_ros](
301  const double angle_radians,
302  const double ros_flat) {
303  if (ros_flat < min_ros)
304  {
305  return false;
306  }
307  auto direction = util::fix_radians(angle_radians + raz);
308  // spread is symmetrical across the center axis, but needs to be adjusted if on a slope
309  // intentionally don't use || because we want both of these to happen all the time
310  auto added = add_offset(direction, ros_flat * correction_factor(direction));
311  direction = util::fix_radians(raz - angle_radians);
312  added |= add_offset(direction, ros_flat * correction_factor(direction));
313  return added;
314  };
315  const auto add_offsets_calc_ros =
316  [&add_offsets, &calculate_ros](const double angle_radians) { return add_offsets(angle_radians, calculate_ros(angle_radians)); };
317  bool added = add_offset(raz, head_ros_);
318  constexpr size_t STEP = 10;
319  size_t i = STEP;
320  while (added && i < 90)
321  {
322  added = add_offsets_calc_ros(util::to_radians(i));
323  i += STEP;
324  }
325  if (added)
326  {
327  added = add_offsets(util::to_radians(90), flank_ros * sqrt(a_sq_sub_c_sq) / a);
328  i = 90 + STEP;
329  while (added && i < 180)
330  {
331  added = add_offsets_calc_ros(util::to_radians(i));
332  i += STEP;
333  }
334  if (added)
335  {
336  //only use back ros if every other angle is spreading since this should be lowest
337  // 180
338  if (back_ros < min_ros)
339  {
340  return;
341  }
342  const auto direction = util::fix_radians(util::RAD_180 + raz);
343  static_cast<void>(!add_offset(direction, back_ros * correction_factor(direction)));
344  }
345  }
346 }
