25 SlopeTableArray make_slope_table() noexcept
26 {
27  // HACK: slope can be infinite, but anything > max is the same as max
28  SlopeTableArray result{};
29  for (size_t i = 0; i <= MAX_SLOPE_FOR_FACTOR; ++i)
30  {
31  result.at(i) = exp(3.533 * pow(i / 100.0, 1.2));
32  }
33  static_assert(result.size() == MAX_SLOPE_FOR_DISTANCE + 1);
34  for (size_t i = MAX_SLOPE_FOR_FACTOR + 1; i < result.size(); ++i)
35  {
36  result.at(i) = result.at(MAX_SLOPE_FOR_FACTOR);
37  }
38  return result;
39 }

50 {

54 static double calculate_standard_wsv(const double v) noexcept
55 {
56  return v < 40.0
57  ? exp(0.05039 * v)
58  : 12.0 * (1.0 - exp(-0.0818 * (v - 28)));
59 }

61 SpreadInfo::SpreadInfo(const Scenario& scenario,
62  const double time,
63  const topo::Cell& cell,
64  const int nd,
65  const wx::FwiWeather* weather)
66  : cell_(cell),
67  weather_(weather),
68  time_(time),
69  nd_(nd)
70 {
71  max_intensity_ = -1;
72  const auto ffmc_effect = ffmcEffect();
73  // needs to be non-const so that we can update if slopeEffect changes direction
74  auto raz = wind().heading();
75  const auto isz = 0.208 * ffmc_effect;
76  const auto slope_azimuth = cell_.aspect();
77  const auto fuel = fuel::check_fuel(cell);
78  const auto has_no_slope = 0 == percentSlope();
79  auto wsv = wind().speed().asDouble();
80  if (!has_no_slope)
81  {
82  const auto isf1 = fuel->calculateIsf(*this, isz);
83  // const auto isf = (0.0 == isf1) ? isz : isf1;
84  // we know const auto isz = 0.208 * ffmc_effect;
85  auto wse = 0.0 == isf1 ? 0 : log(isf1 / isz) / 0.05039;
86  if (wse > 40)
87  {
88  wse = 28.0 - log(1.0 - min(0.999 * 2.496 * ffmc_effect, isf1) / (2.496 * ffmc_effect)) / 0.0818;
89  }
90  const auto heading = util::to_heading(
91  util::to_radians(static_cast<double>(slope_azimuth)));
92  // we know that at->raz is already set to be the wind heading
93  const auto wsv_x = wind().wsvX() + wse * _sin(heading);
94  const auto wsv_y = wind().wsvY() + wse * _cos(heading);
95  wsv = sqrt(wsv_x * wsv_x + wsv_y * wsv_y);
96  raz = (0 == wsv) ? 0 : acos(wsv_y / wsv);
97  if (wsv_x < 0)
98  {
99  raz = util::RAD_360 - raz;
100  }
101  }
102  const auto isi = isz * STANDARD_WSV(wsv);
103  const auto bui_eff = fuel->buiEffect(bui().asDouble());
104  head_ros_ = fuel->calculateRos(nd,
105  *weather,
106  isi)
107  * bui_eff;
108  const auto min_ros = Settings::minimumRos();
109  if (min_ros > head_ros_)
110  {
111  head_ros_ = -1;
112  return;
113  }
114  const auto sfc = fuel->surfaceFuelConsumption(*this);
115  const auto critical_surface_intensity = fuel->criticalSurfaceIntensity(*this);
116  const auto rso = fuel::FuelType::criticalRos(sfc, critical_surface_intensity);
117  const auto is_crown = fuel::FuelType::isCrown(critical_surface_intensity,
118  fuel::fire_intensity(sfc, head_ros_));
119  const auto back_isi = ffmc_effect * STANDARD_BACK_ISI_WSV(wsv);
120  auto back_ros = fuel->calculateRos(nd,
121  *weather,
122  back_isi)
123  * bui_eff;
124  if (is_crown)
125  {
126  head_ros_ = fuel->finalRos(*this,
127  isi,
128  fuel->crownFractionBurned(head_ros_, rso),
129  head_ros_);
130  back_ros = fuel->finalRos(*this,
131  back_isi,
132  fuel->crownFractionBurned(back_ros, rso),
133  back_ros);
134  }
135  // do everything we can to avoid calling trig functions unnecessarily
136  const auto b_semi = has_no_slope ? 0 : _cos(atan(percentSlope() / 100.0));
137  const auto slope_radians = util::to_radians(slope_azimuth);
138  // do check once and make function just return 1.0 if no slope
139  const auto no_correction = [](const double) noexcept
140  {
141  return 1.0;
142  };
143  const auto do_correction = [b_semi, slope_radians](const double theta) noexcept
144  {
145  // never gets called if isInvalid() so don't check
146  // figure out how far the ground distance is in map distance horizontally
147  auto angle_unrotated = theta - slope_radians;
148  if (util::to_degrees(angle_unrotated) == 270 || util::to_degrees(angle_unrotated) == 90)
149  {
150  // CHECK: if we're going directly across the slope then horizontal distance is same as spread distance
151  return 1.0;
152  }
153  const auto tan_u = tan(angle_unrotated);
154  const auto y = b_semi / sqrt(b_semi * tan_u * (b_semi * tan_u) + 1.0);
155  const auto x = y * tan_u;
156  // CHECK: Pretty sure you can't spread farther horizontally than the spread distance, regardless of angle?
157  return min(1.0, sqrt(x * x + y * y));
158  };
159  const auto correction_factor = has_no_slope
160  ? std::function<double(double)>(no_correction)
161  : std::function<double(double)>(do_correction);
162  const auto cell_size = scenario.cellSize();
163  const auto add_offset = [this, cell_size, min_ros](const double direction,
164  const double ros)
165  {
166  if (ros < min_ros)
167  {
168  return false;
169  }
170  // spreading, so figure out offset from current point
171  const auto ros_cell = ros / cell_size;
172  offsets_.emplace_back(ros_cell * _sin(direction), ros_cell * _cos(direction));
173  return true;
174  };
175  const auto threshold = scenario.spreadThresholdByRos(time_);
176  // if not over spread threshold then don't spread
177  // HACK: assume there is no fuel where a crown fire's sfc is < COMPARE_LIMIT and its fc is >
178  double ros{};
179  // HACK: set ros in boolean if we get that far so that we don't have to repeat the if body
180  if (head_ros_ < threshold
181  || sfc < COMPARE_LIMIT
182  || !add_offset(raz, ros = (head_ros_ * correction_factor(raz))))
183  {
184  // mark as invalid
185  head_ros_ = -1;
186  return;
187  }
188  auto fc = sfc;
189  // don't need to re-evaluate if crown with new head_ros_ because it would only go up if is_crown
190  if (fuel->canCrown() && is_crown)
191  {
192  // wouldn't be crowning if ros is 0 so that's why this is in an else
193  fc += fuel->crownConsumption(fuel->crownFractionBurned(head_ros_, rso));
194  }
195  // max intensity should always be at the head
196  max_intensity_ = fuel::fire_intensity(fc, ros);
197  const auto a = (head_ros_ + back_ros) / 2.0;
198  const auto c = a - back_ros;
199  const auto flank_ros = a / fuel->lengthToBreadth(wsv);
200  const auto a_sq = a * a;
201  const auto flank_ros_sq = flank_ros * flank_ros;
202  const auto a_sq_sub_c_sq = a_sq - (c * c);
203  const auto ac = a * c;
204  const auto calculate_ros =
205  [a, c, ac, flank_ros, a_sq, flank_ros_sq, a_sq_sub_c_sq](const double theta) noexcept
206  {
207  const auto cos_t = _cos(theta);
208  const auto cos_t_sq = cos_t * cos_t;
209  const auto f_sq_cos_t_sq = flank_ros_sq * cos_t_sq;
210  // 1.0 = cos^2 + sin^2
211  // const auto sin_t_sq = 1.0 - cos_t_sq;
212  const auto sin_t = _sin(theta);
213  const auto sin_t_sq = sin_t * sin_t;
214  return abs((a * ((flank_ros * cos_t * sqrt(f_sq_cos_t_sq + a_sq_sub_c_sq * sin_t_sq) - ac * sin_t_sq) / (f_sq_cos_t_sq + a_sq * sin_t_sq)) + c) / cos_t);
215  };
216  const auto add_offsets =
217  [&correction_factor, &add_offset, raz, threshold](
218  const double angle_radians,
219  const double ros_flat)
220  {
221  if (ros_flat < threshold)
222  {
223  return false;
224  }
225  auto direction = util::fix_radians(angle_radians + raz);
226  // spread is symmetrical across the center axis, but needs to be adjusted if on a slope
227  // intentionally don't use || because we want both of these to happen all the time
228  auto added = add_offset(direction, ros_flat * correction_factor(direction));
229  direction = util::fix_radians(raz - angle_radians);
230  added |= add_offset(direction, ros_flat * correction_factor(direction));
231  return added;
232  };
233  const auto add_offsets_calc_ros =
234  [&add_offsets, &calculate_ros](const double angle_radians)
235  {
236  return add_offsets(angle_radians, calculate_ros(angle_radians));
237  };
238  bool added = add_offset(raz, head_ros_);
239  constexpr size_t STEP = 10;
240  size_t i = STEP;
241  while (added && i < 90)
242  {
243  added = add_offsets_calc_ros(util::to_radians(i));
244  i += STEP;
245  }
246  if (added)
247  {
248  added = add_offsets(util::to_radians(90), flank_ros * sqrt(a_sq_sub_c_sq) / a);
249  i = 90 + STEP;
250  while (added && i < 180)
251  {
252  added = add_offsets_calc_ros(util::to_radians(i));
253  i += STEP;
254  }
255  if (added)
256  {
257  //only use back ros if every other angle is spreading since this should be lowest
258  // 180
259  if (back_ros < threshold)
260  {
261  return;
262  }
263  const auto direction = util::fix_radians(util::RAD_180 + raz);
264  static_cast<void>(!add_offset(direction, back_ros * correction_factor(direction)));
265  }
266  }
