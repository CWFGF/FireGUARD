22 static vector<T> parse_list(string str, T (*convert)(const string& s))
23 {
24  vector<int> result{};
25  // static const vector<int> OUTPUT_DATE_OFFSETS = { 1, 2, 3, 7, 14, 28 };
26  logging::check_fatal(str[0] != '{', "Expected list starting with '{'");
27  istringstream iss(str.substr(1));
28  while (getline(iss, str, ','))
29  {
30  // need to make sure this isn't an empty list
31  if (0 != strcmp("}", str.c_str()))
32  {
33  result.push_back(convert(str));
34  }
35  }
36  return result;

396 string get_value(unordered_map<string, string>& settings, const string& key)
397 {
398  const auto found = settings.find(key);
399  if (found != settings.end())
400  {
401  auto result = found->second;
402  settings.erase(found);
403  return result;
404  }
405  logging::fatal("Missing setting for %s", key.c_str());
406  // HACK: use return to avoid compiler warning
407  static const string Invalid = "INVALID";
408  return Invalid;
409 }
410 SettingsImplementation::SettingsImplementation(const char* filename) noexcept
411 {
412  try
413  {
414  unordered_map<string, string> settings{};
415  ifstream in;
416  in.open(filename);
417  if (in.is_open())
418  {
419  string str;
420  logging::info("Reading settings from '%s'", filename);
421  while (getline(in, str))
422  {
423  istringstream iss(str);
424  if (getline(iss, str, '#'))
425  {
426  iss = istringstream(str);
427  }
428  if (getline(iss, str, '='))
429  {
430  const auto key = util::trim_copy(str);
431  getline(iss, str, '\n');
432  const auto value = util::trim_copy(str);
433  settings.emplace(key, value);
434  }
435  }
436  in.close();
437  }
438  raster_root_ = get_value(settings, "RASTER_ROOT");
439  fuel_lookup_table_file_ = get_value(settings, "FUEL_LOOKUP_TABLE");
440  // HACK: run into fuel consumption being too low if we don't have a minimum ros
441  static const auto MinRos = 0.05;
442  // HACK: make sure this is always > 0 so that we don't have to check
443  // specifically for 0 to avoid div error
444  minimum_ros_ = max(stod(get_value(settings, "MINIMUM_ROS")), MinRos);
445  maximum_spread_distance_ = stod(get_value(settings, "MAX_SPREAD_DISTANCE"));
446  minimum_ffmc_ = stod(get_value(settings, "MINIMUM_FFMC"));
447  minimum_ffmc_at_night_ = stod(get_value(settings, "MINIMUM_FFMC_AT_NIGHT"));
448  offset_sunrise_ = stod(get_value(settings, "OFFSET_SUNRISE"));
449  offset_sunset_ = stod(get_value(settings, "OFFSET_SUNSET"));
450  confidence_level_ = stod(get_value(settings, "CONFIDENCE_LEVEL"));
451  maximum_time_seconds_ = stol(get_value(settings, "MAXIMUM_TIME"));
452  threshold_scenario_weight_ = stod(get_value(settings, "THRESHOLD_SCENARIO_WEIGHT"));
453  threshold_daily_weight_ = stod(get_value(settings, "THRESHOLD_DAILY_WEIGHT"));
454  threshold_hourly_weight_ = stod(get_value(settings, "THRESHOLD_HOURLY_WEIGHT"));
455  setOutputDateOffsets(get_value(settings, "OUTPUT_DATE_OFFSETS").c_str());
456  default_percent_conifer_ = stoi(get_value(settings, "DEFAULT_PERCENT_CONIFER"));
457  default_percent_dead_fir_ = stoi(get_value(settings, "DEFAULT_PERCENT_DEAD_FIR"));
458  intensity_max_low_ = stoi(get_value(settings, "INTENSITY_MAX_LOW"));
459  intensity_max_moderate_ = stoi(get_value(settings, "INTENSITY_MAX_MODERATE"));
460  if (!settings.empty())
461  {
462  logging::warning("Unused settings in settings file %s", filename);
463  for (const auto& kv : settings)
464  {
465  logging::warning("%s = %s", kv.first.c_str(), kv.second.c_str());
466  }
467  }
468  }
469  catch (...)
470  {
471  std::terminate();
472  }
473 }
