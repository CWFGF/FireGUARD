151  const string& fuel_name,

160 {
161  static const auto Latitude = 49.3911;
162  static const auto Longitude = -84.7395;
163  static const topo::StartPoint ForPoint(Latitude, Longitude);
164  const auto start_date = 123;
165  const auto end_date = start_date + static_cast<double>(num_hours) / DAY_HOURS;
166  util::make_directory_recursive(output_directory);
167  const auto fuel = Settings::fuelLookup().byName(fuel_name);
168  auto values = vector<topo::Cell>();
169  // values.reserve(static_cast<size_t>(MAX_ROWS) * MAX_COLUMNS);
170  for (Idx r = 0; r < MAX_ROWS; ++r)
171  {
172  for (Idx c = 0; c < MAX_COLUMNS; ++c)
173  {
174  values.emplace_back(r, c, slope, aspect, fuel::FuelType::safeCode(fuel));
175  }
176  }
177  const auto cells = new data::ConstantGrid<topo::Cell>{
178  TEST_GRID_SIZE,
179  MAX_ROWS,
180  MAX_COLUMNS,
181  topo::Cell{},
182  -1,
183  TEST_XLLCORNER,
184  TEST_YLLCORNER,
185  TEST_XLLCORNER + TEST_GRID_SIZE * MAX_COLUMNS,
186  TEST_YLLCORNER + TEST_GRID_SIZE * MAX_ROWS,
187  TEST_PROJ4,
188  std::move(values)};
189  TestEnvironment env(cells);
190  const Location start_location(static_cast<Idx>(MAX_ROWS / 2),
191  static_cast<Idx>(MAX_COLUMNS / 2));
192  Model model(ForPoint, &env);
193  const auto start_cell = make_shared<topo::Cell>(model.cell(start_location));
194  TestWeather weather(fuel, start_date, dc, bui, dmc, ffmc, wind);
195  TestScenario scenario(&model, start_cell, ForPoint, start_date, end_date, &weather);
196  auto info = SpreadInfo(scenario,
197  start_date,
198  *start_cell,
199  model.nd(start_date),
200  weather.at(start_date));
201  map<double, ProbabilityMap*> probabilities{};
202  logging::debug("Starting simulation");
203  // NOTE: don't want to reset first because TestScenario handles what that does
204  scenario.run(&probabilities);
205  scenario.saveObservers("");
206  logging::note("Final Size: %0.0f, ROS: %0.2f",
207  scenario.currentFireSize(),
208  info.headRos());
209  return 0;
210 }
