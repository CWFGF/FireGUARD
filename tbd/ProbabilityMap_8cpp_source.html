52 void ProbabilityMap::addProbabilities(const ProbabilityMap& rhs)
53 {
54 #ifndef NDEBUG
55  logging::check_fatal(rhs.time_ != time_, "Wrong time");
56  logging::check_fatal(rhs.start_time_ != start_time_, "Wrong start time");
57  logging::check_fatal(rhs.min_value_ != min_value_, "Wrong min value");
58  logging::check_fatal(rhs.max_value_ != max_value_, "Wrong max value");
59  logging::check_fatal(rhs.low_max_ != low_max_, "Wrong low max value");
60  logging::check_fatal(rhs.med_max_ != med_max_, "Wrong med max value");
61 #endif
62  lock_guard<mutex> lock(mutex_);
63  if (Settings::saveIntensity())
64  {
65  for (auto&& kv : rhs.low_.data)
66  {
67  low_.data[kv.first] += kv.second;
68  }
69  for (auto&& kv : rhs.med_.data)
70  {
71  med_.data[kv.first] += kv.second;
72  }
73  for (auto&& kv : rhs.high_.data)
74  {
75  high_.data[kv.first] += kv.second;
76  }
77  }
78  for (auto&& kv : rhs.all_.data)
79  {
80  all_.data[kv.first] += kv.second;
81  }
82  for (auto size : rhs.sizes_)
83  {
84  static_cast<void>(util::insert_sorted(&sizes_, size));
85  }
86 }
87 void ProbabilityMap::addProbability(const IntensityMap& for_time)
88 {
89  lock_guard<mutex> lock(mutex_);
90  std::for_each(
91  for_time.cbegin(),
92  for_time.cend(),
93  [this](auto&& kv) {
94  const auto k = kv.first;
95  const auto v = kv.second;
96  all_.data[k] += 1;
97  if (Settings::saveIntensity())
98  {
99  if (v >= min_value_ && v <= low_max_)
100  {
101  low_.data[k] += 1;
102  }
103  else if (v > low_max_ && v <= med_max_)
104  {
105  med_.data[k] += 1;
106  }
107  else if (v > med_max_ && v <= max_value_)
108  {
109  high_.data[k] += 1;
110  }
111  else
112  {
113  logging::fatal("Value %d doesn't fit into any range", v);
114  }
115  }
116  });
117  const auto size = for_time.fireSize();
118  static_cast<void>(util::insert_sorted(&sizes_, size));
119 }

177 void ProbabilityMap::saveAll(const Model& model,
178  const tm& start_time,
179  const double time,
180  const double start_day) const
181 {
182  auto t = start_time;
183  auto ticks = mktime(&t);
184  const auto day = static_cast<int>(round(time));
185  ticks += (static_cast<size_t>(day) - t.tm_yday - 1) * DAY_SECONDS;
186  t = *localtime(&ticks);
187  if (sim::Settings::runAsync())
188  {
189  vector<std::future<void>> results{};
190  if (Settings::saveProbability())
191  {
192  results.push_back(async(launch::async,
193  &ProbabilityMap::saveTotal,
194  this,
195  make_string("probability", t, day)));
196  }
197  if (Settings::saveOccurrence())
198  {
199  results.push_back(async(launch::async,
200  &ProbabilityMap::saveTotalCount,
201  this,
202  make_string("occurrence", t, day)));
203  }
204  if (Settings::saveIntensity())
205  {
206  results.push_back(async(launch::async,
207  &ProbabilityMap::saveLow,
208  this,
209  make_string("intensity_L", t, day)));
210  results.push_back(async(launch::async,
211  &ProbabilityMap::saveModerate,
212  this,
213  make_string("intensity_M", t, day)));
214  results.push_back(async(launch::async,
215  &ProbabilityMap::saveHigh,
216  this,
217  make_string("intensity_H", t, day)));
218  }
219  results.push_back(async(launch::async,
220  &ProbabilityMap::saveSizes,
221  this,
222  make_string("sizes", t, day)));
223  for (auto& result : results)
224  {
225  result.wait();
226  }
227  }
228  else
229  {
230  if (Settings::saveProbability())
231  {
232  saveTotal(make_string("probability", t, day));
233  }
234  if (Settings::saveOccurrence())
235  {
236  saveTotalCount(make_string("occurrence", t, day));
237  }
238  if (Settings::saveIntensity())
239  {
240  saveLow(make_string("intensity_L", t, day));
241  saveModerate(make_string("intensity_M", t, day));
242  saveHigh(make_string("intensity_H", t, day));
243  }
244  saveSizes(make_string("sizes", t, day));
245  }
246  const auto nd = model.nd(day);
247  logging::note("Fuels for day %d are %s green-up and grass has %d%% curing",
248  day - static_cast<int>(start_day),
249  fuel::calculate_is_green(nd) ? "after" : "before",
250  fuel::calculate_grass_curing(nd));
251 }
