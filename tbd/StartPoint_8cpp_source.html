45 static double sunrise_sunset(const int jd,
46  const double latitude,
47  const double longitude,
48  const bool for_sunrise) noexcept
49 {
50  static const auto Zenith = util::to_radians(96);
51  static const auto LocalOffset = -5;
52  const auto t_hour = for_sunrise ? 6 : 18;
53  // http://edwilliams.org/sunrise_sunset_algorithm.htm
54  const auto lng_hour = longitude / 15;
55  const auto t = jd + (t_hour - lng_hour) / 24;
56  const auto m = 0.9856 * t - 3.289;
57  const auto l = fix_degrees(
58  m + 1.916 * _sin(util::to_radians(m)) + 0.020 * _sin(util::to_radians(2 * m)) + 282.634);
59  auto ra = fix_degrees(util::to_degrees(atan(0.91764 * tan(util::to_radians(l)))));
60  const auto l_quadrant = floor(l / 90) * 90;
61  const auto ra_quadrant = floor(ra / 90) * 90;
62  ra += l_quadrant - ra_quadrant;
63  ra /= 15;
64  const auto sin_dec = 0.39782 * _sin(util::to_radians(l));
65  const auto cos_dec = _cos(asin(sin_dec));
66  const auto cos_h = (_cos(Zenith) - sin_dec * _sin(util::to_radians(latitude))) / (cos_dec * _cos(util::to_radians(latitude)));
67  if (cos_h > 1)
68  {
69  // sun never rises
70  return for_sunrise ? -1 : 25;
71  }
72  if (cos_h < -1)
73  {
74  // sun never sets
75  return for_sunrise ? 25 : -1;
76  }
77  auto h = util::to_degrees(acos(cos_h));
78  if (for_sunrise)
79  {
80  h = 360 - h;
81  }
82  h /= 15;
83  const auto mean_t = h + ra - 0.06571 * t - 6.622;
84  const auto ut = mean_t - lng_hour;
85  return fix_hours(ut + LocalOffset);
86 }

100 {
