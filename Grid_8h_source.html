554 template <class R>
555 string saveToAsciiFile(const string& dir,
556 const string& base_name,
557 std::function<R(T value)> convert) const
558 {
559#ifdef DEBUG_GRIDS
560 // enforce converting to an int and back produces same V
561 const auto n0 = this->nodataInput();
562 const auto n1 = static_cast<NodataIntType>(n0);
563 const auto n2 = static_cast<V>(n1);
564 const auto n3 = static_cast<NodataIntType>(n2);
565 const auto v0 = this->nodataValue();
566 logging::check_equal(
567 n1,
568 n3,
569 "nodata_input_ as int");
570 logging::check_equal(
571 n0,
572 n2,
573 "nodata_input_ from int");
574 logging::check_equal(
575 convert(v0),
576 n0,
577 "convert nodata");
578#endif
579 tuple<Idx, Idx, Idx, Idx> bounds = dataBounds();
580 auto min_column = std::get<0>(bounds);
581 auto min_row = std::get<1>(bounds);
582 auto max_column = std::get<2>(bounds);
583 auto max_row = std::get<3>(bounds);
584 logging::note(
585 "Bounds are (%d, %d), (%d, %d)",
586 min_column,
587 min_row,
588 max_column,
589 max_row);
590 logging::extensive("Lower left corner is (%d, %d)", min_column, min_row);
591 logging::extensive("Upper right corner is (%d, %d)", max_column, max_row);
592 const double xll = this->xllcorner() + min_column * this->cellSize();
593 // offset is different for y since it's flipped
594 const double yll = this->yllcorner() + (min_row) * this->cellSize();
595 logging::extensive("Lower left corner is (%f, %f)", xll, yll);
596 // HACK: make sure it's always at least 1
597 const auto num_rows = static_cast<double>(max_row) - min_row + 1;
598 const auto num_columns = static_cast<double>(max_column) - min_column + 1;
599 ofstream out;
600 string filename = dir + base_name + ".asc";
601 out.open(filename.c_str());
602 write_ascii_header(
603 out,
604 num_columns,
605 num_rows,
606 xll,
607 yll,
608 this->cellSize(),
609 static_cast<double>(this->nodataInput()));
610 for (Idx ro = 0; ro < num_rows; ++ro)
611 {
612 // HACK: do this so that we always get at least one pixel in output
613 // need to output in reverse order since (0,0) is bottom left
614 const Idx r = static_cast<Idx>(max_row) - ro;
615 for (Idx co = 0; co < num_columns; ++co)
616 {
617 const Location idx(static_cast<Idx>(r), static_cast<Idx>(min_column + co));
618 // HACK: use + here so that it gets promoted to a printable number
619 // prevents char type being output as characters
620 out << +(convert(this->at(idx)))
621 << " ";
622 }
623 out << "\n";
624 }
625 out.close();
626 this->createPrj(dir, base_name);
627 return filename;
628 }

651 template <class R>
652 string _saveToTiffFile(const string& dir,
653 const string& base_name,
654 std::function<R(T value)> convert) const
655 {
656#ifdef DEBUG_GRIDS
657 // enforce converting to an int and back produces same V
658 const auto n0 = this->nodataInput();
659 const auto n1 = static_cast<NodataIntType>(n0);
660 const auto n2 = static_cast<V>(n1);
661 const auto n3 = static_cast<NodataIntType>(n2);
662 const auto v0 = this->nodataValue();
663 logging::check_equal(
664 n1,
665 n3,
666 "nodata_input_ as int");
667 logging::check_equal(
668 n0,
669 n2,
670 "nodata_input_ from int");
671 logging::check_equal(
672 convert(v0),
673 n0,
674 "convert nodata");
675#endif
676 uint32_t tileWidth = min((int)(this->columns()), 256);
677 uint32_t tileHeight = min((int)(this->rows()), 256);
678 tuple<Idx, Idx, Idx, Idx> bounds = dataBounds();
679 auto min_column = std::get<0>(bounds);
680 auto min_row = std::get<1>(bounds);
681 auto max_column = std::get<2>(bounds);
682 auto max_row = std::get<3>(bounds);
683 logging::check_fatal(
684 min_column > max_column,
685 "Invalid bounds for columns with %d => %d",
686 min_column,
687 max_column);
688 logging::check_fatal(
689 min_row > max_row,
690 "Invalid bounds for rows with %d => %d",
691 min_row,
692 max_row);
693#ifdef DEBUG_GRIDS
694 logging::note(
695 "Bounds are (%d, %d), (%d, %d) initially",
696 min_column,
697 min_row,
698 max_column,
699 max_row);
700#endif
701 Idx c_min = 0;
702 while (c_min + static_cast<Idx>(tileWidth) <= min_column)
703 {
704 c_min += static_cast<Idx>(tileWidth);
705 }
706 Idx c_max = c_min + static_cast<Idx>(tileWidth);
707 while (c_max < max_column)
708 {
709 c_max += static_cast<Idx>(tileWidth);
710 }
711 min_column = c_min;
712 max_column = c_max;
713 Idx r_min = 0;
714 while (r_min + static_cast<Idx>(tileHeight) <= min_row)
715 {
716 r_min += static_cast<Idx>(tileHeight);
717 }
718 Idx r_max = r_min + static_cast<Idx>(tileHeight);
719 while (r_max < max_row)
720 {
721 r_max += static_cast<Idx>(tileHeight);
722 }
723 min_row = r_min;
724 max_row = r_max;
725 logging::check_fatal(
726 min_column >= max_column,
727 "Invalid bounds for columns with %d => %d",
728 min_column,
729 max_column);
730 logging::check_fatal(
731 min_row >= max_row,
732 "Invalid bounds for rows with %d => %d",
733 min_row,
734 max_row);
735#ifdef DEBUG_GRIDS
736 logging::note(
737 "Bounds are (%d, %d), (%d, %d) after correction",
738 min_column,
739 min_row,
740 max_column,
741 max_row);
742#endif
743 logging::extensive("(%d, %d) => (%d, %d)", min_column, min_row, max_column, max_row);
744 logging::check_fatal((max_row - min_row) % tileHeight != 0, "Invalid start and end rows");
745 logging::check_fatal((max_column - min_column) % tileHeight != 0, "Invalid start and end columns");
746 logging::extensive("Lower left corner is (%d, %d)", min_column, min_row);
747 logging::extensive("Upper right corner is (%d, %d)", max_column, max_row);
748 const double xll = this->xllcorner() + min_column * this->cellSize();
749 // offset is different for y since it's flipped
750 const double yll = this->yllcorner() + (min_row) * this->cellSize();
751 logging::extensive("Lower left corner is (%f, %f)", xll, yll);
752 const auto num_rows = static_cast<size_t>(max_row - min_row);
753 const auto num_columns = static_cast<size_t>(max_column - min_column);
754 // ensure this is always divisible by tile size
755 logging::check_fatal(0 != (num_rows % tileWidth), "%d rows not divisible by tiles", num_rows);
756 logging::check_fatal(0 != (num_columns % tileHeight), "%d columns not divisible by tiles", num_columns);
757 string filename = dir + base_name + ".tif";
758 TIFF* tif = GeoTiffOpen(filename.c_str(), "w");
759 auto gtif = GTIFNew(tif);
760 logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
761 const double xul = xll;
762 const double yul = this->yllcorner() + (this->cellSize() * max_row);
763 double tiePoints[6] = {
764 0.0,
765 0.0,
766 0.0,
767 xul,
768 yul,
769 0.0};
770 double pixelScale[3] = {
771 this->cellSize(),
772 this->cellSize(),
773 0.0};
774 uint32_t bps = sizeof(R) * 8;
775 // make sure to use floating point if values are
776 if (std::is_floating_point<R>::value)
777 {
778 TIFFSetField(tif, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
779 }
780 // FIX: was using double, and that usually doesn't make sense, but sometime it might?
781 // use buffer big enought to fit any (V + '.000\0') + 1
782 constexpr auto n = std::numeric_limits<V>::digits10;
783 static_assert(n > 0);
784 char str[n + 6]{0};
785 const auto nodata_as_int = static_cast<int>(this->nodataInput());
786 sxprintf(str, "%d.000", nodata_as_int);
787 logging::extensive(
788 "%s using nodata string '%s' for nodata value of (%d, %f)",
789 typeid(this).name(),
790 str,
791 nodata_as_int,
792 static_cast<double>(this->nodataInput()));
793 TIFFSetField(tif, TIFFTAG_GDAL_NODATA, str);
794 logging::extensive("%s takes %d bits", base_name.c_str(), bps);
795 TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, num_columns);
796 TIFFSetField(tif, TIFFTAG_IMAGELENGTH, num_rows);
797 TIFFSetField(tif, TIFFTAG_SAMPLESPERPIXEL, 1);
798 TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, bps);
799 TIFFSetField(tif, TIFFTAG_TILEWIDTH, tileWidth);
800 TIFFSetField(tif, TIFFTAG_TILELENGTH, tileHeight);
801 TIFFSetField(tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
802 TIFFSetField(tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
803 TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_LZW);
804 GTIFSetFromProj4(gtif, this->proj4().c_str());
805 TIFFSetField(tif, TIFFTAG_GEOTIEPOINTS, 6, tiePoints);
806 TIFFSetField(tif, TIFFTAG_GEOPIXELSCALE, 3, pixelScale);
807 size_t tileSize = tileWidth * tileHeight;
808 const auto buf_size = tileSize * sizeof(R);
809 logging::extensive("%s has buffer size %d", base_name.c_str(), buf_size);
810 R* buf = (R*)_TIFFmalloc(buf_size);
811 for (size_t co = 0; co < num_columns; co += tileWidth)
812 {
813 for (size_t ro = 0; ro < num_rows; ro += tileHeight)
814 {
815 // NOTE: shouldn't need to check if writing outside of tile because we made bounds on tile edges above
816 // need to put data from grid into buffer, but flipped vertically
817 for (size_t x = 0; x < tileWidth; ++x)
818 {
819 for (size_t y = 0; y < tileHeight; ++y)
820 {
821 const Idx r = static_cast<Idx>(max_row) - (ro + y + 1);
822 const Idx c = static_cast<Idx>(min_column) + co + x;
823 const Location idx(r, c);
824 // might be out of bounds if not divisible by number of tiles
825 const R value =
826 (this->rows() <= r
827 || 0 > r
828 || this->columns() <= c
829 || 0 > c)
830 ? this->nodataInput()
831 : convert(this->at(idx));
832 buf[x + y * tileWidth] = value;
833 }
834 }
835 logging::check_fatal(TIFFWriteTile(tif, buf, co, ro, 0, 0) < 0, "Cannot write tile to %s", filename.c_str());
836 }
837 }
838 GTIFWriteKeys(gtif);
839 if (gtif)
840 {
841 GTIFFree(gtif);
842 }
843 _TIFFfree(buf);
844 TIFFClose(tif);
845 return filename;
846 }
847 template <class R>
848 string saveToTiffFile(const string& dir,
849 const string& base_name,
850 std::function<R(T value)> convert) const
851 {
852 // HACK: (hopefully) ensure that write works
853 try

862 logging::error(err.what());
863 return _saveToTiffFile<R>(dir, base_name, convert);
864 }
865 }
