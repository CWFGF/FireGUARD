271 [[nodiscard]] double crownFractionBurned(const double rss,
272 const double rso) const noexcept override
273 {
274 return max(0.0, 1.0 - exp(-0.230 * (rss - rso)));
275 }
281 [[nodiscard]] double probabilityPeat(const double mc_fraction) const noexcept override
282 {
283 // Anderson table 1
284 constexpr auto pb = bulkDensity();
285 // Anderson table 1
286 constexpr auto fi = inorganicPercent();
287 constexpr auto pi = fi * pb;
288 // Inorganic ratio
289 constexpr auto ri = fi / (1 - fi);
290 constexpr auto const_part = -19.329 + 1.7170 * ri + 23.059 * pi;
291 // Anderson eq 1
292 return 1 / (1 + exp(17.047 * mc_fraction / (1 - fi) + const_part));
293 }
299 [[nodiscard]] double survivalProbability(const wx::FwiWeather& wx) const noexcept
300 override
301 {
302 // divide by 100 since we need moisture ratio
303 // IFERROR(((1 / (1 + EXP($G$43 + $I$43 *
304 // (Q$44 * $O$43 + $N$43)))) -
305 // (1 / (1 + EXP($G$43 + $I$43 * (2.5 * $O$43 + $N$43)))))
306 // / (1 / (1 + EXP($G$43 + $I$43 * $N$43))), 0)
307 // HACK: use same constants for all fuels because they seem to work nicer than
308 // using the ratios, but they change anyway because of the other fuel attributes
309 static const auto WFfmc = 0.25;
310 static const auto WDmc = 1.0;
311 static const auto RatioHartford = 0.5;
312 static const auto RatioFrandsen = 1.0 - RatioHartford;
313 static const auto RatioAspen = 0.5;
314 static const auto RatioFuel = 1.0 - RatioAspen;
315 const auto mc_ffmc = wx.mcFfmc() * WFfmc + WDmc;
316 static const auto McFfmcSaturated = 2.5 * WFfmc + WDmc;
317 static const auto McDmc = WDmc;
318 const auto prob_ffmc_peat = probabilityPeat(mc_ffmc);
319 const auto prob_ffmc_peat_saturated = probabilityPeat(McFfmcSaturated);
320 const auto prob_ffmc_peat_zero = probabilityPeat(McDmc);
321 const auto prob_ffmc_peat_weighted = (prob_ffmc_peat - prob_ffmc_peat_saturated) / prob_ffmc_peat_zero;
322 const auto prob_ffmc = duffFfmcType()->probabilityOfSurvival(mc_ffmc * 100);
323 const auto prob_ffmc_saturated = duffFfmcType()->probabilityOfSurvival(
324 McFfmcSaturated * 100);
325 const auto prob_ffmc_zero = duffFfmcType()->probabilityOfSurvival(McDmc);
326 const auto prob_ffmc_weighted = (prob_ffmc - prob_ffmc_saturated) / prob_ffmc_zero;
327 const auto term_otway = exp(-3.11 + 0.12 * wx.dmc().asDouble());
328 const auto prob_otway = term_otway / (1 + term_otway);
329 const auto mc_pct = wx.mcDmcPct() * dmcRatio() + wx.mcFfmcPct() * ffmcRatio();
330 const auto prob_weight_ffmc = duffFfmcType()->probabilityOfSurvival(mc_pct);
331 const auto prob_weight_ffmc_peat = probabilityPeat(mc_pct / 100);
332 const auto prob_weight_dmc = duffDmcType()->probabilityOfSurvival(wx.mcDmcPct());
333 const auto prob_weight_dmc_peat = probabilityPeat(wx.mcDmc());
334 // chance of survival is 1 - chance of it not surviving in every fuel
335 const auto tot_prob = 1 - (1 - prob_ffmc_peat_weighted) * (1 - prob_ffmc_weighted) * ((1 - prob_otway) * RatioAspen + ((1 - prob_weight_ffmc_peat) * RatioHartford + (1 - prob_weight_ffmc) * RatioFrandsen) * ((1 - prob_weight_dmc_peat) * RatioHartford + (1 - prob_weight_dmc) * RatioFrandsen) * RatioFuel);
336 return tot_prob;
337 }
