281[[nodiscard]] constexpr T pow_int(const T& base)
282{
283 // https://stackoverflow.com/questions/16443682/c-power-of-integer-template-meta-programming
284 return N == 0
285 ? 1
286 : N % 2 == 0
287 ? pow_int<N / 2, T>(base) * pow_int<N / 2, T>(base)

429template <typename T, typename V>
430T binary_find(const T lower,
431 const T upper,
432 const double value,
433 const std::function<V(T)>& fct)
434{
435 const auto mid = lower + (upper - lower) / 2;
436 if (lower == upper)
437 {
438 return lower;
439 }
440 if (fct(mid) < value)
441 {
442 return binary_find(lower, mid, value, fct);
443 }
444 return binary_find(mid + 1, upper, value, fct);
445}
456template <typename T, typename V>
457T binary_find_checked(const T lower,
458 const T upper,
459 const double value,
460 const std::function<V(T)>& fct)
461{
462 if (fct(lower) < value)
463 {
464 return lower;
465 }
466 if (fct(upper) >= value)
467 {
468 return upper;
469 }
470 return binary_find(lower, upper, value, fct);
471}
