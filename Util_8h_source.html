350 template <typename T, typename V>
351 T binary_find(const T lower,
352  const T upper,
353  const double value,
354  const std::function<V(T)>& fct)
355 {
356  const auto mid = lower + (upper - lower) / 2;
357  if (lower == upper)
358  {
359  return lower;
360  }
361  if (fct(mid) < value)
362  {
363  return binary_find(lower, mid, value, fct);
364  }
365  return binary_find(mid + 1, upper, value, fct);
366 }
