287[[nodiscard]] constexpr T pow_int(const T& base)
288{
289 // https://stackoverflow.com/questions/16443682/c-power-of-integer-template-meta-programming
290 return N == 0
291 ? 1
292 : N % 2 == 0

435template <typename T, typename V>
436T binary_find(const T lower,
437 const T upper,
438 const MathSize value,
439 const std::function<V(T)>& fct)
440{
441 const auto mid = lower + (upper - lower) / 2;
442 if (lower == upper)
443 {
444 return lower;
445 }
446 if (fct(mid) < value)
447 {
448 return binary_find(lower, mid, value, fct);
449 }
450 return binary_find(mid + 1, upper, value, fct);
451}
462template <typename T, typename V>
463T binary_find_checked(const T lower,
464 const T upper,
465 const MathSize value,
466 const std::function<V(T)>& fct)
467{
468 if (fct(lower) < value)
469 {
470 return lower;
471 }
472 if (fct(upper) >= value)
473 {
474 return upper;
475 }
476 return binary_find(lower, upper, value, fct);
477}
