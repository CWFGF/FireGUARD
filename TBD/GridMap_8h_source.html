234  template <class R>
235  void saveToAsciiFile(const string& dir,
236  const string& base_name,
237  std::function<R(V value)> convert) const
238  {
239  Idx min_row = this->rows();
240  int16 max_row = 0;
241  Idx min_column = this->columns();
242  Idx max_column = 0;
243  for (const auto& kv : this->data)
244  {
245  const Idx r = kv.first.row();
246  const Idx c = kv.first.column();
247  min_row = min(min_row, r);
248  max_row = max(max_row, r);
249  min_column = min(min_column, c);
250  max_column = max(max_column, c);
251  }
252  // do this so that we take the center point when there's no data since it should
253  // stay the same if the grid is centered on the fire
254  if (min_row > max_row)
255  {
256  min_row = max_row = this->rows() / 2;
257  }
258  if (min_column > max_column)
259  {
260  min_column = max_column = this->columns() / 2;
261  }
262  logging::extensive("Lower left corner is (%d, %d)", min_column, min_row);
263  logging::extensive("Upper right corner is (%d, %d)", max_column, max_row);
264  const double xll = this->xllcorner() + min_column * this->cellSize();
265  // offset is different for y since it's flipped
266  const double yll = this->yllcorner() + (min_row) * this->cellSize();
267  logging::extensive("Lower left corner is (%f, %f)", xll, yll);
268  // HACK: make sure it's always at least 1
269  const auto num_rows = static_cast<double>(max_row) - min_row + 1;
270  const auto num_columns = static_cast<double>(max_column) - min_column + 1;
271  ofstream out;
272  out.open(dir + base_name + ".asc");
273  write_ascii_header(out,
274  num_columns,
275  num_rows,
276  xll,
277  yll,
278  this->cellSize(),
279  static_cast<double>(this->noDataInt()));
280  for (Idx ro = 0; ro < num_rows; ++ro)
281  {
282  // HACK: do this so that we always get at least one pixel in output
283  // need to output in reverse order since (0,0) is bottom left
284  const Idx r = static_cast<Idx>(max_row) - ro;
285  for (Idx co = 0; co < num_columns; ++co)
286  {
287  const Location idx(static_cast<Idx>(r), static_cast<Idx>(min_column + co));
288  // HACK: use + here so that it gets promoted to a printable number
289  // prevents char type being output as characters
290  out << +((this->data.find(idx) != this->data.end())
291  ? convert(this->data.at(idx))
292  : this->noData())
293  << " ";
294  }
295  out << "\n";
296  }
297  out.close();
298  this->createPrj(dir, base_name);
299  }

320  template <class R>
321  void saveToTiffFile(const string& dir,
322  const string& base_name,
323  std::function<R(V value)> convert) const
324  {
325  constexpr uint32 tileWidth = 256;
326  constexpr uint32 tileHeight = 256;
327  Idx min_row = this->rows();
328  int16 max_row = 0;
329  Idx min_column = this->columns();
330  Idx max_column = 0;
331  for (const auto& kv : this->data)
332  {
333  const Idx r = kv.first.row();
334  const Idx c = kv.first.column();
335  min_row = min(min_row, r);
336  max_row = max(max_row, r);
337  min_column = min(min_column, c);
338  max_column = max(max_column, c);
339  }
340  // do this so that we take the center point when there's no data since it should
341  // stay the same if the grid is centered on the fire
342  if (min_row > max_row)
343  {
344  min_row = max_row = this->rows() / 2;
345  }
346  if (min_column > max_column)
347  {
348  min_column = max_column = this->columns() / 2;
349  }
350  Idx c_min = 0;
351  while (c_min + static_cast<Idx>(tileWidth) <= min_column)
352  {
353  c_min += static_cast<Idx>(tileWidth);
354  }
355  Idx c_max = c_min + static_cast<Idx>(tileWidth);
356  while (c_max < max_column)
357  {
358  c_max += static_cast<Idx>(tileWidth);
359  }
360  min_column = c_min;
361  max_column = c_max;
362  Idx r_min = 0;
363  while (r_min + static_cast<Idx>(tileHeight) <= min_row)
364  {
365  r_min += static_cast<Idx>(tileHeight);
366  }
367  Idx r_max = r_min + static_cast<Idx>(tileHeight);
368  while (r_max < max_row)
369  {
370  r_max += static_cast<Idx>(tileHeight);
371  }
372  min_row = r_min;
373  max_row = r_max;
374  logging::extensive("(%d, %d) => (%d, %d)", min_column, min_row, max_column, max_row);
375  logging::check_fatal((max_row - min_row) % tileHeight != 0,"Invalid start and end rows");
376  logging::check_fatal((max_column - min_column) % tileHeight != 0,"Invalid start and end columns");
377  logging::extensive("Lower left corner is (%d, %d)", min_column, min_row);
378  logging::extensive("Upper right corner is (%d, %d)", max_column, max_row);
379  const double xll = this->xllcorner() + min_column * this->cellSize();
380  // offset is different for y since it's flipped
381  const double yll = this->yllcorner() + (min_row) * this->cellSize();
382  logging::extensive("Lower left corner is (%f, %f)", xll, yll);
383  const auto num_rows = static_cast<size_t>(max_row - min_row);
384  const auto num_columns = static_cast<size_t>(max_column - min_column);
385  // ensure this is always divisible by tile size
386  logging::check_fatal(0 != (num_rows % tileWidth), "%d rows not divisible by tiles", num_rows);
387  logging::check_fatal(0 != (num_columns % tileHeight), "%d columns not divisible by tiles", num_columns);
388  string filename = dir + base_name + ".tif";
389  TIFF* tif = GeoTiffOpen(filename.c_str(), "w");
390  auto gtif = GTIFNew(tif);
391  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
392  const double xul = xll;
393  const double yul = this->yllcorner() + (this->cellSize() * max_row);
394  double tiePoints[6] = {
395  0.0,
396  0.0,
397  0.0,
398  xul,
399  yul,
400  0.0};
401  double pixelScale[3] = {
402  this->cellSize(),
403  this->cellSize(),
404  0.0};
405  uint32 bps = sizeof(R) * 8;
406  // make sure to use floating point if values are
407  if (std::is_floating_point<R>::value)
408  {
409  TIFFSetField(tif, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
410  }
411  int length = snprintf(NULL, 0, "%f", this->noDataInt());
412  char* str = static_cast<char*>(malloc(length + 1));
413  snprintf(str, length + 1, "%f", this->noDataInt());
414  TIFFSetField(tif, TIFFTAG_GDAL_NODATA, str);
415  free(str);
416  logging::extensive("%s takes %d bits", base_name.c_str(), bps);
417  TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, num_columns);
418  TIFFSetField(tif, TIFFTAG_IMAGELENGTH, num_rows);
419  TIFFSetField(tif, TIFFTAG_SAMPLESPERPIXEL, 1);
420  TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, bps);
421  TIFFSetField(tif, TIFFTAG_TILEWIDTH, tileWidth);
422  TIFFSetField(tif, TIFFTAG_TILELENGTH, tileHeight);
423  TIFFSetField(tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
424  TIFFSetField(tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
425  TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_LZW);
426  GTIFSetFromProj4(gtif, this->proj4().c_str());
427  TIFFSetField(tif, TIFFTAG_GEOTIEPOINTS, 6, tiePoints);
428  TIFFSetField(tif, TIFFTAG_GEOPIXELSCALE, 3, pixelScale);
429  size_t tileSize = tileWidth * tileHeight;
430  const auto buf_size = tileSize * sizeof(R);
431  logging::extensive("%s has buffer size %d", base_name.c_str(), buf_size);
432  R* buf = (R*)_TIFFmalloc(buf_size);
433  for (size_t co = 0; co < num_columns; co += tileWidth)
434  {
435  for (size_t ro = 0; ro < num_rows; ro += tileHeight)
436  {
437  // NOTE: shouldn't need to check if writing outside of tile because we made bounds on tile edges above
438  // need to put data from grid into buffer, but flipped vertically
439  for (size_t x = 0; x < tileWidth; ++x)
440  {
441  for (size_t y = 0; y < tileHeight; ++y)
442  {
443  const Idx r = static_cast<Idx>(max_row) - (ro + y + 1);
444  const Idx c = static_cast<Idx>(min_column) + co + x;
445  const Location idx(r, c);
446  const R value = convert(at(idx));
447  buf[x + y * tileWidth] = value;
448  }
449  }
450  logging::check_fatal(TIFFWriteTile(tif, buf, co, ro, 0, 0) < 0, "Cannot write tile to %s", filename.c_str());
451  }
452  }
453  GTIFWriteKeys(gtif);
454  if (gtif)
455  {
456  GTIFFree(gtif);
457  }
458  _TIFFfree(buf);
459  TIFFClose(tif);
460  }

500  [[nodiscard]] list<Location> makeEdge() const
501  {
502  list<Location> edge{};
503  for (const auto& kv : this->data)
504  {
505  auto loc = kv.first;
506  auto on_edge = false;
507  for (Idx r = -1; !on_edge && r <= 1; ++r)
508  {
509  const Idx row_index = loc.row() + r;
510  if (!(row_index < 0 || row_index >= this->rows()))
511  {
512  for (Idx c = -1; !on_edge && c <= 1; ++c)
513  {
514  const Idx col_index = loc.column() + c;
515  if (!(col_index < 0 || col_index >= this->columns())
516  && this->data.find(Location(row_index, col_index)) == this->data.end())
517  {
518  on_edge = true;
519  }
520  }
521  }
522  }
523  if (on_edge)
524  {
525  edge.push_back(loc);
526  }
527  }
528  logging::info("Created edge for perimeter with length %lu m",
529  static_cast<size_t>(this->cellSize() * edge.size()));
530  return edge;
531  }
