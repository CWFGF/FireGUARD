274  [[nodiscard]] double crownFractionBurned(const double rss,
275  const double rso) const noexcept override
276  {
277  return max(0.0, 1.0 - exp(-0.230 * (rss - rso)));
278  }
284  [[nodiscard]] double probabilityPeat(const double mc_fraction) const noexcept override
285  {
286  // Anderson table 1
287  constexpr auto pb = bulkDensity();
288  // Anderson table 1
289  constexpr auto fi = inorganicPercent();
290  constexpr auto pi = fi * pb;
291  // Inorganic ratio
292  constexpr auto ri = fi / (1 - fi);
293  constexpr auto const_part = -19.329 + 1.7170 * ri + 23.059 * pi;
294  // Anderson eq 1
295  return 1 / (1 + exp(17.047 * mc_fraction / (1 - fi) + const_part));
296  }
302  [[nodiscard]] double survivalProbability(const wx::FwiWeather& wx) const noexcept
303  override
304  {
305  // divide by 100 since we need moisture ratio
306  // IFERROR(((1 / (1 + EXP($G$43 + $I$43 *
307  // (Q$44 * $O$43 + $N$43)))) -
308  // (1 / (1 + EXP($G$43 + $I$43 * (2.5 * $O$43 + $N$43)))))
309  // / (1 / (1 + EXP($G$43 + $I$43 * $N$43))), 0)
310  // HACK: use same constants for all fuels because they seem to work nicer than
311  // using the ratios but they change anyway because of the other fuel attributes
312  static const auto WFfmc = 0.25;
313  static const auto WDmc = 1.0;
314  static const auto RatioHartford = 0.5;
315  static const auto RatioFrandsen = 1.0 - RatioHartford;
316  static const auto RatioAspen = 0.5;
317  static const auto RatioFuel = 1.0 - RatioAspen;
318  const auto mc_ffmc = wx.mcFfmc() * WFfmc + WDmc;
319  static const auto McFfmcSaturated = 2.5 * WFfmc + WDmc;
320  static const auto McDmc = WDmc;
321  const auto prob_ffmc_peat = probabilityPeat(mc_ffmc);
322  const auto prob_ffmc_peat_saturated = probabilityPeat(McFfmcSaturated);
323  const auto prob_ffmc_peat_zero = probabilityPeat(McDmc);
324  const auto prob_ffmc_peat_weighted = (prob_ffmc_peat - prob_ffmc_peat_saturated) /
325  prob_ffmc_peat_zero;
326  const auto prob_ffmc = duffFfmcType()->probabilityOfSurvival(mc_ffmc * 100);
327  const auto prob_ffmc_saturated = duffFfmcType()->probabilityOfSurvival(
328  McFfmcSaturated * 100);
329  const auto prob_ffmc_zero = duffFfmcType()->probabilityOfSurvival(McDmc);
330  const auto prob_ffmc_weighted = (prob_ffmc - prob_ffmc_saturated) / prob_ffmc_zero;
331  const auto term_otway = exp(-3.11 + 0.12 * wx.dmc().asDouble());
332  const auto prob_otway = term_otway / (1 + term_otway);
333  const auto mc_pct = wx.mcDmcPct() * dmcRatio() + wx.mcFfmcPct() * ffmcRatio();
334  const auto prob_weight_ffmc = duffFfmcType()->probabilityOfSurvival(mc_pct);
335  const auto prob_weight_ffmc_peat = probabilityPeat(mc_pct / 100);
336  const auto prob_weight_dmc = duffDmcType()->probabilityOfSurvival(wx.mcDmcPct());
337  const auto prob_weight_dmc_peat = probabilityPeat(wx.mcDmc());
338  // chance of survival is 1 - chance of it not surviving in every fuel
339  const auto tot_prob = 1 - (1 - prob_ffmc_peat_weighted) * (1 - prob_ffmc_weighted)
340  * (
341  (1 - prob_otway) * RatioAspen +
342  ((1 - prob_weight_ffmc_peat) * RatioHartford + (1 - prob_weight_ffmc) *
343  RatioFrandsen) *
344  ((1 - prob_weight_dmc_peat) * RatioHartford + (1 - prob_weight_dmc) *
345  RatioFrandsen)
346  * RatioFuel);
347  return tot_prob;
348  }
353  [[nodiscard]] static constexpr double bulkDensity() { return BulkDensity / 1000.0; }
358  [[nodiscard]] static constexpr double inorganicPercent()
359  {
360  return InorganicPercent / 100.0;
361  }
366  [[nodiscard]] static constexpr double duffDepth() { return DuffDepth / 10.0; }
371  [[nodiscard]] constexpr const Duff* duffDmcType() const { return duff_dmc_; }
376  [[nodiscard]] constexpr const Duff* duffFfmcType() const { return duff_ffmc_; }
381  [[nodiscard]] static constexpr double ffmcRatio() { return 1 - dmcRatio(); }
386  [[nodiscard]] static constexpr double dmcRatio()
387  {
388  return (duffDepth() - DUFF_FFMC_DEPTH) / duffDepth();
389  }
390 private:
394  const Duff* duff_ffmc_;
398  const Duff* duff_dmc_;
399 };
