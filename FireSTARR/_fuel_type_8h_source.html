280  [[nodiscard]] double crownFractionBurned(const double rss,
281  const double rso) const noexcept override
282  {
283  return max(0.0, 1.0 - exp(-0.230 * (rss - rso)));
284  }
290  [[nodiscard]] double probabilityPeat(const double mc_fraction) const noexcept override
291  {
292  // Anderson table 1
293  constexpr auto pb = bulkDensity();
294  // Anderson table 1
295  constexpr auto fi = inorganicPercent();
296  constexpr auto pi = fi * pb;
297  // Inorganic ratio
298  constexpr auto ri = fi / (1 - fi);
299  constexpr auto const_part = -19.329 + 1.7170 * ri + 23.059 * pi;
300  // Anderson eq 1
301  return 1 / (1 + exp(17.047 * mc_fraction / (1 - fi) + const_part));
302  }
308  [[nodiscard]] double survivalProbability(const wx::FwiWeather& wx) const noexcept
309  override
310  {
311  // divide by 100 since we need moisture ratio
312  // IFERROR(((1 / (1 + EXP($G$43 + $I$43 *
313  // (Q$44 * $O$43 + $N$43)))) -
314  // (1 / (1 + EXP($G$43 + $I$43 * (2.5 * $O$43 + $N$43)))))
315  // / (1 / (1 + EXP($G$43 + $I$43 * $N$43))), 0)
316  // HACK: use same constants for all fuels because they seem to work nicer than
317  // using the ratios but they change anyway because of the other fuel attributes
318  static const auto WFfmc = 0.25;
319  static const auto WDmc = 1.0;
320  static const auto RatioHartford = 0.5;
321  static const auto RatioFrandsen = 1.0 - RatioHartford;
322  static const auto RatioAspen = 0.5;
323  static const auto RatioFuel = 1.0 - RatioAspen;
324  const auto mc_ffmc = wx.mcFfmc() * WFfmc + WDmc;
325  static const auto McFfmcSaturated = 2.5 * WFfmc + WDmc;
326  static const auto McDmc = WDmc;
327  const auto prob_ffmc_peat = probabilityPeat(mc_ffmc);
328  const auto prob_ffmc_peat_saturated = probabilityPeat(McFfmcSaturated);
329  const auto prob_ffmc_peat_zero = probabilityPeat(McDmc);
330  const auto prob_ffmc_peat_weighted = (prob_ffmc_peat - prob_ffmc_peat_saturated) / prob_ffmc_peat_zero;
331  const auto prob_ffmc = duffFfmcType()->probabilityOfSurvival(mc_ffmc * 100);
332  const auto prob_ffmc_saturated = duffFfmcType()->probabilityOfSurvival(
333  McFfmcSaturated * 100);
334  const auto prob_ffmc_zero = duffFfmcType()->probabilityOfSurvival(McDmc);
335  const auto prob_ffmc_weighted = (prob_ffmc - prob_ffmc_saturated) / prob_ffmc_zero;
336  const auto term_otway = exp(-3.11 + 0.12 * wx.dmc().asDouble());
337  const auto prob_otway = term_otway / (1 + term_otway);
338  const auto mc_pct = wx.mcDmcPct() * dmcRatio() + wx.mcFfmcPct() * ffmcRatio();
339  const auto prob_weight_ffmc = duffFfmcType()->probabilityOfSurvival(mc_pct);
340  const auto prob_weight_ffmc_peat = probabilityPeat(mc_pct / 100);
341  const auto prob_weight_dmc = duffDmcType()->probabilityOfSurvival(wx.mcDmcPct());
342  const auto prob_weight_dmc_peat = probabilityPeat(wx.mcDmc());
343  // chance of survival is 1 - chance of it not surviving in every fuel
344  const auto tot_prob = 1 - (1 - prob_ffmc_peat_weighted) * (1 - prob_ffmc_weighted) * ((1 - prob_otway) * RatioAspen + ((1 - prob_weight_ffmc_peat) * RatioHartford + (1 - prob_weight_ffmc) * RatioFrandsen) * ((1 - prob_weight_dmc_peat) * RatioHartford + (1 - prob_weight_dmc) * RatioFrandsen) * RatioFuel);
345  return tot_prob;
346  }

402  }
