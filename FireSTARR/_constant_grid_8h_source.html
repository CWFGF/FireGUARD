145  [[nodiscard]] static ConstantGrid<T, V>* readTiff(const string& filename,
146  TIFF* tif,
147  GTIF* gtif,
148  const topo::Point& point,
149  std::function<T(int, int)> convert)
150  {
151  logging::info("Reading file %s", filename.c_str());
152 #ifndef NDEBUG
153  V min_value = std::numeric_limits<V>::max();
154  V max_value = std::numeric_limits<V>::min();
155 #endif
156  const GridBase grid_info = read_header<T>(tif, gtif);
157  const auto coordinates = grid_info.findCoordinates(point, false);
158  auto min_column = max(static_cast<Idx>(0),
159  static_cast<Idx>(std::get<1>(*coordinates) - MAX_COLUMNS /
160  static_cast<Idx>(2)));
161  if (min_column + MAX_COLUMNS >= grid_info.columns())
162  {
163  min_column = grid_info.columns() - MAX_COLUMNS;
164  }
165  const auto max_column = min_column + MAX_COLUMNS;
166  auto min_row = max(static_cast<Idx>(0),
167  static_cast<Idx>(std::get<0>(*coordinates) - MAX_COLUMNS /
168  static_cast<Idx>(2)));
169  if (min_row + MAX_COLUMNS >= grid_info.rows())
170  {
171  min_row = grid_info.rows() - MAX_COLUMNS;
172  }
173  const auto max_row = min_row + MAX_COLUMNS;
174  const Idx offset_x = -min_column;
175  const Idx offset_y = -min_row;
176  T no_data = convert(grid_info.nodata(), grid_info.nodata());
177  vector<T> values(MAX_COLUMNS * MAX_COLUMNS, no_data);
178  int tile_width;
179  int tile_length;
180  TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
181  TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
182  logging::warning("%s: malloc start", filename.c_str());
183  const auto buf = _TIFFmalloc(TIFFTileSize(tif));
184  logging::warning("%s: read start", filename.c_str());
185  const tsample_t smp{};
186  for (auto h = 0; h < grid_info.rows(); h += tile_length)
187  {
188  const auto y_min = static_cast<Idx>(max(0, min_row - h));
189  const auto y_limit = min(tile_length,
190  min(static_cast<int>(grid_info.rows()), max_row) - h);
191  for (auto w = 0; w < grid_info.columns(); w += tile_width)
192  {
193  TIFFReadTile(tif, buf, static_cast<uint32>(w), static_cast<uint32>(h), 0, smp);
194  const auto x_min = static_cast<Idx>(max(0, min_column - w));
195  const auto x_limit = min(tile_width,
196  min(static_cast<int>(grid_info.columns()),
197  max_column) - w);
198  for (auto y = y_min; y < y_limit; ++y)
199  {
200  // read in so that (0, 0) has a hash of 0
201  const Idx i = MAX_COLUMNS - (static_cast<Idx>(h) + y + offset_y + 1);
202  for (auto x = x_min; x < x_limit; ++x)
203  {
204  const auto cur_hash = static_cast<HashSize>(i) * MAX_COLUMNS + w + x +
205  offset_x;
206  const auto offset = y * tile_length + x;
207  V cur = *(static_cast<V*>(buf) + offset);
208 #ifndef NDEBUG
209  min_value = min(cur, min_value);
210  max_value = max(cur, max_value);
211 #endif
212  values.at(cur_hash) = convert(cur, grid_info.nodata());
213  }
214  }
215  }
216  }
217  logging::warning("%s: read end", filename.c_str());
218  _TIFFfree(buf);
219  logging::warning("%s: free end", filename.c_str());
220  const auto new_xll = grid_info.xllcorner() - offset_x * grid_info.cellSize();
221  const auto new_yll = grid_info.yllcorner() + (static_cast<double>(grid_info.rows()) -
222  max_row) * grid_info.cellSize();
223  logging::check_fatal(new_yll < grid_info.yllcorner(),
224  "New yllcorner is outside original grid");
225  logging::note("Translated lower left is (%f, %f) from (%f, %f)",
226  new_xll,
227  new_yll,
228  grid_info.xllcorner(),
229  grid_info.yllcorner());
230  auto result = new ConstantGrid<T, V>(grid_info.cellSize(),
231  MAX_COLUMNS,
232  MAX_COLUMNS,
233  no_data,
234  grid_info.nodata(),
235  new_xll,
236  new_yll,
237  string(grid_info.proj4()),
238  std::move(values));
239  auto new_location = result->findCoordinates(point, true);
240  logging::check_fatal(nullptr == new_location, "Invalid location after reading");
241  logging::note("Coordinates are (%d, %d => %f, %f)",
242  std::get<0>(*new_location),
243  std::get<1>(*new_location),
244  std::get<0>(*new_location) + std::get<3>(*new_location) / 1000.0,
245  std::get<1>(*new_location) + std::get<2>(*new_location) / 1000.0);
246 #ifndef NDEBUG
247  logging::note("Values for %s range from %d to %d",
248  filename.c_str(),
249  min_value,
250  max_value);
251 #endif
252  return result;
253  }
