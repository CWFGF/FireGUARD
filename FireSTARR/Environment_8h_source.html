267  const FuelGrid& fuel,

271  logging::check_fatal(fuel.yllcorner() != slope.yllcorner(),
272  "Expected yllcorner %f but got %f",
273  fuel.yllcorner(),

275  logging::check_fatal(fuel.yllcorner() != aspect.yllcorner(),

279  static Cell nodata{};
280  auto values = vector<Cell>{fuel.data.size()};
281  //fuel.saveToAsciiFile("C:/FireGUARD/", "fuel");
282  vector<HashSize> hashes{};
283  for (HashSize h = 0; h < static_cast<size_t>(fuel.rows()) * fuel.columns(); ++h)
284  {
285  hashes.emplace_back(h);
286  }
287  std::for_each(
288  std::execution::par_unseq,
289  hashes.begin(),
290  hashes.end(),
291  [&fuel, &slope, &aspect, &values](auto&& h)
292  {
293  const auto s = min(static_cast<SlopeSize>(MAX_SLOPE_FOR_DISTANCE),
294  slope.at(h));
295  const auto a = 0 == s ? static_cast<AspectSize>(0) : aspect.at(h);
296  const auto f = fuel::FuelType::safeCode(fuel.at(h));
297  const auto cell = Cell(h, s, a, f);
298  values.at(h) = cell;
299 #ifndef NDEBUG
300 #ifndef VLD_RPTHOOK_INSTALL
301  const topo::Location loc{h};
302  const auto r = loc.row();
303  const auto c = loc.column();
304  logging::check_fatal(cell.row() != r, "Cell row %d not %d", cell.row(), r);
305  logging::check_fatal(cell.column() != c, "Cell column %d not %d", cell.column(), c);
306  logging::check_fatal(cell.slope() != s, "Cell slope %d not %d", cell.slope(), s);
307  logging::check_fatal(cell.aspect() != a, "Cell aspect %d not %d", cell.aspect(), a);
308  logging::check_fatal(cell.fuelCode() != f, "Cell fuel %d not %d", cell.fuelCode(), f);
309  const auto v = values.at(h);
310  logging::check_fatal(v.row() != r, "Row %d not %d", v.row(), r);
311  logging::check_fatal(v.column() != c, "Column %d not %d", v.column(), c);
312  logging::check_fatal(v.slope() != s, "Slope %d not %d", v.slope(), s);
313  if (0 != s)
314  {
315  logging::check_fatal(v.aspect() != a, "Aspect %d not %d", v.aspect(), a);
316  }
317  else
318  {
319  logging::check_fatal(v.aspect() != 0, "Aspect %d not %d", v.aspect(), 0);
320  }
321  logging::check_fatal(v.fuelCode() != f, "Fuel %d not %d", v.fuelCode(), f);
322 #endif
323 #endif
324  });
325  return new data::ConstantGrid<Cell>(fuel.cellSize(),
326  fuel.rows(),
327  fuel.columns(),
328  nodata,
329  -1,

332  string(fuel.proj4()),
333  std::move(values));
