478 template <typename T>
479 [[nodiscard]] GridBase read_header(TIFF* tif, GTIF* gtif)
480 {
481  GTIFDefn definition;
482  if (GTIFGetDefn(gtif, &definition))
483  {
484  void* data;
485  uint32 count;
486  int columns;
487  int rows;
488  TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &columns);
489  // logging::check_fatal(columns > numeric_limits<Idx>::max(),
490  // "Cannot use grids with more than %d columns",
491  // numeric_limits<Idx>::max());
492  TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &rows);
493  // logging::check_fatal(rows > numeric_limits<Idx>::max(),
494  // "Cannot use grids with more than %d rows",
495  // numeric_limits<Idx>::max());
496  TIFFGetField(tif, GDAL_NODATA, &count, &data);
497  const auto nodata = stoi(string(static_cast<char*>(data)));
498  double x = 0.0;
499  double y = rows;
500  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
501  "Unable to translate image to PCS coordinates.");
502  const auto yllcorner = y;
503  const auto xllcorner = x;
504  logging::debug("Lower left for header is (%f, %f)", xllcorner, yllcorner);
505  double adf_coefficient[6] = {0};
506  x = 0.5;
507  y = 0.5;
508  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
509  "Unable to translate image to PCS coordinates.");
510  adf_coefficient[4] = x;
511  adf_coefficient[5] = y;
512  x = 1.5;
513  y = 0.5;
514  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
515  "Unable to translate image to PCS coordinates.");
516  const auto cell_width = x - adf_coefficient[4];
517  x = 0.5;
518  y = 1.5;
519  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
520  "Unable to translate image to PCS coordinates.");
521  const auto cell_height = y - adf_coefficient[5];
522  logging::check_fatal(cell_width != -cell_height,
523  "Can only use grids with square pixels");
524  const auto proj4_char = GTIFGetProj4Defn(&definition);
525  auto proj4 = string(proj4_char);
526  delete proj4_char;
527  const auto zone_pos = proj4.find("+zone=");
528  if (string::npos != zone_pos && string::npos != proj4.find("+proj=utm"))
529  {
530  // convert from utm zone to tmerc
531  const auto zone_str = proj4.substr(zone_pos + 6);
532  const auto zone = stoi(zone_str);
533  // zone 15 is -93 and other zones are 6 degrees difference
534  const auto degrees = static_cast<int>(6.0 * (zone - 15.0) - 93);
535  // HACK: assume utm zone is at start
536  proj4 = string(
537  "+proj=tmerc +lat_0=0.000000000 +lon_0=" + to_string(degrees) + ".000000000 +k=0.999600 +x_0=500000.000 +y_0=0.000");
538  }
539  const auto xurcorner = xllcorner + cell_width * columns;
540  const auto yurcorner = yllcorner + cell_width * rows;
541  return GridBase(cell_width,
542  nodata,
543  xllcorner,
544  yllcorner,
545  xurcorner,
546  yurcorner,
547  string(proj4));
548  }
549  throw runtime_error("Cannot read TIFF header");
550 }
551 template <class R>
552 [[nodiscard]] R with_tiff(const string& filename, function<R(TIFF*, GTIF*)> fct)
553 {
554  logging::debug("Reading file %s", filename.c_str());
555  // suppress warnings about geotiff tags that aren't found
556  TIFFSetWarningHandler(nullptr);
557  auto tif = XTIFFOpen(filename.c_str(), "r");
558  logging::check_fatal(!tif, "Cannot open file %s as a TIF", filename.c_str());
559  auto gtif = GTIFNew(tif);
560  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
561  try
562  {
563  R result = fct(tif, gtif);
564  if (tif)
565  XTIFFClose(tif);
566  if (gtif)
567  GTIFFree(gtif);
568  GTIFDeaccessCSV();
569  return result;
570  }
571  catch (std::exception&)
572  {
573  return logging::fatal<R>("Unable to process file %s", filename.c_str());
574  }
575 }
