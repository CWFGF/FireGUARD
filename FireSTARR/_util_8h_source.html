385 template <typename T, typename V>
386 T binary_find(const T lower,
387  const T upper,
388  const double value,
389  const std::function<V(T)>& fct)
390 {
391  const auto mid = lower + (upper - lower) / 2;
392  if (lower == upper)
393  {
394  return lower;
395  }
396  if (fct(mid) < value)
397  {
398  return binary_find(lower, mid, value, fct);
399  }
400  return binary_find(mid + 1, upper, value, fct);
401 }
