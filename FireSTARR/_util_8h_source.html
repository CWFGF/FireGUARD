338 template <typename T, typename V>
339 T binary_find(const T lower,
340  const T upper,
341  const double value,
342  const std::function<V(T)>& fct)
343 {
344  const auto mid = lower + (upper - lower) / 2;
345  if (lower == upper)
346  {
347  return lower;
348  }
349  if (fct(mid) < value)
350  {
351  return binary_find(lower, mid, value, fct);
352  }
353  return binary_find(mid + 1, upper, value, fct);
354 }
