142 protected:
143  /*
144  * Field Natural Range Used Range Bits Bit Range
145  * Row 0 - 2047 0 - 2047 11 0 - 2047
146  * Column 0 - 2047 0 - 2047 11 0 - 2047
147  * PADDING 10
148  * Fuel 0 - 56 0 - 56 6 0 - 63
149  * Aspect 0 - 359 0 - 359 9 0 - 511
150  * Slope 0 - infinity 0 - 127 7 0 - 127
151  * Extra 10
152  *
153  * Rows and Columns are restricted to 2048 since that's what gets clipped out of
154  * the GIS outputs.
155  *
156  * Fuel is tied to how many variations of percent conifer/dead fir we want to use, and
157  * if we want to allow M1/M2/M3/M4 on their own, or just use the ones that are
158  * automatically tied to the green-up.
159  *
160  * Aspect is calculated to be in degrees, so 0 - 359.
161  *
162  * Slope is truncated to 0 - 60 because that's the range that affects effective wind
163  * speed for slopes, but there's an issue with this when it tries to calculate the
164  * horizontal rate of spread since then the slope has been truncated and the distance
165  * calculated will be wrong.
166  */
170  static constexpr uint32 FuelShift = LocationBits + 10;
174  static constexpr uint32 FuelBits = 6;
178  static constexpr Topo FuelBitMask = util::bit_mask<FuelBits, Topo>();
179  static_assert(FuelBitMask == 0x3F);
180  static_assert(FuelBitMask >= NUMBER_OF_FUELS);
184  static constexpr Topo FuelMask = FuelBitMask << FuelShift;
188  static constexpr uint32 AspectShift = FuelBits + FuelShift;
192  static constexpr uint32 AspectBits = 9;
196  static constexpr Topo AspectBitMask = util::bit_mask<AspectBits, Topo>();
197  static_assert(AspectBitMask == 0x1FF);
201  static constexpr Topo AspectMask = AspectBitMask << AspectShift;
205  static constexpr uint32 SlopeShift = AspectBits + AspectShift;
209  static constexpr uint32 SlopeBits = 7;
213  static constexpr Topo SlopeBitMask = util::bit_mask<SlopeBits, Topo>();
214  static_assert(SlopeBitMask == 0x7F);
215  static_assert(SlopeBitMask >= MAX_SLOPE_FOR_FACTOR);
219  static constexpr Topo SlopeMask = SlopeBitMask << SlopeShift;
223  static constexpr Topo CellMask = HashMask | FuelMask | AspectMask | SlopeMask;
224  static_assert(CellMask == 0x3FFFFF003FFFFF);
225 };
