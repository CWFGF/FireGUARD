22 inline double distPtPt(firestarr::sim::InnerPos& a, firestarr::sim::InnerPos& b)
23 {
24  const auto abX = (b.x - a.x);
25  const auto abY = (b.y - a.y);
26  return (abX * abX + abY * abY);
27 }

122 void quickHull(const vector<firestarr::sim::InnerPos>& a, set<firestarr::sim::InnerPos>& hullPoints, firestarr::sim::InnerPos& n1, firestarr::sim::InnerPos& n2)
123 {
124 #ifdef DEBUG_HULL
125  firestarr::logging::warning("Checking %d points", a->size());
126 #endif
127  double maxD = -1.0; //just make sure it's not >= 0
128  firestarr::sim::InnerPos maxPos{std::numeric_limits<double>::min(), std::numeric_limits<double>::min()};
129  vector<firestarr::sim::InnerPos> usePoints{};
130  // worst case scenario
131  usePoints.reserve(a.size());
132 
133  //since we do distLinePt so often, calculate the parts that are always the same
134  const auto abX = (n2.x - n1.x);
135  const auto abY = (n2.y - n1.y);
136  /* so instead of:
137  * return ( (b->x - a->x)*(a->y - p->y) - (a->x - p->x)*(b->y - a->y) );
138  * we can do the equivalent of:
139  * return ( abX*(a->y - p->y) - (a->x - p->x)*abY );
140  * for distance from the line n1n2 to the current point
141  */
142 
143  for (const auto p : a)
144  {
145  //loop through points, looking for furthest
146  const auto d = (abX * (n1.y - p.y) - (n1.x - p.x) * abY);
147  if (d >= 0)
148  {
149  if (d > maxD)
150  { // if further away
151  maxD = d; // update max dist
152  maxPos = p;
153  }
154  // only use in next step if on positive side of line
155 #ifdef DEBUG_HULL
156  firestarr::logging::warning("Adding point (%d, %d) (%f, %f)",
157  p.x,
158  p.y,
159  p.x,
160  p.y);
161 #endif
162  usePoints.emplace_back(p);
163  }
164  }
165  if (maxD == 0)
166  {
167 //we have co-linear points
168 #ifdef DEBUG_HULL
169  size_t before = usePoints->size();
170 #endif
171  usePoints.erase(std::remove(usePoints.begin(), usePoints.end(), maxPos), usePoints.end());
172 #ifdef DEBUG_HULL
173  size_t after = usePoints->size();
174  firestarr::logging::check_fatal(before == after, "Remove did not get rid of point (%d, %d) (%f, %f)", maxPos.x, maxPos.y, maxPos.x, maxPos.y);
175 #endif
176  //need to figure out which direction we're going in
177  const auto d1 = distPtPt(n1, maxPos);
178  const auto d2 = distPtPt(n1, n2);
179 
180  // if either of these isn't true then this must be an edge
181  auto is_not_edge = (d1 < d2) && (distPtPt(maxPos, n2) < d2);
182  if (is_not_edge)
183  {
184  // maxNode is between n1 & n2
185 #ifdef DEBUG_HULL
186  firestarr::logging::check_fatal(usePoints->size() == a->size(), "Recursing without eliminating any points");
187 #endif
188  quickHull(usePoints, hullPoints, n1, maxPos);
189  quickHull(usePoints, hullPoints, maxPos, n2);
190  }
191  //n1 -> n2 must be an edge, but then maxNode is on one side of them
192  else
193  {
194  hullPoints.emplace(n1);
195  hullPoints.emplace(n2);
196  }
197  }
198  else if (maxD < 0)
199  {
200  //no valid points, this must be edge
201  hullPoints.emplace(n1);
202  hullPoints.emplace(n2);
203  }
204  else
205  {
206  //this is not an edge, so recurse on the lines between n1, n2, & maxPos
207 #ifdef DEBUG_HULL
208  size_t before = usePoints->size();
209 #endif
210  usePoints.erase(std::remove(usePoints.begin(), usePoints.end(), maxPos), usePoints.end());
211 #ifdef DEBUG_HULL
212  size_t after = usePoints->size();
213  firestarr::logging::check_fatal(before == after, "Remove did not get rid of point (%d, %d) (%f, %f)", maxPos.x, maxPos.y, maxPos.x, maxPos.y);
214  firestarr::logging::check_fatal(usePoints->size() == a->size(), "Recursing without eliminating any points");
215 #endif
216  quickHull(usePoints, hullPoints, n1, maxPos);
217  quickHull(usePoints, hullPoints, maxPos, n2);
218  }
219 }
