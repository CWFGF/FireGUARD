26 void hull(vector<firestarr::sim::InnerPos>& a) noexcept
27 {
28  vector<firestarr::sim::InnerPos> hullPoints{};
29  firestarr::sim::InnerPos maxPos{MIN_X, MIN_X};
30  firestarr::sim::InnerPos minPos{MAX_X, MAX_X};
31 
32  for (const auto p : a)
33  {
34  if (p.x > maxPos.x)
35  {
36  maxPos = p;
37  }
38  // don't use else if because first point should be set for both
39  if (p.x < minPos.x)
40  {
41  minPos = p;
42  }
43  }
44 
45  //get rid of max & min nodes & call quickhull
46  if (maxPos != minPos)
47  {
48  a.erase(std::remove(a.begin(), a.end(), maxPos), a.end());
49  a.erase(std::remove(a.begin(), a.end(), minPos), a.end());
50  quickHull(a, hullPoints, minPos, maxPos);
51  quickHull(a, hullPoints, maxPos, minPos);
52  // make sure we have unique points
53  std::sort(hullPoints.begin(), hullPoints.end());
54  hullPoints.erase(std::unique(hullPoints.begin(), hullPoints.end()), hullPoints.end());
55  std::swap(a, hullPoints);
56  }
57 }

59 void quickHull(const vector<firestarr::sim::InnerPos>& a,
60  vector<firestarr::sim::InnerPos>& hullPoints,
61  const firestarr::sim::InnerPos& n1,
62  const firestarr::sim::InnerPos& n2) noexcept
63 {
64  double maxD = -1.0; //just make sure it's not >= 0
65  firestarr::sim::InnerPos maxPos{MIN_X, MIN_X};
66  vector<firestarr::sim::InnerPos> usePoints{};
67  // worst case scenario
68  usePoints.reserve(a.size());
69 
70  //since we do distLinePt so often, calculate the parts that are always the same
71  const auto abX = (n2.x - n1.x);
72  const auto abY = (n2.y - n1.y);
73  /* so instead of:
74  * return ( (b->x - a->x)*(a->y - p->y) - (a->x - p->x)*(b->y - a->y) );
75  * we can do the equivalent of:
76  * return ( abX*(a->y - p->y) - (a->x - p->x)*abY );
77  * for distance from the line n1n2 to the current point
78  */
79 
80  for (const auto p : a)
81  {
82  //loop through points, looking for furthest
83  const auto d = (abX * (n1.y - p.y) - (n1.x - p.x) * abY);
84  if (d >= 0)
85  {
86  if (d > maxD)
87  {
88  // if further away
89  if (maxD >= 0)
90  {
91  // already have one, so add old one to the list
92  // NOTE: delayed add instead of erasing maxPos later
93  usePoints.emplace_back(maxPos);
94  }
95  // update max dist
96  maxD = d;
97  maxPos = p;
98  }
99  else
100  {
101  // only use in next step if on positive side of line
102  usePoints.emplace_back(p);
103  }
104  }
105  }
106  if (maxD > 0
107  || (0 == maxD
108  //we have co-linear points
109  // if either of these isn't true then this must be an edge
110  && (distPtPt(n1, maxPos) < distPtPt(n1, n2))
111  && (distPtPt(maxPos, n2) < distPtPt(n1, n2))))
112  {
113  // this is not an edge, so recurse on the lines between n1, n2, & maxPos
114  quickHull(usePoints, hullPoints, n1, maxPos);
115  quickHull(usePoints, hullPoints, maxPos, n2);
116  }
117  else
118  {
119  // n1 -> n2 must be an edge
120  hullPoints.emplace_back(n1);
121  // Must add n2 as the first point of a different line
122  }
123 }
