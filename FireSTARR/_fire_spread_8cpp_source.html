28 SlopeTableArray make_slope_table() noexcept
29 {
30  // HACK: slope can be infinite, but anything > max is the same as max
31  SlopeTableArray result{};
32  for (size_t i = 0; i <= MAX_SLOPE_FOR_FACTOR; ++i)
33  {
34  result.at(i) = exp(3.533 * pow(i / 100.0, 1.2));
35  }
36  static_assert(result.size() == MAX_SLOPE_FOR_DISTANCE + 1);
37  for (size_t i = MAX_SLOPE_FOR_FACTOR + 1; i < result.size(); ++i)
38  {
39  result.at(i) = result.at(MAX_SLOPE_FOR_FACTOR);
40  }
41  return result;
42 }

54 static double calculate_standard_back_isi_wsv(const double v) noexcept
55 {
56  return 0.208 * exp(-0.05039 * v);
57 }
58 static const util::LookupTable<&calculate_standard_back_isi_wsv> STANDARD_BACK_ISI_WSV{};
59 static double calculate_standard_wsv(const double v) noexcept
60 {
61  return v < 40.0
62  ? exp(0.05039 * v)
63  : 12.0 * (1.0 - exp(-0.0818 * (v - 28)));
64 }
65 static const util::LookupTable<&calculate_standard_wsv> STANDARD_WSV{};
66 SpreadInfo::SpreadInfo(const Scenario& scenario,
67  const double time,
68  const topo::Cell& cell,
69  const int nd,
70  const wx::FwiWeather* weather)
71  : cell_(cell),
72  weather_(weather),
73  time_(time),
74  nd_(nd)
75 {
76  max_intensity_ = -1;
77  const auto ffmc_effect = ffmcEffect();
78  // needs to be non-const so we can update if slopeEffect changes direction
79  auto raz = wind().heading();
80  const auto isz = 0.208 * ffmc_effect;
81  const auto slope_azimuth = cell_.aspect();
82  const auto fuel = fuel::check_fuel(cell);
83  const auto has_no_slope = 0 == percentSlope();
84  auto wsv = wind().speed().asDouble();
85  if (!has_no_slope)
86  {
87  const auto isf1 = fuel->calculateIsf(*this, isz);
88  // const auto isf = (0.0 == isf1) ? isz : isf1;
89  // we know const auto isz = 0.208 * ffmc_effect;
90  auto wse = 0.0 == isf1 ? 0 : log(isf1 / isz) / 0.05039;
91  if (wse > 40)
92  {
93  wse = 28.0 - log(
94  1.0 - min(0.999 * 2.496 * ffmc_effect, isf1) / (2.496 * ffmc_effect)) / 0.0818;
95  }
96  const auto heading = util::to_heading(
97  util::to_radians(static_cast<double>(slope_azimuth)));
98  // we know that at->raz is already set to be the wind heading
99  const auto wsv_x = wind().wsvX() + wse * cos(heading);
100  const auto wsv_y = wind().wsvY() + wse * sin(heading);
101  wsv = sqrt(wsv_x * wsv_x + wsv_y * wsv_y);
102  raz = acos(wsv_y / wsv);
103  if (wsv_x < 0)
104  {
105  raz = util::RAD_360 - raz;
106  }
107  }
108  const auto isi = isz * STANDARD_WSV(wsv);
109  const auto bui_eff = fuel->buiEffect(bui().asDouble());
110  head_ros_ = fuel->calculateRos(nd,
111  *weather,
112  isi) * bui_eff;
113  const auto min_ros = Settings::minimumRos();
114  if (min_ros > head_ros_)
115  {
116  head_ros_ = -1;
117  return;
118  }
119  const auto sfc = fuel->surfaceFuelConsumption(*this);
120  const auto critical_surface_intensity = fuel->criticalSurfaceIntensity(*this);
121  const auto rso = fuel::FuelType::criticalRos(sfc, critical_surface_intensity);
122  const auto is_crown = fuel::FuelType::isCrown(critical_surface_intensity,
123  fuel::fire_intensity(sfc, head_ros_));
124  const auto back_isi = ffmc_effect * STANDARD_BACK_ISI_WSV(wsv);
125  auto back_ros = fuel->calculateRos(nd,
126  *weather,
127  back_isi) * bui_eff;
128  if (is_crown)
129  {
130  head_ros_ = fuel->finalRos(*this,
131  isi,
132  fuel->crownFractionBurned(head_ros_, rso),
133  head_ros_);
134  back_ros = fuel->finalRos(*this,
135  back_isi,
136  fuel->crownFractionBurned(back_ros, rso),
137  back_ros);
138  }
139  // do everything we can to avoid calling trig functions unnecessarily
140  const auto b_semi = has_no_slope ? 0 : cos(atan(percentSlope() / 100.0));
141  const auto slope_radians = util::to_radians(slope_azimuth);
142  // do check once and make function just return 1.0 if no slope
143  const auto no_correction = [](const double) noexcept { return 1.0; };
144  const auto do_correction = [b_semi, slope_radians](const double theta) noexcept
145  {
146  // never gets called if isInvalid() so don't check
147  // figure out how far the ground distance is in map distance horizontally
148  const auto angle_unrotated = theta - slope_radians;
149  const auto tan_u = tan(angle_unrotated);
150  const auto y = b_semi / sqrt(b_semi * tan_u * (b_semi * tan_u) + 1.0);
151  const auto x = y * tan_u;
152  return sqrt(x * x + y * y);
153  };
154  const auto correction_factor = has_no_slope
155  ? std::function<double(double)>(no_correction)
156  : std::function<double(double)>(do_correction);
157  const auto cell_size = scenario.cellSize();
158  const auto add_offset = [this, cell_size, min_ros](const double direction,
159  const double ros)
160  {
161  if (ros < min_ros)
162  {
163  return false;
164  }
165  // spreading, so figure out offset from current point
166  const auto ros_cell = ros / cell_size;
167  offsets_.emplace_back(ros_cell * sin(direction), ros_cell * cos(direction));
168  return true;
169  };
170  const auto threshold = scenario.spreadThresholdByRos(time_);
171  // if not over spread threshold then don't spread
172  // HACK: assume there is no fuel where a crown fire's sfc is < COMPARE_LIMIT but it's fc is >
173  double ros{};
174  // HACK: set ros in boolean if we get that far so we don't have to repeat the if body
175  if (head_ros_ < threshold
176  || sfc < COMPARE_LIMIT
177  || !add_offset(raz, ros = (head_ros_ * correction_factor(raz))))
178  {
179  // mark as invalid
180  head_ros_ = -1;
181  return;
182  }
183  auto fc = sfc;
184  // don't need to re-evaluate if crown with new head_ros_ because it would only go up if is_crown
185  if (fuel->canCrown() && is_crown)
186  {
187  // wouldn't be crowning if ros is 0 so that's why this is in an else
188  fc += fuel->crownConsumption(fuel->crownFractionBurned(head_ros_, rso));
189  }
190  // max intensity should always be at the head
191  max_intensity_ = fuel::fire_intensity(fc, ros);
192  const auto a = (head_ros_ + back_ros) / 2.0;
193  const auto c = a - back_ros;
194  const auto flank_ros = a / fuel->lengthToBreadth(wsv);
195  const auto a_sq = a * a;
196  const auto flank_ros_sq = flank_ros * flank_ros;
197  const auto a_sq_sub_c_sq = a_sq - (c * c);
198  const auto ac = a * c;
199  const auto calculate_ros =
200  [a, c, ac, flank_ros, a_sq, flank_ros_sq, a_sq_sub_c_sq](const double theta) noexcept
201  {
202  const auto cos_t = cos(theta);
203  const auto cos_t_sq = cos_t * cos_t;
204  const auto f_sq_cos_t_sq = flank_ros_sq * cos_t_sq;
205  // 1.0 = cos^2 + sin^2
206  const auto sin_t_sq = 1.0 - cos_t_sq;
207  return abs((a * ((flank_ros * cos_t *
208  sqrt(f_sq_cos_t_sq + a_sq_sub_c_sq * sin_t_sq) - ac * sin_t_sq)
209  / (f_sq_cos_t_sq + a_sq * sin_t_sq)) + c) / cos_t);
210  };
211  const auto add_offsets =
212  [&correction_factor, &add_offset, raz, threshold](
213  const double angle_radians,
214  const double ros_flat)
215  {
216  if (ros_flat < threshold)
217  {
218  return false;
219  }
220  auto direction = util::fix_radians(angle_radians + raz);
221  // spread is symmetrical across the center axis, but needs to be adjusted if on a slope
222  // intentionally don't use || because we want both of these to happen all the time
223  auto added = add_offset(direction, ros_flat * correction_factor(direction));
224  direction = util::fix_radians(raz - angle_radians);
225  added |= add_offset(direction, ros_flat * correction_factor(direction));
226  return added;
227  };
228  const auto add_offsets_calc_ros =
229  [&add_offsets, &calculate_ros](const double angle_radians)
230  {
231  return add_offsets(angle_radians, calculate_ros(angle_radians));
232  };
233  // HACK: rely on && to stop when first ros is too low
234  if (add_offsets_calc_ros(util::to_radians(10))
235  && add_offsets_calc_ros(util::to_radians(20))
236  && add_offsets_calc_ros(util::to_radians(30))
237  && add_offsets_calc_ros(util::to_radians(60))
238  && add_offsets(util::to_radians(90.0), flank_ros * sqrt(a_sq_sub_c_sq) / a)
239  && add_offsets_calc_ros(util::to_radians(120))
240  && add_offsets_calc_ros(util::to_radians(150)))
241  {
242  //only use back ros if every other angle is spreading since this should be lowest
243  // 180
244  if (back_ros < threshold)
245  {
246  return;
247  }
248  const auto direction = util::fix_radians(util::RAD_180 + raz);
249  static_cast<void>(!add_offset(direction, back_ros * correction_factor(direction)));
250  }
251 }
