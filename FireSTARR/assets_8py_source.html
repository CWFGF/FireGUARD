28 def getZone(spatialReference):
29  """!
30  Determine UTM zone for given spatial reference
31  @param spatialReference Spatial reference to determine zone from
32  @return UTM zone represented by spatial reference
33  """
34  r = spatialReference.exporttostring()
35  r = r[r.index('Central_Meridian'):]
36  r = r[r.index(',') + 1:r.index(']') - 1]
37  c = float(r)
38  return (c + 183.0) / 6.0
