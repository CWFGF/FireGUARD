32 Environment Environment::load(const Point& point,
33  const string& in_fuel,
34  const string& in_elevation)
35 {
36  logging::note("Fuel raster is %s", in_fuel.c_str());
37  if (sim::Settings::runAsync())
38  {
39  logging::debug("Loading grids async");
40  auto fuel = async(launch::async,
41  [&in_fuel, &point]()
42  {
43  logging::info("Loading %s", in_fuel.c_str());
44  return FuelGrid::readTiff(in_fuel, point, sim::Settings::fuelLookup());
45  });
46  auto elevation = async(launch::async,
47  [&in_elevation, &point]()
48  {
49  logging::info("Loading %s", in_elevation.c_str());
50  return ElevationGrid::readTiff(in_elevation, point);
51  });
52  logging::debug("Waiting for grids");
53  return Environment(*unique_ptr<FuelGrid>(fuel.get()),
54  *unique_ptr<ElevationGrid>(elevation.get()),
55  point);
56  }
57  logging::warning("Loading grids synchronously");
58  // HACK: need to copy strings since closures do that above
59  return Environment(*unique_ptr<FuelGrid>(
60  FuelGrid::readTiff(string(in_fuel), point, sim::Settings::fuelLookup())),
61  *unique_ptr<ElevationGrid>(
62  ElevationGrid::readTiff(string(in_elevation), point)),
63  point);
64 }

82 Environment Environment::loadEnvironment(const string& path,
83  const Point& point,
84  const string& perimeter,
85  const int year)
86 {
87  logging::note("Using ignition point (%f, %f)", point.latitude(), point.longitude());
88  logging::info("Running using inputs directory '%s'", path.c_str());
89  auto rasters = util::find_rasters(path, year);
90  auto best_x = numeric_limits<double>::max();
91  auto best_y = numeric_limits<FullIdx>::max();
92  unique_ptr<const EnvironmentInfo> env_info = nullptr;
93  unique_ptr<data::GridBase> for_info = nullptr;
94  if (!perimeter.empty())
95  {
96  for_info = make_unique<data::GridBase>(data::read_header<unsigned char>(perimeter));
97  logging::info("Perimeter projection is %s", for_info->proj4().c_str());
98  }
99  for (const auto& raster : rasters)
100  {
101  auto fuel = raster;
102  // make sure we're using a consistent directory separator
103  std::replace(fuel.begin(), fuel.end(), '\\', '/');
104  // HACK: assume there's only one instance of 'fuel' in the file name we want to change
105  const auto find_what = string("fuel");
106  const auto find_len = find_what.length();
107  const auto find_start = fuel.find(find_what, fuel.find_last_of('/'));
108  const auto elevation = string(fuel).replace(find_start, find_len, "dem");
109  unique_ptr<const EnvironmentInfo> cur_info = EnvironmentInfo::loadInfo(
110  fuel,
111  elevation);
112  const auto cur_x = abs(point.longitude() - cur_info->meridian());
113  logging::verbose("Zone %0.1f meridian is %0.2f degrees from point",
114  cur_info->zone(),
115  cur_x);
116  // HACK: assume floating point is going to always be exactly the same result
117  if ((nullptr == for_info || cur_info->meridian() == for_info->meridian())
118  && cur_x <= best_x)
119  {
120  logging::debug("SWITCH X");
121  if (cur_x != best_x)
122  {
123  // if we're switching zones then we need to reset this
124  best_y = numeric_limits<FullIdx>::max();
125  }
126  best_x = cur_x;
127  // overwrite should delete current pointer
128  const auto coordinates = cur_info->findFullCoordinates(point, false);
129  if (nullptr != coordinates)
130  {
131  logging::debug("CHECK Y");
132  const auto cur_y = static_cast<FullIdx>(abs(
133  std::get<0>(*coordinates) - cur_info->calculateRows() / static_cast<FullIdx>(2)));
134  logging::debug(("Current y value is " + std::to_string(cur_y)).c_str());
135  if (cur_y < best_y)
136  {
137  logging::debug("SWITCH Y");
138  env_info = std::move(cur_info);
139  best_y = cur_y;
140  }
141  }
142  else
143  {
144  logging::debug("NULLPTR");
145  }
146  }
147  }
148  logging::check_fatal(nullptr == env_info,
149  "Could not find an environment to use for (%f, %f)",
150  point.latitude(),
151  point.longitude());
152  logging::debug("Best match for (%f, %f) is zone %0.2f with a meridian of %0.2f",
153  point.latitude(),
154  point.longitude(),
155  env_info->zone(),
156  env_info->meridian());
157  logging::note("Projection is %s", env_info->proj4().c_str());
158  // envInfo should get deleted automatically because it uses unique_ptr
159  return env_info->load(point);
160 }
