92 def save_http(to_dir, url, save_as=None, mode='wb', ignore_existing=False):
93  """!
94  Save file at given URL into given directory using an HTTP connection
95  @param to_dir Directory to save to
96  @param url URL to save from
97  @param save_as File name to save to
98  @param mode Mode to use while saving
99  @param ignore_existing Whether or not to save if file already exists
100  @return Path that file was saved to
101  """
102  logging.debug("Saving {}".format(url))
103  if save_as is None:
104  save_as = os.path.join(to_dir, os.path.basename(url))
105  if ignore_existing and os.path.exists(save_as):
106  logging.debug('Ignoring existing file')
107  return save_as
108  ensure_dir(to_dir)
109  # we want to keep modified times matching on both ends
110  do_save = True
111  response = urllib2.urlopen(url)
112  modlocal = None
113  if 'last-modified' in response.headers.keys():
114  mod = response.headers['last-modified']
115  modtime = dateutil.parser.parse(mod)
116  modlocal = fix_timezone_offset(modtime)
117  # if file exists then compare mod times
118  if os.path.isfile(save_as):
119  filetime = os.path.getmtime(save_as)
120  filedatetime = datetime.datetime.fromtimestamp(filetime)
121  do_save = modlocal != filedatetime
122  # NOTE: need to check file size too? Or should it not matter because we
123  # only change the timestamp after it's fully written
124  if do_save:
125  logging.info("Downloading {}".format(save_as))
126  try:
127  with open(save_as, mode) as f:
128  f.write(response.read())
129  except:
130  try_remove(save_as)
131  raise
132  if modlocal is not None:
133  tt = modlocal.timetuple()
134  usetime = time.mktime(tt)
135  os.utime(save_as, (usetime, usetime))
136  return save_as

138 def save_ftp(to_dir, url, user="anonymous", password=""):
139  """!
140  Save file at given URL into given directory using an FTP connection
141  @param to_dir Directory to save to
142  @param url URL to save from
143  @param user User name for FTP login
144  @param password Password for FTP login
145  @return Path that file was saved to
146  """
147  urlp = urlparse(url)
148  folder = os.path.dirname(urlp.path)
149  site = urlp.netloc
150  filename = os.path.basename(urlp.path)
151  logging.debug("Saving {}".format(filename))
152  import ftplib
153  ftp = ftplib.FTP(site)
154  ftp.login(user, password)
155  ftp.cwd(folder)
156  save_as = os.path.join(to_dir, filename)
157  do_save = True
158  ftptime = ftp.sendcmd('MDTM {}'.format(filename))
159  ftpdatetime = datetime.datetime.strptime(ftptime[4:], '%Y%m%d%H%M%S')
160  ftplocal = fix_timezone_offset(ftpdatetime)
161  # if file exists then compare mod times
162  if os.path.isfile(save_as):
163  filetime = os.path.getmtime(save_as)
164  filedatetime = datetime.datetime.fromtimestamp(filetime)
165  do_save = ftplocal != filedatetime
166  # NOTE: need to check file size too? Or should it not matter because we
167  # only change the timestamp after it's fully written
168  if do_save:
169  logging.debug("Downloading {}".format(filename))
170  with open(save_as, 'wb') as f:
171  ftp.retrbinary('RETR {}'.format(filename), f.write)
172  tt = ftplocal.timetuple()
173  usetime = time.mktime(tt)
174  os.utime(save_as, (usetime, usetime))
175  return save_as

303 def fixK(s):
304  """!
305  Fix precision being shown for a value
306  @param s Value to format
307  @return String formatted to magnitude of it
308  """
309  if s >= 100000000000:
310  return '{:,}B'.format(int(s / 1000000000.0))
311  if s >= 1000000000:
312  return '{:1.1f}B'.format(s / 1000000000.0)
313  if s >= 100000000:
314  return '{:1.0f}M'.format(s / 1000000.0)
315  if s >= 1000000:
316  return '{:1.1f}M'.format(s / 1000000.0)
317  if s >= 100000:
318  return '{:1.0f}K'.format(s / 1000.0)
319  if s >= 1000:
320  return '{:1.1f}K'.format(s / 1000.0)
321  if s < 1:
322  return '< 1'
323  if s < 10:
324  return '{}'.format(int(round(s, 0)))
325  return str(int(s))
