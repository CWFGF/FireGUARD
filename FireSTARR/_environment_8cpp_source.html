32 template <class V, class T = V>
33 static V read_tiff_point(TIFF* tif,
34  GTIF* gtif,
35  const Point& point,
36  std::function<T(int, int)> convert)
37 {
38  const data::GridBase grid_info = data::read_header<T>(tif, gtif);
39  const auto coordinates = grid_info.findCoordinates(point, false);
40  auto min_column = max(static_cast<Idx>(0),
41  static_cast<Idx>(std::get<1>(*coordinates) - MAX_COLUMNS /
42  static_cast<Idx>(2)));
43  if (min_column + MAX_COLUMNS >= grid_info.columns())
44  {
45  min_column = grid_info.columns() - MAX_COLUMNS;
46  }
47  auto min_row = max(static_cast<Idx>(0),
48  static_cast<Idx>(std::get<0>(*coordinates) - MAX_COLUMNS /
49  static_cast<Idx>(2)));
50  if (min_row + MAX_COLUMNS >= grid_info.rows())
51  {
52  min_row = grid_info.rows() - MAX_COLUMNS;
53  }
54  int tile_width;
55  int tile_length;
56  TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
57  TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
58  const auto buf = _TIFFmalloc(TIFFTileSize(tif));
59  const tsample_t smp{};
60  const auto mid = MAX_COLUMNS / 2;
61  const auto h = (min_row + mid) / tile_length * tile_length;
62  const auto w = (min_column + mid) / tile_width * tile_width;
63  const auto f_y = h - min_row - (h - min_row) / tile_length * tile_length;
64  const auto y = static_cast<Idx>(tile_length - f_y);
65  const auto f_x = w - min_column - (w - min_column) / tile_width * tile_width;
66  const auto x = static_cast<Idx>(tile_width - f_x);
67  TIFFReadTile(tif, buf, static_cast<uint32>(w), static_cast<uint32>(h), 0, smp);
68  const auto offset = y * tile_length + x;
69  V cur = *(static_cast<V*>(buf) + offset);
70  _TIFFfree(buf);
71  return convert(cur, grid_info.nodata());
72 }

74 static V read_tiff_point(const string& filename,
75  const Point& point,
76  std::function<T(int, int)> convert)
77 {
78  return data::with_tiff<V>(filename,
79  [&convert, &point](TIFF* tif, GTIF* gtif)

90 Environment Environment::load(const fuel::FuelLookup& lookup,
91  const Point& point,
92  const string& in_fuel,
93  const string& in_slope,
94  const string& in_aspect,
95  const string& in_elevation)

99  {
100  auto fuel = async(launch::async,
101  [&in_fuel, &point, lookup]()
102  {
103  return FuelGrid::readTiff(in_fuel, point, lookup);
104  });
105  auto slope = async(launch::async,
106  [&in_slope, &point]()
107  {
108  return SlopeGrid::readTiff(in_slope, point);
109  });
110  auto aspect = async(launch::async,
111  [&in_aspect, &point]()
112  {
113  return AspectGrid::readTiff(in_aspect, point);
114  });
115  auto elevation = async(launch::async,
116  [&in_elevation, &point]()
117  {
118  return read_tiff_point<ElevationSize>(
119  in_elevation,
120  point);
121  });
122  return Environment(*unique_ptr<FuelGrid>(fuel.get()),
123  *unique_ptr<SlopeGrid>(slope.get()),
124  *unique_ptr<AspectGrid>(aspect.get()),
125  elevation.get());
126  }
127  // HACK: need to copy strings since closures do that above
128  return Environment(*unique_ptr<FuelGrid>(
129  FuelGrid::readTiff(string(in_fuel), point, lookup)),
130  *unique_ptr<SlopeGrid>(SlopeGrid::readTiff(string(in_slope), point)),
131  *unique_ptr<AspectGrid>(
132  AspectGrid::readTiff(string(in_aspect), point)),
133  read_tiff_point<ElevationSize>(string(in_elevation), point));
134 }

152 Environment Environment::loadEnvironment(const fuel::FuelLookup& lookup,
153  const string& path,
154  const Point& point,
155  const string& perimeter,
156  const int year)
157 {
158  logging::note("Using ignition point (%f, %f)", point.latitude(), point.longitude());
159  logging::info("Running using inputs directory '%s'", path.c_str());
160  auto rasters = util::find_rasters(path, year);
161  auto best_x = numeric_limits<double>::max();
162  auto best_y = numeric_limits<Idx>::max();
163  unique_ptr<const EnvironmentInfo> env_info = nullptr;
164  unique_ptr<data::GridBase> for_info = nullptr;
165  if (!perimeter.empty())
166  {
167  for_info = make_unique<data::GridBase>(data::read_header<unsigned char>(perimeter));
168  logging::info("Perimeter projection is %s", for_info->proj4().c_str());
169  }
170  for (const auto& raster : rasters)
171  {
172  auto fuel = raster;
173  // make sure we're using a consistent directory separator
174  std::replace(fuel.begin(), fuel.end(), '\\', '/');
175  // HACK: assume there's only one instance of 'fuel' in the file name we want to change
176  const auto find_what = string("fuel");
177  const auto find_len = find_what.length();
178  const auto find_start = fuel.find(find_what, fuel.find_last_of('/'));
179  const auto aspect = string(fuel).replace(find_start, find_len, "aspect");
180  const auto slope = string(fuel).replace(find_start, find_len, "slope");
181  const auto elevation = string(fuel).replace(find_start, find_len, "dem");
182  unique_ptr<const EnvironmentInfo> cur_info = EnvironmentInfo::loadInfo(
183  fuel,
184  slope,
185  aspect,
186  elevation);
187  const auto cur_x = abs(point.longitude() - cur_info->meridian());
188  logging::verbose("Zone %0.1f meridian is %0.2f degrees from point",
189  cur_info->zone(),
190  cur_x);
191  // HACK: assume floating point is going to always be exactly the same result
192  if ((nullptr == for_info || cur_info->meridian() == for_info->meridian())
193  && cur_x <= best_x)
194  {
195  logging::debug("SWITCH X");
196  if (cur_x != best_x)
197  {
198  // if we're switching zones then we need to reset this
199  best_y = numeric_limits<Idx>::max();
200  }
201  best_x = cur_x;
202  // overwrite should delete current pointer
203  const auto coordinates = cur_info->findCoordinates(point, false);
204  if (nullptr != coordinates)
205  {
206  logging::debug("CHECK Y");
207  const auto cur_y = static_cast<Idx>(abs(
208  std::get<0>(*coordinates) - cur_info->rows() / static_cast<Idx>(2)));
209  logging::debug(("Current y value is " + std::to_string(cur_y)).c_str());
210  if (cur_y < best_y)
211  {
212  logging::debug("SWITCH Y");
213  env_info = std::move(cur_info);
214  best_y = cur_y;
215  }
216  }
217  else
218  {
219  logging::debug("NULLPTR");
220  }
221  }
222  }
223  logging::check_fatal(nullptr == env_info,
224  "Could not find an environment to use for (%f, %f)",
225  point.latitude(),
226  point.longitude());
227  logging::debug("Best match for (%f, %f) is zone %0.2f with a meridian of %0.2f",
228  point.latitude(),
229  point.longitude(),
230  env_info->zone(),
231  env_info->meridian());
232  logging::note("Projection is %s", env_info->proj4().c_str());
233  // envInfo should get deleted automatically because it uses unique_ptr
234  return env_info->load(lookup, point);
235 }
