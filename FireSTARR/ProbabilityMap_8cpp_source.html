56 void ProbabilityMap::addProbabilities(const ProbabilityMap& rhs)
57 {
58  logging::check_fatal(rhs.time_ != time_, "Wrong time");
59  logging::check_fatal(rhs.start_time_ != start_time_, "Wrong start time");
60  logging::check_fatal(rhs.min_value_ != min_value_, "Wrong min value");
61  logging::check_fatal(rhs.max_value_ != max_value_, "Wrong max value");
62  logging::check_fatal(rhs.low_max_ != low_max_, "Wrong low max value");
63  logging::check_fatal(rhs.med_max_ != med_max_, "Wrong med max value");
64  lock_guard<mutex> lock(mutex_);
65  if (Settings::saveIntensity())
66  {
67  for (auto&& kv : rhs.low_.data)
68  {
69  low_.data[kv.first] += kv.second;
70  }
71  for (auto&& kv : rhs.med_.data)
72  {
73  med_.data[kv.first] += kv.second;
74  }
75  for (auto&& kv : rhs.high_.data)
76  {
77  high_.data[kv.first] += kv.second;
78  }
79  }
80  for (auto&& kv : rhs.all_.data)
81  {
82  all_.data[kv.first] += kv.second;
83  }
84  for (auto size : rhs.sizes_)
85  {
86  static_cast<void>(util::insert_sorted(&sizes_, size));
87  }
88 }
89 void ProbabilityMap::addProbability(const IntensityMap& for_time)
90 {
91  lock_guard<mutex> lock(mutex_);
92  std::for_each(
93  for_time.cbegin(),
94  for_time.cend(),
95  [this](auto&& kv)
96  {
97  const auto k = kv.first;
98  const auto v = kv.second;
99  all_.data[k] += 1;
100  if (Settings::saveIntensity())
101  {
102  if (v >= min_value_ && v <= low_max_)
103  {
104  low_.data[k] += 1;
105  }
106  else if (v > low_max_ && v <= med_max_)
107  {
108  med_.data[k] += 1;
109  }
110  else if (v > med_max_ && v <= max_value_)
111  {
112  high_.data[k] += 1;
113  }
114  else
115  {
116  logging::fatal("Value %d doesn't fit into any range", v);
117  }
118  }
119  });
120  const auto size = for_time.fireSize();
121  static_cast<void>(util::insert_sorted(&sizes_, size));
122 }

166 void ProbabilityMap::saveAll(const Model& model,
167  const tm& start_time,
168  const bool for_actuals,
169  const double time,
170  const double start_day) const
171 {
172  auto t = start_time;
173  auto ticks = mktime(&t);
174  const auto day = static_cast<int>(round(time));
175  ticks += (static_cast<size_t>(day) - t.tm_yday - 1) * DAY_SECONDS;
176  t = *localtime(&ticks);
177  // t.tm_yday = day + 1;
178  TIMESTAMP_STRUCT for_time{};
179  util::to_ts(t, &for_time);
180  const auto make_string = [&for_time, &day](const char* name)
181  {
182  constexpr auto mask = "%s_%03d_%04d-%02d-%02d";
183  static constexpr size_t OutLength = 100;
184  char tmp[OutLength];
185  sprintf(tmp,
186  mask,
187  name,
188  day,
189  for_time.year,
190  for_time.month,
191  for_time.day);
192  return string(tmp);
193  };
194  if (sim::Settings::runAsync())
195  {
196  vector<std::future<void>> results{};
197  if (Settings::saveProbability())
198  {
199  results.push_back(async(launch::async,
200  &ProbabilityMap::saveTotal,
201  this,
202  make_string(for_actuals ? "actuals" : "probability")));
203  }
204  if (Settings::saveOccurrence())
205  {
206  results.push_back(async(launch::async,
207  &ProbabilityMap::saveTotalCount,
208  this,
209  make_string("occurrence")));
210  }
211  if (Settings::saveIntensity())
212  {
213  results.push_back(async(launch::async,
214  &ProbabilityMap::saveLow,
215  this,
216  make_string("intensity_L")));
217  results.push_back(async(launch::async,
218  &ProbabilityMap::saveModerate,
219  this,
220  make_string("intensity_M")));
221  results.push_back(async(launch::async,
222  &ProbabilityMap::saveHigh,
223  this,
224  make_string("intensity_H")));
225  }
226  results.push_back(async(launch::async,
227  &ProbabilityMap::saveSizes,
228  this,
229  make_string("sizes")));
230  for (auto& result : results)
231  {
232  result.wait();
233  }
234  }
235  else
236  {
237  if (Settings::saveProbability())
238  {
239  saveTotal(make_string(for_actuals ? "actuals" : "probability"));
240  }
241  if (Settings::saveOccurrence())
242  {
243  saveTotalCount(make_string("occurrence"));
244  }
245  if (Settings::saveIntensity())
246  {
247  saveLow(make_string("intensity_L"));
248  saveModerate(make_string("intensity_M"));
249  saveHigh(make_string("intensity_H"));
250  }
251  saveSizes(make_string("sizes"));
252  }
253  const auto nd = model.nd(day);
254  logging::note("Fuels for day %d are %s green-up and grass has %d%% curing",
255  day - static_cast<int>(start_day),
256  fuel::calculate_is_green(nd) ? "after" : "before",
257  fuel::calculate_grass_curing(nd));
258 }
