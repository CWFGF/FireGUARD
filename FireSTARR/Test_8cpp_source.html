160  static const auto Latitude = 49.3911;
161  static const auto Longitude = -84.7395;
162  static const topo::StartPoint ForPoint(Latitude, Longitude);
163  const auto start_date = 123;
164  const auto end_date = start_date + static_cast<double>(num_hours) / DAY_HOURS;
165  util::make_directory_recursive(output_directory);
166  const auto fuel = Settings::fuelLookup().byName(fuel_name);
167  auto values = vector<topo::Cell>();
168  // values.reserve(static_cast<size_t>(MAX_ROWS) * MAX_COLUMNS);

170  {
171  for (Idx c = 0; c < MAX_COLUMNS; ++c)
172  {
173  values.emplace_back(r, c, slope, aspect, fuel::FuelType::safeCode(fuel));
174  }
175  }
176  const auto cells = new data::ConstantGrid<topo::Cell>{
177  TEST_GRID_SIZE,
178  MAX_ROWS,
179  MAX_COLUMNS,
180  topo::Cell{},
181  -1,
182  TEST_XLLCORNER,
183  TEST_YLLCORNER,
184  TEST_XLLCORNER + TEST_GRID_SIZE * MAX_COLUMNS,
185  TEST_YLLCORNER + TEST_GRID_SIZE * MAX_ROWS,
186  TEST_PROJ4,
187  std::move(values)};
188  TestEnvironment env(cells);
189  const Location start_location(static_cast<Idx>(MAX_ROWS / 2),
190  static_cast<Idx>(MAX_COLUMNS / 2));
191  Model model(ForPoint, &env);
192  const auto start_cell = make_shared<topo::Cell>(model.cell(start_location));
193  TestWeather weather(fuel, start_date, dc, bui, dmc, ffmc, wind);
194  TestScenario scenario(&model, start_cell, ForPoint, start_date, end_date, &weather);
195  auto info = SpreadInfo(scenario,
196  start_date,
197  *start_cell,
198  model.nd(start_date),
199  weather.at(start_date));
200  map<double, ProbabilityMap*> probabilities{};
201  logging::debug("Starting simulation");
202  // NOTE: don't want to reset first because TestScenario handles what that does
203  scenario.run(&probabilities);
204  scenario.saveObservers("");
205  logging::note("Final Size: %0.0f, ROS: %0.2f",
206  scenario.currentFireSize(),
207  info.headRos());
208  return 0;
209 }
