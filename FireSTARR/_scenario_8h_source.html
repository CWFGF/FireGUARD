399  const topo::Cell& cell,
400  const double time_at_location) const
401  {
402  const auto wx = weather_->at(time);
403  // use mike's table
404  const auto mc = wx->mcDmcPct();
405  if (100 > mc
406  || 109 >= mc && 5 > time_at_location
407  || 119 >= mc && 4 > time_at_location
408  || 131 >= mc && 3 > time_at_location
409  || 145 >= mc && 2 > time_at_location
410  || 218 >= mc && 1 > time_at_location)
411  {
412  return true;
413  }
414  // we can look by fuel type because the entire landscape shares the weather
415  return extinctionThreshold(time) < weather_->survivalProbability(
416  time,
417  cell.fuelCode());
418  }
