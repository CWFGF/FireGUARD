68 def applyExtent(extent, theMXD, spatialReference):
69  """!
70  Apply extent to data frame in map
71  @param extent Extent to apply
72  @param theMXD mxd to apply to data frame of
73  @param spatialReference Spatial reference of extent
74  @return None
75  """
76  df = arcpy.mapping.ListDataFrames(theMXD, "Layers")[0]
77  df.spatialReference = spatialReference
78  df.extent = extent
79  #set scale based on predetermined list
80  s = 0
81  # HACK: round so that if applying extent makes it really close then it should still work
82  while s < len(SCALES) - 1 and round(df.scale / 1000.0, 0) > round(SCALES[s] / 1000.0, 0):
83  s += 1
84  df.scale = SCALES[s]
85  df.panToExtent(extent)
86  inset = arcpy.mapping.ListDataFrames(theMXD, "Inset")[0]
87  inset.spatialReference = spatialReference

129 def setDayCommon(i, is_actuals, theMXD, run_output, fire, perim, extent):
130  """!
131  Set common attributes on maps that are for a specific day
132  @param i Index of date that map is for
133  @param is_actuals Whether or not this is for actual weather
134  @param theMXD mxd to apply settings to
135  @param run_output Folder that simulation outputs are in
136  @param fire Fire that simulation was for
137  @param perim Perimeter to show on map
138  @param extent Extent to apply to map
139  @return Extent after applying to map
140  @return Probability contour that was used
141  """
142  prefix = 'actuals_' if is_actuals else 'wxshield_'
143  fire_prefix = fire + "_" + ('actual_' if is_actuals else '')
144  probs = [x for x in os.listdir(os.path.join(Settings.HOME_DIR, run_output)) if x.startswith(prefix) and x[-3:] == "asc"]
145  day0 = find_day(probs[0]) - 1
146  jds = map(find_day, probs)
147  dates = map(find_date, probs)
148  days = map(lambda x: x - day0, jds)
149  sim_output = readSimOutput(run_output)
150  txtSize, txtAssumptions = getSizeAndAssumptions(i, sim_output, days, dates)
151  map_output = getMapOutput(run_output)
152  f = os.path.join(run_output, probs[i].replace(prefix, 'sizes_').replace('.asc', '.csv'))
153  size_graph = makeSizeGraphName(f)
154  setSourceImage(theMXD, "graph_stats", size_graph)
155  prob_source = os.path.join(map_output, probs[i].replace(".asc", "_class_poly.shp").replace("-", "_"))
156  low_prob = setDataSource(theMXD, "Low Probability", prob_source)
157  gtr10 = setDataSource(theMXD, "> 10%", prob_source)
158  df = arcpy.mapping.ListDataFrames(theMXD, "Layers")[0]
159  df.spatialReference = arcpy.Describe(prob_source).spatialReference
160  if not extent:
161  # only set extent based on final map so that they all cover the same area
162  #~ extent = gtr10.getSelectedExtent()
163  extent = low_prob.getSelectedExtent()
164  # HACK: do this here so the same extent gets applied to subsequent maps
165  applyExtent(extent, theMXD, df.spatialReference)
166  extent = df.extent
167  setCommon(theMXD,
168  fire + "\n" + dates[i] + ' - Day ' + str(days[i]),
169  txtAssumptions,
170  txtSize,
171  run_output,
172  fire,
173  perim,
174  extent,
175  prob_source)
176  return df.extent, probs[i]
