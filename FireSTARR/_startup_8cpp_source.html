46 Startup read_startup(util::Database* db)
47 {
48  // HACK: do it this way so that we know database calls are in this order
49  const auto station = db->getString();
50  const auto gen_utc = db->getTimestamp();
51  tm utc{};
52  util::to_tm_gm(gen_utc, &utc);
53  const auto local_t = mktime(&utc);
54  tm local{};
55  localtime_s(&local, &local_t);
56  TIMESTAMP_STRUCT result;
57  util::to_ts(local, &result);
58  const auto generated = result;
59  const auto latitude = db->getDouble();
60  const auto longitude = db->getDouble();
61  const auto distance_from = db->getDouble<0>();
62  const Ffmc ffmc(db->getDouble<2>());
63  const Dmc dmc(db->getDouble<2>());
64  const Dc dc(db->getDouble<2>());
65  const AccumulatedPrecipitation apcp_0800(db->getDouble<2>());
66  return Startup(station,
67  generated,
68  topo::Point(latitude, longitude),
69  distance_from,
70  ffmc,
71  dmc,
72  dc,
73  apcp_0800,
74  false);
75 }
