36  BurnedMap(const Grid<unsigned char, unsigned char>& perim_grid, const Environment& env)
37  : GridMap<unsigned char, unsigned char>(
38  *env.makeMap<unsigned char>(static_cast<unsigned char>(0)))
39  {
40  // HACK: fix offset if the perimeter raster is different from this one
41  logging::check_fatal(0 != strcmp(perim_grid.proj4().c_str(), this->proj4().c_str()),
42  "Invalid projection for input perimeter raster - %s instead of %s",
43  perim_grid.proj4().c_str(),
44  this->proj4().c_str());
45  logging::check_fatal(perim_grid.cellSize() != this->cellSize(),
46  "Invalid cell size for input perimeter raster - %f instead of %f",
47  perim_grid.cellSize(),
48  this->cellSize());
49  const auto offset_x = static_cast<Idx>((this->xllcorner() - perim_grid.xllcorner()) / this->cellSize());
50  const auto perim_origin = static_cast<Idx>(perim_grid.rows() + perim_grid.yllcorner() / this->cellSize());
51  const auto this_origin = static_cast<Idx>(this->rows() + this->yllcorner() / this->cellSize());
52  const auto offset_y = static_cast<Idx>((perim_origin - this_origin));
53  // make sure we don't go out of bounds on grid
54  const auto min_column = static_cast<Idx>(offset_x < 0 ? abs(offset_x) : 0);
55  const auto max_columns = min(this->columns(), perim_grid.columns());
56  const auto min_row = static_cast<Idx>(offset_y < 0 ? abs(offset_y) : 0);
57  const auto max_rows = min(this->rows(),
58  static_cast<Idx>(perim_grid.rows() - abs(offset_y)));
59  logging::note("Correcting perimeter raster offset by %dx%d cells",
60  offset_x,
61  offset_y);
62  size_t count = 0;
63  // since it was read in as a vector we need to check all the cells
64  for (auto r = min_row; r < max_rows; ++r)
65  {
66  for (auto c = min_column; c < max_columns; ++c)
67  {
68  const auto loc = env.cell(r, c);
69  const Location fixed_loc(static_cast<Idx>(r + offset_y),
70  static_cast<Idx>(c + offset_x));
71  const auto value = perim_grid.at(fixed_loc);
72  if (value != perim_grid.noData() && !fuel::is_null_fuel(loc))
73  {
74  this->GridMap<unsigned char, unsigned char>::set(loc, value);
75  ++count;
76  }
77  }
78  }
79  for (auto& kv : data)
80  {
81  logging::check_fatal(fuel::is_null_fuel(env.cell(kv.first)),
82  "Null fuel in BurnedData");
83  }
84  logging::info("Loaded burned area of size %lu ha",

86  }

100 Perimeter::Perimeter(const Location& location,
101  const size_t size,
102  const wx::FwiWeather&,
103  const Environment& env)
104 {
105  // NOTE: FwiWeather is unused but could change this to try doing length to breadth ratio
106  auto perim_grid = env.makeMap<unsigned char>(0);
107  // want to find cells in the area that fill up the size we're looking for
108  size_t count = 0;
109  // convert into number of cells
110  const auto num_cells = size / (100.0 * 100.0 / (perim_grid->cellSize() * perim_grid->cellSize()));
111  auto max_distance = sqrt(num_cells / M_PI);
112  perim_grid->set(location, 1);
113  ++count;
114  // HACK: assume fuel for origin matches the rest of the fire
115  while (num_cells > count)
116  {
117  const auto range = static_cast<Idx>(ceil(max_distance));
118  for (auto x = -range; x <= range && num_cells > count; ++x)
119  {
120  for (auto y = -range; y <= range && num_cells > count; ++y)
121  {
122  // look at any cell that's within the range
123  if (sqrt(util::pow_int<2>(x) + util::pow_int<2>(y)) < max_distance)
124  {
125  const auto r = static_cast<Idx>(location.row() + x);
126  const auto c = static_cast<Idx>(location.column() + y);
127  const auto loc = env.cell(r, c);
128  if (1 != perim_grid->at(loc) && !fuel::is_null_fuel(loc))
129  {
130  perim_grid->set(loc, 1);
131  ++count;
132  }
133  }
134  }
135  }
136  max_distance += 0.1;
137  }
138  const BurnedMap burned(*perim_grid, env);
139  burned_ = burned.makeList();
140  edge_ = burned.makeEdge();
141 }
