25 FwiStream::FwiStream(const vector<TIMESTAMP_STRUCT>& dates,
26  Startup* startup,
27  const double latitude,
28  vector<Weather> cur_member)
29 {
30  static const AccumulatedPrecipitation ZeroFfmcPrecipitationAmount(2);
31  static const AccumulatedPrecipitation ZeroDmcPrecipitationAmount(10);
32  assert(cur_member.size() <= dates.size());
33  int month = dates.at(0).month;
34  auto started = (nullptr != startup) || (month > 9 || (9 == month && 15 <= dates.at(0).day));
35  auto shutdown = !started;
36  static auto fake_startup = Startup("",
37  TIMESTAMP_STRUCT{},
38  topo::Point(0, 0),
39  0,
40  Ffmc(0),
41  Dmc(0),
42  Dc(0),
43  AccumulatedPrecipitation(0),
44  false);
45  // HACK: could be started but now ended and marked as started because of month/day
46  const Startup* const use_startup = (!started || nullptr == startup)
47  ? &fake_startup
48  : startup;
49  auto last_ffmc = use_startup->ffmc();
50  auto last_dmc = use_startup->dmc();
51  auto last_dc = use_startup->dc();
52  Dc shutdown_dc(-1);
53  AccumulatedPrecipitation since_shutdown_accumulation(-1);
54  for (size_t date_index = 0; date_index < cur_member.size(); ++date_index)
55  {
56  const auto& wx = cur_member.at(date_index);
57  const auto for_date = dates.at(date_index);
58  month = for_date.month;
59  if (date_index >= 2)
60  {
61  // if we rolled into next year then start checking for startup conditions again
62  if (shutdown && started && 1 == month)
63  {
64  started = false;
65  }
66  if (!started)
67  {
68  // try to see if we can start things
69  double sum = 0;
70  for (auto j = date_index - 2; j <= date_index; ++j)
71  {
72  sum += cur_member.at(j).tmp().asDouble();
73  }
74  const auto avg = sum / 3;
75  // start stations if average temperature is at or over 11C for 3 consecutive days
76  started = avg >= 11;
77  if (started)
78  {
79  // use default startup values for FWI indices
80  // FIX: check that we're doing this right - should these be yesterday's indices
81  // (i.e. lastValues) or today's?
82  last_ffmc = Ffmc(85.0);
83  last_dmc = Dmc(6.0);
84  // FIX: DC adjustment isn't correct over winter
85  if (Dc(0) <= shutdown_dc)
86  {
87  last_dc = Dc(15.0);
88  }
89  }
90  // we aren't shutdown if we just started
91  shutdown = !started;
92  if (started)
93  {
94  shutdown_dc = Dc(-1);
95  since_shutdown_accumulation = AccumulatedPrecipitation(-1);
96  }
97  }
98  else if (((!shutdown && month > 9)
99  || (9 == month && 15 <= for_date.day))
100  && date_index >= 2)
101  {
102  // try to see if we can stop things
103  double sum = 0;
104  for (auto j = date_index - 2; j <= date_index; ++j)
105  {
106  sum += cur_member.at(j).tmp().asDouble();
107  }
108  auto avg = sum / 3;
109  // shut down station if DMC < 10 and temperature <= 2.5C for 3 days
110  shutdown = Dmc(10) > last_dmc && 2.5 >= avg;
111  // check if past week has been 2.5C
112  // NOTE: need to be able to average at least 7 days
113  if (!shutdown && date_index >= 6)
114  {
115  // try to see if we can stop things
116  sum = 0;
117  for (auto j = date_index - 6; j <= date_index; ++j)
118  {
119  sum += cur_member.at(j).tmp().asDouble();
120  }
121  avg = sum / 7;
122  // shut down station if temperature <= 2.5C for 7 days
123  shutdown = 2.5 >= avg;
124  }
125  }
126  else if (10 <= month)
127  {
128  shutdown = true;
129  }
130  }
131  auto apcp = wx.apcp();
132  // have to add apcp because it isn't in accumulation yet
133  if (shutdown)
134  {
135  since_shutdown_accumulation += apcp;
136  }
137  // if no startup then we don't need to add since it's 0
138  if (0 == date_index && nullptr != startup)
139  {
140  // need to add in 0800 obs for day 1
141  apcp += startup->apcp0800();
142  }
143  Ffmc ffmc(wx.tmp(), wx.rh(), wx.wind().speed(), apcp, last_ffmc);
144  if (shutdown)
145  {
146  // HACK: if we got 2mm of pcp since shutdown then assume 0 indices
147  if (since_shutdown_accumulation > ZeroFfmcPrecipitationAmount)
148  {
149  ffmc = Ffmc(0.0);
150  }
151  else
152  {
153  // want to decrease the value but never increase it
154  ffmc = min(last_ffmc, ffmc);
155  }
156  }
157  ffmc = max(Ffmc(0.0), ffmc);
158  Dmc dmc(wx.tmp(), wx.rh(), apcp, last_dmc, month, latitude);
159  if (shutdown)
160  {
161  // HACK: if we got 10mm of pcp since shutdown then assume 0 indices
162  if (since_shutdown_accumulation > ZeroDmcPrecipitationAmount)
163  {
164  dmc = Dmc(0.0);
165  }
166  else
167  {
168  // want to decrease the value but never increase it
169  dmc = min(last_dmc, dmc);
170  }
171  }
172  dmc = max(Dmc(0.0), dmc);
173  Dc dc(0);
174  if (shutdown)
175  {
176  assert(Dc(0) <= shutdown_dc);
177  assert(AccumulatedPrecipitation(0) <= since_shutdown_accumulation);
178  const auto a = 1.0;
179  // NOTE: concerned about the lack of overwinter precipitation present
180  // in reanalysis data
181  // NOTE: use 0.9 due to deep duff fuels
182  const auto b = 0.9;
183  const auto smi_f = 800 * exp(-shutdown_dc.asDouble() / 400.0);
184  const auto smi_s = a * smi_f + b * since_shutdown_accumulation.asDouble();
185  dc = min(last_dc, Dc(400 * log(800 / smi_s)));
186  }
187  else
188  {
189  dc = Dc(wx.tmp(), apcp, last_dc, month, latitude);
190  }
191  dc = max(Dc(0.0), dc);
192  const Isi isi(wx.wind().speed(), ffmc);
193  const Bui bui(dmc, dc);
194  const Fwi fwi(isi, bui);
195  // need to add in check for station startup and shutdown
196  wx_.emplace_back(wx.tmp(),
197  wx.rh(),
198  wx.wind(),
199  apcp,
200  ffmc,
201  dmc,
202  dc,
203  isi,
204  bui,
205  fwi);
206  last_ffmc = ffmc;
207  last_dmc = dmc;
208  last_dc = dc;
209  if (started && !shutdown)
210  {
211  shutdown_dc = last_dc;
212  since_shutdown_accumulation = AccumulatedPrecipitation(0.0);
213  }
214  }
215 }
