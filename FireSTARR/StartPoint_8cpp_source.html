47 static double sunrise_sunset(const int jd,
48  const double latitude,
49  const double longitude,
50  const bool for_sunrise) noexcept
51 {
52  static const auto Zenith = util::to_radians(96);
53  static const auto LocalOffset = -5;
54  const auto t_hour = for_sunrise ? 6 : 18;
55  // http://edwilliams.org/sunrise_sunset_algorithm.htm
56  const auto lng_hour = longitude / 15;
57  const auto t = jd + (t_hour - lng_hour) / 24;
58  const auto m = 0.9856 * t - 3.289;
59  const auto l = fix_degrees(
60  m + 1.916 * sin(util::to_radians(m)) + 0.020 * sin(util::to_radians(2 * m)) + 282.634);
61  auto ra = fix_degrees(util::to_degrees(atan(0.91764 * tan(util::to_radians(l)))));
62  const auto l_quadrant = floor(l / 90) * 90;
63  const auto ra_quadrant = floor(ra / 90) * 90;
64  ra += l_quadrant - ra_quadrant;
65  ra /= 15;
66  const auto sin_dec = 0.39782 * sin(util::to_radians(l));
67  const auto cos_dec = cos(asin(sin_dec));
68  const auto cos_h = (cos(Zenith) - sin_dec * sin(util::to_radians(latitude))) / (cos_dec * cos(util::to_radians(latitude)));
69  if (cos_h > 1)
70  {
71  // sun never rises
72  return for_sunrise ? -1 : 25;
73  }
74  if (cos_h < -1)
75  {
76  // sun never sets
77  return for_sunrise ? 25 : -1;
78  }
79  auto h = util::to_degrees(acos(cos_h));
80  if (for_sunrise)
81  {
82  h = 360 - h;
83  }
84  h /= 15;
85  const auto mean_t = h + ra - 0.06571 * t - 6.622;
86  const auto ut = mean_t - lng_hour;
87  return fix_hours(ut + LocalOffset);
88 }

99 static array<tuple<double, double>, MAX_DAYS> make_days(
100  const double latitude,
101  const double longitude) noexcept
102 {
103  array<tuple<double, double>, MAX_DAYS> days{};
104  array<double, MAX_DAYS> day_length_hours{};
105  for (size_t i = 0; i < day_length_hours.size(); ++i)
106  {
107  days[i] = make_tuple(
108  fix_hours(
109  sunrise(static_cast<int>(i), latitude, longitude) + sim::Settings::offsetSunrise()),
110  fix_hours(
111  sunset(static_cast<int>(i),
112  latitude,
113  longitude)
114  - sim::Settings::offsetSunset()));
115  day_length_hours[i] = get<1>(days[i]) - get<0>(days[i]);
116  }
117  return days;
