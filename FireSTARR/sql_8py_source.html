35 def get_current_fires():
36  """!
37  Get fires that are currently active from DFOSS
38  @return Current fires from DFOSS
39  """
40  stmt_select = r'''SELECT
41  f.FIREID,
42  c.ORGUNITCODE || TO_CHAR(f.FIRESEQ, '000') AS FIRENAME,
43  f.LATITUDE,
44  f.LONGITUDE,
45  f.CONFIRMDATE,
46  f.SMOKECOLUMN,
47  (SELECT s.CURRENTSIZE
48  FROM FIRESTATUS s
49  WHERE s.FIREID = f.FIREID
50  AND s.STATUSDATE = (
51  SELECT MAX(s2.STATUSDATE)
52  FROM FIRESTATUS s2
53  WHERE s2.FIREID = f.FIREID)
54  ) as CURRENTSIZE,
55  (SELECT s.CONDITION
56  FROM FIRESTATUS s
57  WHERE s.FIREID = f.FIREID
58  AND s.STATUSDATE = (
59  SELECT MAX(s2.STATUSDATE)
60  FROM FIRESTATUS s2
61  WHERE s2.FIREID = f.FIREID)
62  ) as CONDITION
63 FROM
64  (SELECT *
65  FROM FIRE f2
66  WHERE f2.FIREID NOT IN
67  (SELECT DISTINCT FIREID
68  FROM FIRESTATUS s
69  WHERE CONDITION = 'OUT'
70  AND f2.FIRETYPE <> 'PB')) f
71  LEFT JOIN CT_ORGUNIT c ON c.ORGUNIT=f.ORGUNIT
72 WHERE f.FIRETYPE <> 'PB'
73 '''
74  index = ['FIREID']
75  delete_all = True
76  # need to fix longitude because they're all positive even though they're in the west
77  df = read_remote(stmt_select).set_index(index)
78  df['LONGITUDE'] = df['LONGITUDE'].apply(lambda x: -abs(x))
79  return df

164  )

168  AND f.FIRETYPE <> 'PB'
