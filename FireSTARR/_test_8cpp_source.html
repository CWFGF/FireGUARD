162 int run_test(const char* output_directory,
163  const fuel::FuelLookup& fuel_lookup,
164  const string& fuel_name,
165  const SlopeSize slope,
166  const AspectSize aspect,
167  const int num_hours,
168  const wx::Dc& dc,
169  const wx::Bui& bui,
170  const wx::Dmc& dmc,
171  const wx::Ffmc& ffmc,
172  const wx::Wind& wind)
173 {
174  static const auto Latitude = 49.3911;
175  static const auto Longitude = -84.7395;
176  static const topo::StartPoint ForPoint(Latitude, Longitude);
177  const auto start_date = 123;
178  const auto end_date = start_date + static_cast<double>(num_hours) / DAY_HOURS;
179  util::make_directory_recursive(output_directory);
180  const auto fuel = fuel_lookup.byName(fuel_name);
181  auto values = vector<topo::Cell>();
182  values.reserve(MAX_ROWS * MAX_COLUMNS);
183  for (Idx r = 0; r < MAX_ROWS; ++r)
184  {
185  for (Idx c = 0; c < MAX_COLUMNS; ++c)
186  {
187  values.emplace_back(r, c, slope, aspect, fuel::FuelType::safeCode(fuel));
188  }
189  }
190  const auto cells = new data::ConstantGrid<topo::Cell>{
191  TEST_GRID_SIZE,
192  MAX_ROWS,
193  MAX_COLUMNS,
194  topo::Cell{},
195  -1,
196  TEST_XLLCORNER,
197  TEST_YLLCORNER,
198  TEST_PROJ4,
199  std::move(values)};
200  TestEnvironment env(cells);
201  const Location start_location(static_cast<Idx>(MAX_ROWS / 2),
202  static_cast<Idx>(MAX_COLUMNS / 2));
203  Model model(ForPoint, output_directory, &env);
204  const auto start_cell = make_shared<topo::Cell>(model.cell(start_location));
205  TestWeather weather(fuel, start_date, dc, bui, dmc, ffmc, wind);
206  TestScenario scenario(&model, start_cell, ForPoint, start_date, end_date, &weather);
207  auto info = SpreadInfo(scenario,
208  start_date,
209  *start_cell,
210  model.nd(start_date),
211  weather.at(start_date));
212  map<double, ProbabilityMap*> probabilities{};
213  scenario.run(&probabilities);
214  scenario.saveObservers("");
215  logging::note("Final Size: %0.0f, ROS: %0.2f",
216  scenario.currentFireSize(),
217  info.headRos());
218  return 0;
219 }

270  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), '/'),
271  simple_fuel_name.end());

333  "\tWind Speed:\t\t%f\n"

335  num_hours,
336  slope,
337  aspect,
338  wind_speed,
