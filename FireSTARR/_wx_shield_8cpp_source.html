180 template <class T>
181 TIMESTAMP_STRUCT date_by_offset(T offset) noexcept
182 {
183  TIMESTAMP_STRUCT result{};
184  const auto t = time(nullptr) + DAY_SECONDS * static_cast<time_t>(offset);
185  // convert to local time
186  tm local{};
187  localtime_s(&local, &t);
188  // HACK: always use 13:00
189  local.tm_hour = 13;
190  local.tm_min = 0;
191  local.tm_sec = 0;
192  util::to_ts(local, &result);
193  return result;
194 }
195 int offset_by_date(const TIMESTAMP_STRUCT& d) noexcept
196 {
197  const auto t = time(nullptr);
198  tm tmd{0, 0, 0, d.day, d.month - 1, d.year - 1900, 0, 0, 0};
199  const auto seconds = difftime(mktime(&tmd), t);
200  return static_cast<int>(seconds / DAY_SECONDS);
201 }
202 void WxShield::preProcessEnsembles(ModelMap& hindcasts, vector<int>& years) const
203 {
204  unordered_map<int, bool> remove_years{};
205  const size_t year_length = 365;
206  auto members = &hindcasts.at(hindcasts.begin()->first);
207  const auto original_length = min(year_length,
208  (*members).at(members->begin()->first).size());
209  for (const auto& kv : hindcasts)
210  {
211  auto& cur_model = kv.first;
212  members = &hindcasts.at(cur_model);
213  // do this to deal with leap years possibly being in there (so 366 days)
214  for (const auto& mv : *members)
215  {
216  auto m = mv.first;
217  // make this a pointer so we don't copy it
218  // don't use mv.second since it doesn't seem to append to the vector
219  auto cur_member = &((*members).at(m));
220  auto current_end = cur_member->size() - 1;
221  auto offset = 1;
222  while (current_end < (dates_.size() - 1) && (remove_years.find(m) == remove_years.
223  end()))
224  {
225  const auto days_left = (dates_.size() - 1) - current_end;
226  const auto f = m + offset;
227  // >= since leap years might be longer
228  if (members->find(f) != members->end() && (*members).at(f).size() >=
229  original_length)
230  {
231  const auto from_member = &((*members).at(f));
232  for (size_t i = 0; i <
233  min(days_left, min(year_length, from_member->size() - 1)); ++i)
234  {
235  const auto wx = (*from_member).at(i);
236  // need to fix date but copy everything else
237  cur_member->emplace_back(wx);
238  }
239  }
240  else
241  {
242  // need to remove this year from everything because it goes too far
243  const auto now = date_by_offset(0);
244  const int cur_year = now.year;
245  for (auto i = m; i <= min(f, cur_year - 1); ++i)
246  {
247  remove_years.emplace(i, true);
248  }
249  }
250  current_end = cur_member->size() - 1;
251  offset++;
252  }
253  assert(cur_member->size() <= dates_.size());
254  }
255  for (const auto& r : remove_years)
256  {
257  // get rid of all members that can't go far enough for the projection we asked for
258  auto i = r.first;
259  members->erase(i);
260  }
261  }
262  for (auto y : years)
263  {
264  for (auto& kv : hindcasts)
265  {
266  auto h = kv.second;
267  const auto& seek = h.find(y);
268  logging::check_fatal(seek == h.end(),
269  "Hindcast %s missing year %d",
270  kv.first.name().c_str(),
271  y);
272  }
273  }
274 }

301 class DecimalFormatter
302 {
303  // this exists so that we can output a maximum number of places after the
304  // decimal place but not if it's 0
305 public:
306  ~DecimalFormatter() = default;
307  DecimalFormatter(const DecimalFormatter& rhs) = delete;
308  DecimalFormatter(DecimalFormatter&& rhs) = delete;
309  DecimalFormatter& operator=(const DecimalFormatter& rhs) = delete;
310  DecimalFormatter& operator=(DecimalFormatter&& rhs) = delete;
316  DecimalFormatter(const double value, const int after_decimal) noexcept
317  : value(value), after_decimal(after_decimal)
318  {
319  }
323  const double value;
324 #pragma warning (push)
325 #pragma warning (disable: 4820)
329  const int after_decimal;
330 };
331 #pragma warning (pop)
332 ostream& operator<<(ostream& os, const DecimalFormatter& df)
333 {
334  ostringstream ss{};
335  // let this figure out how the number should be rounded
336  ss << fixed << setprecision(df.after_decimal) << df.value;
337  auto s = ss.str();
338  // remove trailing 0's
339  while (s.at(s.size() - 1) == '0')
340  {
341  s = s.substr(0, s.size() - 1);
342  }
343  // if last character is the decimal the remove it
344  if (s.at(s.size() - 1) == '.')
345  {
346  s = s.substr(0, s.size() - 1);
347  }
348  os << s;
349  return os;
350 }

386 void WxShield::processEnsembles(const vector<Weather>& actuals,
387  ModelMap& hindcasts,
388  ModelMap& forecasts,
389  vector<int>& years,
390  const size_t end_of_ensemble)
391 {
392  map<wstring, FwiStream> scenarios{};
393  map<wstring, FwiStream> reanalysis{};
394  // HACK: override the APCP_0800 since the actuals are already counting the whole day
395  const auto actuals_startup = nullptr == startup_
396  ? nullptr
397  : make_unique<Startup>(startup_->station(),
398  startup_->generated(),
399  startup_->point(),
400  startup_->distanceFrom(),
401  startup_->ffmc(),
402  startup_->dmc(),
403  startup_->dc(),
404  AccumulatedPrecipitation::Zero,
405  startup_->isOverridden());
406  auto fix_actuals = processMember(&(*actuals_startup), point_.latitude(), actuals);
407  // find AFFES forecast
408  const vector<Weather>* affes = nullptr;
409  size_t affes_days = 0;
410  for (const auto& kv : forecasts)
411  {
412  const auto model = kv.first;
413  if (model.name() == L"AFFES")
414  {
415  logging::check_fatal(kv.second.size() != 1,
416  "There should only be one AFFES forecast");
417  affes = &kv.second.at(0);
418  affes_days = affes->size();
419  break;
420  }
421  }
422  wstring first;
423  auto i = 0;
424  // vector<vector<Weather*>> affes_members;
425  // static const auto EnsembleMembers = 42;
426  // affes_members.reserve(EnsembleMembers * years_.size() * affes_days);
427  for (const auto& kv : forecasts)
428  {
429  const auto model = kv.first;
430  if (model.name() == L"AFFES")
431  {
432  continue;
433  }
434  auto members = kv.second;
435  for (const auto& mkv : members)
436  {
437  const auto member_id = mkv.first;
438  auto member = mkv.second;
439  for (const auto& hkv : hindcasts)
440  {
441  auto hindcast = hkv.second;
442  for (const auto& year : years)
443  {
444  auto h = &hindcast.at(year);
445  const auto count = min(h->size() - end_of_ensemble,
446  dates_.size() - member.size());
447  vector<Weather> m{member.size() + count};
448  m.reserve(dates_.size());
449  std::copy(member.begin(), member.end(), m.begin());
450  std::copy_n(h->begin() + static_cast<__int64>(end_of_ensemble),
451  static_cast<__int64>(count),
452  m.begin() + static_cast<__int64>(member.size()));
453  const auto mem_id = L"0" + to_wstring(member_id);
454  const auto fake_name = model.name() + L"x" + to_wstring(year) + L"x"
455  + mem_id.substr(mem_id.size() - 2);
456  scenarios.emplace(fake_name, processMember(&(*startup_), point_.latitude(), m));
457  if (first.empty())
458  {
459  // we know that scenario 0 should be first
460  first = fake_name;
461  }
462  ++i;
463  // add in AFFES versions of this member
464  // for (size_t cur_days = 1; cur_days <= affes_days; ++cur_days)
465  if (0 < affes_days)
466  {
467  vector<Weather> affes_m{m.size()};
468  std::copy(m.begin(), m.end(), affes_m.begin());
469  // copy over the first j days with the AFFES forecast
470  // std::copy_n(affes->begin(), cur_days, affes_m.begin());
471  std::copy_n(affes->begin(), affes_days, affes_m.begin());
472  // ReSharper disable once StringLiteralTypo
473  const auto affes_fake_name = L"AFFESX" + model.name() + L"x" +
474  to_wstring(year) + L"x"
475  + mem_id.substr(mem_id.size() - 2);
476  scenarios.emplace(affes_fake_name,
477  processMember(&(*startup_), point_.latitude(), affes_m));
478  }
479  }
480  }
481  }
482  }
483  if (full_wx_)
484  {
485  for (const auto& hkv : hindcasts)
486  {
487  auto hindcast = hkv.second;
488  auto min_year = numeric_limits<int>::max();
489  auto max_year = numeric_limits<int>::min();
490  for (const auto& ykv : hindcast)
491  {
492  const auto year = ykv.first;
493  min_year = min(min_year, year);
494  max_year = max(max_year, year);
495  }
496  for (auto year = min_year; year <= max_year; ++year)
497  {
498  auto h = &hindcast.at(year);
499  const auto count = dates_.size();
500  vector<Weather> m{count};
501  m.reserve(dates_.size());
502  std::copy_n(h->begin(), static_cast<__int64>(count), m.begin());
503  reanalysis.emplace(to_wstring(year),
504  processMember(&(*startup_), point_.latitude(), m));
505  }
506  }
507  }
508  static const auto IndicesHeader =
509  "Scenario,Date,APCP,TMP,RH,WS,WD,FFMC,DMC,DC,ISI,BUI,FWI";
510  if (nullptr != affes)
511  {
512  // do this so we know the scenario was processed and indices calculated
513  for (const auto& s : scenarios)
514  {
515  auto k = s.first;
516  if (0 == k.find(L"AFFES"))
517  {
518  auto v = s.second;
519  FwiVector wx{affes_days};
520  vector<TIMESTAMP_STRUCT> dates{affes_days};
521  // find first scenario that used the AFFES forecast
522  for (size_t d = 0; d < affes_days; ++d)
523  {
524  wx.at(d) = v.wx().at(d);
525  dates.at(d) = dates_.at(d);
526  }
527  const string tmp;
528  stringstream iss(tmp);
529  logging::note("AFFES indices are:");
530  iss << IndicesHeader << endl;
531  const FwiStream fwi_wx(std::move(wx));
532  iss << make_tuple(L"AFFES", fwi_wx, dates);
533  cout << iss.str();
534  break;
535  }
536  }
537  }
538  logging::info("Saving to %s", save_to_.c_str());
539  ofstream out;
540  out.open(save_to_);
541  out << IndicesHeader << endl;
542  out << make_tuple(L"actual", fix_actuals, dates_);
543  i = 0;
544  for (const auto& kv : scenarios)
545  {
546  out << make_tuple(to_wstring(--i), kv.second, dates_);
547  }
548  if (!reanalysis.empty())
549  {
550  auto min_year = numeric_limits<int>::max();
551  auto max_year = numeric_limits<int>::min();
552  for (const auto& ykv : reanalysis)
553  {
554  const auto year = ykv.first;
555  min_year = min(min_year, stoi(year));
556  max_year = max(max_year, stoi(year));
557  }
558  for (auto year = min_year; year <= max_year; ++year)
559  {
560  const auto y = to_wstring(year);
561  out << make_tuple(y, reanalysis.at(y), dates_);
562  }
563  }
564  out.close();
565 }
566 static vector<int> find_years(const int offset)
567 {
568  vector<int> years{};
569  auto scores = WxShield::readScoreByOffset(offset);
570  double sum_grade = 0;
571  const auto target = sim::Settings::maxGrade();
572  logging::note("Target score is %f", target);
573  for (const auto& s : scores)
574  {
575  if (sum_grade >= target)
576  {
577  break;
578  }
579  sum_grade += s.grade();
580  years.push_back(s.year());
581  }
582  // HACK: sort now so that output order is the same as wxshield csv
583  sort(years.begin(), years.end());
584  return years;
585 }

605 static unique_ptr<Startup> override_startup(const topo::Point& point,

608  const Dmc* const dmc,
609  const Dc* const dc,
610  const AccumulatedPrecipitation* const
611  apcp_0800)
612 {
613  // if no overrides then just use original
614  if (nullptr == ffmc && nullptr == dmc && nullptr == dc && nullptr == apcp_0800)
615  {
616  return startup;
617  }
618  const auto& base = (nullptr == startup) ? STARTUP_ZERO : *startup;
619  return make_unique<Startup>(Startup(L"N/A",
620  {},

622  0.0,
623  (nullptr == ffmc) ? base.ffmc() : *ffmc,
