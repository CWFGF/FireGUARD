90 static void make_threshold(vector<double>* thresholds,
91  mt19937* mt,
92  const Day start_day,
93  const Day last_date,
94  double (*convert)(double value))
95 {
96  const auto total_weight = Settings::thresholdScenarioWeight() + Settings::thresholdDailyWeight() + Settings::thresholdHourlyWeight();
97  uniform_real_distribution<double> rand(0.0, 1.0);
98  const auto general = rand(*mt);
99  for (size_t i = start_day; i < MAX_DAYS; ++i)
100  {
101  const auto daily = rand(*mt);
102  for (auto h = 0; h < DAY_HOURS; ++h)
103  {
104  // generate no matter what so if we extend the time period the results
105  // for the first days don't change
106  const auto hourly = rand(*mt);
107  // only save if we're going to use it
108  if (i <= last_date)
109  {
110  // subtract from 1.0 because we want weight to make things more likely not less
111  // ensure we stay between 0 and 1
112  thresholds->at((i - start_day) * DAY_HOURS + h) =
113  convert(
114  max(0.0,
115  min(1.0,
116  1.0 - (Settings::thresholdScenarioWeight() * general + Settings::thresholdDailyWeight() * daily + Settings::thresholdHourlyWeight() * hourly) / total_weight)));
117  // thresholds->at((i - start_day) * DAY_HOURS + h) = 0.0;
118  }
119  }
120  }
121 }

218 void Scenario::evaluate(const Event& event)
219 {
220  log_check_fatal(event.time() < current_time_,
221  "Expected time to be > %f but got %f",
222  current_time_,
223  event.time());
224  const auto& p = event.cell();
225  switch (event.type())
226  {
227  case Event::FIRE_SPREAD:
228  scheduleFireSpread(event);
229  break;
230  case Event::SAVE:
231  saveObservers(event.time());
232  saveStats(event.time());
233  break;
234  case Event::NEW_FIRE:
235  // HACK: don't do this in constructor because scenario creates this in its constructor
236  points_[p].emplace_back(p.column(), p.row(), CELL_CENTER, CELL_CENTER);
237  if (fuel::is_null_fuel(event.cell()))
238  {
239  log_fatal("Trying to start a fire in non-fuel");
240  }
241  log_verbose("Starting fire at point (%f, %f) in fuel type %s at time %f",
242  p.column() + CELL_CENTER,
243  p.row() + CELL_CENTER,
244  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
245  event.time());
246  if (!survives(event.time(), event.cell(), event.timeAtLocation()))
247  {
248  const auto wx = weather(event.time());
249  log_info("Didn't survive ignition in %s with weather %f, %f",
250  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
251  wx->ffmc(),
252  wx->dmc());
253  // HACK: we still want the fire to have existed, so set the intensity of the origin
254  }
255  // fires start with intensity of 1
256  burn(event, 1);
257  scheduleFireSpread(event);
258  break;
259  case Event::END_SIMULATION:
260  log_verbose("End simulation event reached at %f", event.time());
261  endSimulation();
262  break;
263  default:
264  throw runtime_error("Invalid event type");
265  }
266 }

451 Scenario* Scenario::run(map<double, ProbabilityMap*>* probabilities)
452 {
453  log_check_fatal(ran(), "Scenario has already run");
454  log_debug("Starting");
455  CriticalSection _(Model::task_limiter);
456  probabilities_ = probabilities;
457  for (auto time : save_points_)
458  {
459  // NOTE: these happen in this order because of the way they sort based on type
460  addEvent(Event::makeSave(static_cast<double>(time)));
461  }
462  if (nullptr == perimeter_)
463  {
464  addEvent(Event::makeNewFire(start_time_, cell(*start_cell_)));
465  }
466  else
467  {
468  intensity_->applyPerimeter(*perimeter_);
469  const auto& env = model().environment();
470  for (const auto& location : perimeter_->edge())
471  {
472  // const auto cell = env.cell(location.hash());
473  const auto cell = env.cell(location);
474  log_check_fatal(fuel::is_null_fuel(cell), "Null fuel in perimeter");
475  log_verbose("Adding point (%d, %d)",
476  cell.column() + CELL_CENTER,
477  cell.row() + CELL_CENTER);
478  points_[cell].emplace_back(cell.column(), cell.row(), CELL_CENTER, CELL_CENTER);
479  }
480  addEvent(Event::makeFireSpread(start_time_));
481  }
482  // HACK: make a copy of the event so that it still exists after it gets processed
483  // NOTE: sorted so that EventSaveASCII is always just before this
484  // Only run until last time we asked for a save for
485  log_verbose("Creating simulation end event for %f", last_save_);
486  addEvent(Event::makeEnd(last_save_));
487  // mark all original points as burned at start
488  for (auto& kv : points_)
489  {
490  const auto& location = kv.first;
491  // would be burned already if perimeter applied
492  if (canBurn(location))
493  {
494  const auto fake_event = Event::makeFireSpread(
495  start_time_,
496  static_cast<IntensitySize>(1),
497  location);
498  burn(fake_event, static_cast<IntensitySize>(1));
499  }
500  }
501  while (!scheduler_.empty())
502  {
503  evaluateNextEvent();
504  }
505  ++COMPLETED;
506  // HACK: use + to pull value out of atomic
507 #ifdef NDEBUG
508  log_info("[% d of % d] Completed with final size % 0.1f ha",
509  +COMPLETED,
510  +COUNT,
511  currentFireSize());
512 #else
513  // try to make output consistent if in debug mode
514  log_info("Completed with final size %0.1f ha",
515  currentFireSize());
516 #endif
517  ran_ = true;
518  // delete unburnable_;
519  // unburnable_ = nullptr;
520 #ifndef NDEBUG
521  static const size_t BufferSize = 64;
522  char buffer[BufferSize + 1] = {0};
523  sprintf(buffer,
524  "%03zu_%06lld_extinction",
525  id(),
526  simulation());
527  saveProbabilities(Settings::outputDirectory(), string(buffer), extinction_thresholds_);
528  sprintf(buffer,
529  "%03zu_%06lld_spread",
530  id(),
531  simulation());
532  saveProbabilities(Settings::outputDirectory(), string(buffer), spread_thresholds_by_ros_);
533 #endif
534  return this;
535 }

544 inline void Scenario::checkCondense(vector<InnerPos>& a)
545 {
546  if (a.size() > Settings::maxCellPoints())
547  {
548  size_t n_pos = 0;
549  auto n = numeric_limits<double>::min();
550  size_t ne_pos = 0;
551  auto ne = numeric_limits<double>::min();
552  size_t e_pos = 0;
553  auto e = numeric_limits<double>::min();
554  size_t se_pos = 0;
555  auto se = numeric_limits<double>::min();
556  size_t s_pos = 0;
557  auto s = numeric_limits<double>::max();
558  size_t sw_pos = 0;
559  auto sw = numeric_limits<double>::min();
560  size_t w_pos = 0;
561  auto w = numeric_limits<double>::max();
562  size_t nw_pos = 0;
563  auto nw = numeric_limits<double>::min();
564  for (size_t i = 0; i < a.size(); ++i)
565  {
566  const auto& p = a[i];
567  const auto x = p.sub_x - 0.5;
568  const auto y = p.sub_y - 0.5;
569  // NOTE: seems like it should be else if, but want to make sure
570  // all of these are initialized after the first point
571  if (y > n)
572  {
573  n_pos = i;
574  n = y;
575  }
576  if (y < s)
577  {
578  s_pos = i;
579  s = y;
580  }
581  const auto cur_ne = x + y;
582  const auto cur_sw = -x - y;
583  if (cur_ne > ne)
584  {
585  ne_pos = i;
586  ne = cur_ne;
587  }
588  if (cur_sw > sw)
589  {
590  sw_pos = i;
591  sw = cur_sw;
592  }
593  if (x > e)
594  {
595  e_pos = i;
596  e = x;
597  }
598  if (x < w)
599  {
600  w_pos = i;
601  w = x;
602  }
603  const auto cur_se = x - y;
604  const auto cur_nw = -x + y;
605  if (cur_se > se)
606  {
607  se_pos = i;
608  se = cur_se;
609  }
610  if (cur_nw > nw)
611  {
612  nw_pos = i;
613  nw = cur_nw;
614  }
615  }
616  // NOTE: temporarily use a set to ensure no duplicates
617  const std::set<InnerPos> result{
618  a[n_pos],
619  a[ne_pos],
620  a[e_pos],
621  a[se_pos],
622  a[s_pos],
623  a[sw_pos],
624  a[w_pos],
625  a[nw_pos]};
626  a = {};
627  a.assign(result.begin(), result.end());
628  }

630 void Scenario::scheduleFireSpread(const Event& event)
631 {
632  const auto time = event.time();
633  //note("time is %f", time);
634  current_time_ = time;
635  const auto wx = weather(time);
636  // log_note("%d points", points_->size());
637  const auto this_time = util::time_index(time);
638  const auto next_time = static_cast<double>(this_time + 1) / DAY_HOURS;
639  // should be in minutes?
640  const auto max_duration = (next_time - time) * DAY_MINUTES;
641  //note("time is %f, next_time is %f, max_duration is %f",
642  // time,
643  // next_time,
644  // max_duration);
645  const auto max_time = time + max_duration / DAY_MINUTES;
646  if (wx->ffmc().asDouble() < minimumFfmcForSpread(time))
647  {
648  addEvent(Event::makeFireSpread(max_time));
649  log_verbose("Waiting until %f because of FFMC", max_time);
650  return;
651  }
652  if (current_time_index_ != this_time)
653  {
654  current_time_index_ = this_time;
655  // offsets_.clear();
656  // max_intensity_.clear();
657  offsets_ = {};
658  max_intensity_ = {};
659  max_ros_ = 0.0;
660  }
661  auto any_spread = false;
662  for (const auto& kv : points_)
663  {
664  const auto& location = kv.first;
665  // any cell that has the same fuel, slope, and aspect has the same spread
666  const auto key = location.key();
667  const auto seek_spreading = offsets_.find(key);
668  if (seek_spreading == offsets_.end())
669  {
670  // have not calculated spread for this cell yet
671  const SpreadInfo origin(*this, time, location, nd(time), wx);
672  // will be empty if invalid
673  offsets_.emplace(key, origin.offsets());
674  if (!origin.isNotSpreading())
675  {
676  any_spread = true;
677  max_ros_ = max(max_ros_, origin.headRos());
678  max_intensity_[key] = max(max_intensity_[key], origin.maxIntensity());
679  }
680  }
681  else
682  {
683  // already did the lookup so use the result
684  any_spread |= !seek_spreading->second.empty();
685  }
686  }
687  if (!any_spread || max_ros_ < Settings::minimumRos())
688  {
689  log_verbose("Waiting until %f", max_time);
690  addEvent(Event::makeFireSpread(max_time));
691  return;
692  }
693  //note("Max spread is %f, max_ros is %f",
694  // Settings::maximumSpreadDistance() * cellSize(),
695  // max_ros_);
696  const auto duration = ((max_ros_ > 0)
697  ? min(max_duration,
698  Settings::maximumSpreadDistance() * cellSize() / max_ros_)
699  : max_duration);
700  //note("Spreading for %f minutes", duration);
701  const auto new_time = time + duration / DAY_MINUTES;
702  for (auto& kv : points_)
703  {
704  const auto& location = kv.first;
705  const auto key = location.key();
706  auto& offsets = offsets_.at(key);
707  if (!offsets.empty())
708  {
709  for (auto& o : offsets)
710  {
711  // offsets in meters
712  const auto offset_x = o.x * duration;
713  const auto offset_y = o.y * duration;
714  const Offset offset{offset_x, offset_y};
715  //note("%f, %f", offset_x, offset_y);
716  for (auto& p : kv.second)
717  {
718  const InnerPos pos = p.add(offset);
719  const auto for_cell = cell(pos);
720  if (!(fuel::is_null_fuel(for_cell) || (*unburnable_)[for_cell.hash()]))
721  {
722  //log_extensive("Adding point (%f, %f)", pos.x, pos.y);
723  point_map_[for_cell].emplace_back(pos);
724  }
725  }
726  }
727  }
728  else
729  {
730  // can't just keep existing points by swapping because something may have spread into this cell
731  auto& pts = point_map_[location];
732  pts.insert(pts.end(), kv.second.begin(), kv.second.end());
733  std::sort(pts.begin(), pts.end());
734  }
735  // kv.second.clear();
736  kv.second = {};
737  }
738  vector<topo::Cell> erase_what{};
739  for (auto& kv : point_map_)
740  {
741  if (!kv.second.empty())
742  {
743  auto& for_cell = kv.first;
744  if (canBurn(for_cell) && max_intensity_[for_cell.key()] > 0)
745  {
746  // HACK: make sure it can't round down to 0
747  const auto intensity = static_cast<IntensitySize>(max(
748  1.0,
749  max_intensity_[for_cell.key()]));
750  const auto fake_event = Event::makeFireSpread(
751  new_time,
752  intensity,
753  for_cell);
754  burn(fake_event, intensity);
755  }
756  // check if this cell is surrounded by burned cells or non-fuels
757  // if surrounded then just drop all the points inside this cell
758  if (!(*unburnable_)[for_cell.hash()])
759  {
760  if (!isSurrounded(for_cell) && survives(new_time, for_cell, new_time - arrival_[for_cell]))
761  {
762  checkCondense(kv.second);
763  //checkHull(for_cell, kv.second);
764  std::swap(points_[for_cell], kv.second);
765  }
766  else
767  {
768  // whether it went out or is surrounded just mark it as unburnable
769  (*unburnable_)[for_cell.hash()] = true;
770  erase_what.emplace_back(for_cell);
771  }
772  }
773  // kv.second.clear();
774  kv.second = {};
775  }
776  }
777  for (auto& c : erase_what)
778  {
779  point_map_.erase(c);
780  points_.erase(c);
781  }
782  log_verbose("Spreading %d points until %f", points_.size(), new_time);
783  addEvent(Event::makeFireSpread(new_time));
784 }
