83 static void make_threshold(vector<double>* thresholds,
84  mt19937* mt,
85  const Day start_day,
86  const Day last_date,
87  double (*convert)(double value))
88 {
89  const auto total_weight = Settings::thresholdScenarioWeight() + Settings::thresholdDailyWeight() + Settings::thresholdHourlyWeight();
90  uniform_real_distribution<double> rand(0.0, 1.0);
91  const auto general = rand(*mt);
92  for (size_t i = start_day; i < MAX_DAYS; ++i)
93  {
94  const auto daily = rand(*mt);
95  for (auto h = 0; h < DAY_HOURS; ++h)
96  {
97  // generate no matter what so if we extend the time period the results
98  // for the first days don't change
99  const auto hourly = rand(*mt);
100  // only save if we're going to use it
101  if (i <= last_date)
102  {
103  // subtract from 1.0 because we want weight to make things more likely not less
104  // ensure we stay between 0 and 1
105  thresholds->at((i - start_day) * DAY_HOURS + h) =
106  convert(
107  max(0.0,
108  min(1.0,
109  1.0 - (Settings::thresholdScenarioWeight() * general + Settings::thresholdDailyWeight() * daily + Settings::thresholdHourlyWeight() * hourly) / total_weight)));
110  // thresholds->at((i - start_day) * DAY_HOURS + h) = 0.0;
111  }
112  }
113  }
114 }

218 void Scenario::evaluate(const Event& event)
219 {
220  log_check_fatal(event.time() < current_time_,
221  "Expected time to be > %f but got %f",
222  current_time_,
223  event.time());
224  const auto& p = event.cell();
225  switch (event.type())
226  {
227  case Event::FIRE_SPREAD:
228  scheduleFireSpread(event);
229  break;
230  case Event::SAVE:
231  saveObservers(event.time());
232  saveStats(event.time());
233  break;
234  case Event::NEW_FIRE:
235  // HACK: don't do this in constructor because scenario creates this in its constructor
236  points_[p].emplace_back(p.column(), p.row(), CELL_CENTER, CELL_CENTER);
237  if (fuel::is_null_fuel(event.cell()))
238  {
239  log_fatal("Trying to start a fire in non-fuel");
240  }
241  log_verbose("Starting fire at point (%f, %f) in fuel type %s at time %f",
242  p.column() + CELL_CENTER,
243  p.row() + CELL_CENTER,
244  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
245  event.time());
246  if (!survives(event.time(), event.cell(), event.timeAtLocation()))
247  {
248  const auto wx = weather(event.time());
249  log_info("Didn't survive ignition in %s with weather %f, %f",
250  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
251  wx->ffmc(),
252  wx->dmc());
253  // HACK: we still want the fire to have existed, so set the intensity of the origin
254  }
255  // fires start with intensity of 1
256  burn(event, 1);
257  scheduleFireSpread(event);
258  break;
259  case Event::END_SIMULATION:
260  log_verbose("End simulation event reached at %f", event.time());
261  endSimulation();
262  break;
263  default:
264  throw runtime_error("Invalid event type");
265  }
266 }

452 Scenario* Scenario::run(map<double, ProbabilityMap*>* probabilities)
453 {
454  log_check_fatal(ran(), "Scenario has already run");
455  log_verbose("Starting");
456  CriticalSection _(Model::task_limiter);
457  probabilities_ = probabilities;
458  for (auto time : save_points_)
459  {
460  // NOTE: these happen in this order because of the way they sort based on type
461  addEvent(Event::makeSave(static_cast<double>(time)));
462  }
463  if (nullptr == perimeter_)
464  {
465  addEvent(Event::makeNewFire(start_time_, cell(*start_cell_)));
466  }
467  else
468  {
469  intensity_->applyPerimeter(*perimeter_);
470  const auto& env = model().environment();
471  for (const auto& location : perimeter_->edge())
472  {
473  // const auto cell = env.cell(location.hash());
474  const auto cell = env.cell(location);
475  log_check_fatal(fuel::is_null_fuel(cell), "Null fuel in perimeter");
476  log_verbose("Adding point (%d, %d)",
477  cell.column() + CELL_CENTER,
478  cell.row() + CELL_CENTER);
479  points_[cell].emplace_back(cell.column(), cell.row(), CELL_CENTER, CELL_CENTER);
480  }
481  addEvent(Event::makeFireSpread(start_time_));
482  }
483  // HACK: make a copy of the event so that it still exists after it gets processed
484  // NOTE: sorted so that EventSaveASCII is always just before this
485  // Only run until last time we asked for a save for
486  log_verbose("Creating simulation end event for %f", last_save_);
487  addEvent(Event::makeEnd(last_save_));
488  // mark all original points as burned at start
489  for (auto& kv : points_)
490  {
491  const auto& location = kv.first;
492  // would be burned already if perimeter applied
493  if (canBurn(location))
494  {
495  const auto fake_event = Event::makeFireSpread(
496  start_time_,
497  static_cast<IntensitySize>(1),
498  location);
499  burn(fake_event, static_cast<IntensitySize>(1));
500  }
501  }
502  while (!scheduler_.empty())
503  {
504  evaluateNextEvent();
505  }
506  ++COMPLETED;
507  // HACK: use + to pull value out of atomic
508 #ifdef NDEBUG
509  log_info("[% d of % d] Completed with final size % 0.1f ha",
510  +COMPLETED,
511  +COUNT,
512  currentFireSize());
513 #else
514  // try to make output consistent if in debug mode
515  log_info("Completed with final size %0.1f ha",
516  currentFireSize());
517 #endif
518  ran_ = true;
519  // delete unburnable_;
520  // unburnable_ = nullptr;
521 #ifndef NDEBUG
522  static const size_t BufferSize = 64;
523  char buffer[BufferSize + 1] = {0};
524  sprintf(buffer,
525  "%03zu_%06ld_extinction",
526  id(),
527  simulation());
528  saveProbabilities(Settings::outputDirectory(), string(buffer), extinction_thresholds_);
529  sprintf(buffer,
530  "%03zu_%06ld_spread",
531  id(),
532  simulation());
533  saveProbabilities(Settings::outputDirectory(), string(buffer), spread_thresholds_by_ros_);
534 #endif
535  return this;
536 }

545 inline void Scenario::checkCondense(vector<InnerPos>& a)
546 {
547  if (a.size() > Settings::maxCellPoints())
548  {
549  size_t n_pos = 0;
550  auto n = numeric_limits<double>::min();
551  size_t ne_pos = 0;
552  auto ne = numeric_limits<double>::min();
553  size_t e_pos = 0;
554  auto e = numeric_limits<double>::min();
555  size_t se_pos = 0;
556  auto se = numeric_limits<double>::min();
557  size_t s_pos = 0;
558  auto s = numeric_limits<double>::max();
559  size_t sw_pos = 0;
560  auto sw = numeric_limits<double>::min();
561  size_t w_pos = 0;
562  auto w = numeric_limits<double>::max();
563  size_t nw_pos = 0;
564  auto nw = numeric_limits<double>::min();
565  for (size_t i = 0; i < a.size(); ++i)
566  {
567  const auto& p = a[i];
568  const auto x = p.sub_x - 0.5;
569  const auto y = p.sub_y - 0.5;
570  // NOTE: seems like it should be else if, but want to make sure
571  // all of these are initialized after the first point
572  if (y > n)
573  {
574  n_pos = i;
575  n = y;
576  }
577  if (y < s)
578  {
579  s_pos = i;
580  s = y;
581  }
582  const auto cur_ne = x + y;
583  const auto cur_sw = -x - y;
584  if (cur_ne > ne)
585  {
586  ne_pos = i;
587  ne = cur_ne;
588  }
589  if (cur_sw > sw)
590  {
591  sw_pos = i;
592  sw = cur_sw;
593  }
594  if (x > e)
595  {
596  e_pos = i;
597  e = x;
598  }
599  if (x < w)
600  {
601  w_pos = i;
602  w = x;
603  }
604  const auto cur_se = x - y;
605  const auto cur_nw = -x + y;
606  if (cur_se > se)
607  {
608  se_pos = i;
609  se = cur_se;
610  }
611  if (cur_nw > nw)
612  {
613  nw_pos = i;
614  nw = cur_nw;
615  }
616  }
617  // NOTE: temporarily use a set to ensure no duplicates
618  const std::set<InnerPos> result{
619  a[n_pos],
620  a[ne_pos],
621  a[e_pos],
622  a[se_pos],
623  a[s_pos],
624  a[sw_pos],
625  a[w_pos],
626  a[nw_pos]};
627  a = {};
628  a.assign(result.begin(), result.end());
629  }

654 CellIndex relativeIndex(const topo::Cell& for_cell, const topo::Cell& from_cell)
655 {
656  const auto r = for_cell.row();
657  const auto r_o = from_cell.row();
658  const auto c = for_cell.column();
659  const auto c_o = from_cell.column();
660  if (r == r_o)
661  {
662  // center row
663  // same cell, so source is 0
664  if (c == c_o)
665  {
666  return DIRECTION_NONE;
667  }
668  if (c < c_o)
669  {
670  //center right
671  return DIRECTION_E;
672  }
673  // else has to be c > c_o
674  // center left
675  return DIRECTION_W;
676  }
677  if (r < r_o)
678  {
679  // came from the row to the north
680  if (c == c_o)
681  {
682  // center top
683  return DIRECTION_N;
684  }
685  if (c < c_o)
686  {
687  // top right
688  return DIRECTION_NE;
689  }
690  // else has to be c > c_o
691  // top left
692  return DIRECTION_NW;
693  }
694  // else r > r_o
695  // came from the row to the south
696  if (c == c_o)
697  {
698  // center bottom
699  return DIRECTION_S;
700  }
701  if (c < c_o)
702  {
703  // bottom right

710 void Scenario::scheduleFireSpread(const Event& event)
711 {
712  const auto time = event.time();
713  //note("time is %f", time);
714  current_time_ = time;
715  const auto wx = weather(time);
716  // log_note("%d points", points_->size());
717  const auto this_time = util::time_index(time);
718  const auto next_time = static_cast<double>(this_time + 1) / DAY_HOURS;
719  // should be in minutes?
720  const auto max_duration = (next_time - time) * DAY_MINUTES;
721  //note("time is %f, next_time is %f, max_duration is %f",
722  // time,
723  // next_time,
724  // max_duration);
725  const auto max_time = time + max_duration / DAY_MINUTES;
726  if (wx->ffmc().asDouble() < minimumFfmcForSpread(time))
727  {
728  addEvent(Event::makeFireSpread(max_time));
729  log_verbose("Waiting until %f because of FFMC", max_time);
730  return;
731  }
732  if (current_time_index_ != this_time)
733  {
734  current_time_index_ = this_time;
735  // offsets_.clear();
736  // max_intensity_.clear();
737  offsets_ = {};
738  max_intensity_ = {};
739  max_ros_ = 0.0;
740  }
741  auto any_spread = false;
742  for (const auto& kv : points_)
743  {
744  const auto& location = kv.first;
745  // any cell that has the same fuel, slope, and aspect has the same spread
746  const auto key = location.key();
747  const auto seek_spreading = offsets_.find(key);
748  if (seek_spreading == offsets_.end())
749  {
750  // have not calculated spread for this cell yet
751  const SpreadInfo origin(*this, time, location, nd(time), wx);
752  // will be empty if invalid
753  offsets_.emplace(key, origin.offsets());
754  if (!origin.isNotSpreading())
755  {
756  any_spread = true;
757  max_ros_ = max(max_ros_, origin.headRos());
758  max_intensity_[key] = max(max_intensity_[key], origin.maxIntensity());
759  }
760  }
761  else
762  {
763  // already did the lookup so use the result
764  any_spread |= !seek_spreading->second.empty();
765  }
766  }
767  if (!any_spread || max_ros_ < Settings::minimumRos())
768  {
769  log_verbose("Waiting until %f", max_time);
770  addEvent(Event::makeFireSpread(max_time));
771  return;
772  }
773  //note("Max spread is %f, max_ros is %f",
774  // Settings::maximumSpreadDistance() * cellSize(),
775  // max_ros_);
776  const auto duration = ((max_ros_ > 0)
777  ? min(max_duration,
778  Settings::maximumSpreadDistance() * cellSize() / max_ros_)
779  : max_duration);
780  //note("Spreading for %f minutes", duration);
781  map<topo::Cell, CellIndex> sources{};
782  const auto new_time = time + duration / DAY_MINUTES;
783  for (auto& kv : points_)
784  {
785  const auto& location = kv.first;
786  const auto key = location.key();
787  auto& offsets = offsets_.at(key);
788  if (!offsets.empty())
789  {
790  for (auto& o : offsets)
791  {
792  // offsets in meters
793  const auto offset_x = o.x * duration;
794  const auto offset_y = o.y * duration;
795  const Offset offset{offset_x, offset_y};
796  //note("%f, %f", offset_x, offset_y);
797  for (auto& p : kv.second)
798  {
799  const InnerPos pos = p.add(offset);
800  const auto for_cell = cell(pos);
801  const auto source = relativeIndex(for_cell, location);
802  sources[for_cell] |= source;
803  if (!(fuel::is_null_fuel(for_cell) || (*unburnable_)[for_cell.hash()]))
804  {
805  //log_extensive("Adding point (%f, %f)", pos.x, pos.y);
806  point_map_[for_cell].emplace_back(pos);
807  }
808  }
809  }
810  }
811  else
812  {
813  // can't just keep existing points by swapping because something may have spread into this cell
814  auto& pts = point_map_[location];
815  pts.insert(pts.end(), kv.second.begin(), kv.second.end());
816  std::sort(pts.begin(), pts.end());
817  }
818  // kv.second.clear();
819  kv.second = {};
820  }
821  vector<topo::Cell> erase_what{};
822  for (auto& kv : point_map_)
823  {
824  if (!kv.second.empty())
825  {
826  auto& for_cell = kv.first;
827  if (canBurn(for_cell) && max_intensity_[for_cell.key()] > 0)
828  {
829  // HACK: make sure it can't round down to 0
830  const auto intensity = static_cast<IntensitySize>(max(
831  1.0,
832  max_intensity_[for_cell.key()]));
833  // HACK: just use the first cell as the source
834  const auto source = sources[for_cell];
835  const auto fake_event = Event::makeFireSpread(
836  new_time,
837  intensity,
838  for_cell,
839  source);
840  burn(fake_event, intensity);
841  }
842  // check if this cell is surrounded by burned cells or non-fuels
843  // if surrounded then just drop all the points inside this cell
844  if (!(*unburnable_)[for_cell.hash()])
845  {
846  if (!isSurrounded(for_cell) && survives(new_time, for_cell, new_time - arrival_[for_cell]))
847  {
848  checkCondense(kv.second);
849  //checkHull(for_cell, kv.second);
850  std::swap(points_[for_cell], kv.second);
851  }
852  else
853  {
854  // whether it went out or is surrounded just mark it as unburnable
855  (*unburnable_)[for_cell.hash()] = true;
856  erase_what.emplace_back(for_cell);
857  }
858  }
859  // kv.second.clear();
860  kv.second = {};
861  }
862  }
863  for (auto& c : erase_what)
864  {
865  point_map_.erase(c);
866  points_.erase(c);
867  }
868  log_verbose("Spreading %d points until %f", points_.size(), new_time);
869  addEvent(Event::makeFireSpread(new_time));
870 }
