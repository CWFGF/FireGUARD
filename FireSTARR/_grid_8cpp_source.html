25 string find_value(const string& key, const string& within)
26 {
27  const auto c = within.find(key);
28  if (c != string::npos)
29  {
30  const string str = &within.c_str()[c + string(key).length()];
31  return str.substr(0, str.find(' '));
32  }
33  return "";
34 }

58 double find_meridian(const string& proj4) noexcept
59 {
60  try
61  {
62  int zone{};
63  auto meridian = 0.0;
64  if (find_value("+zone=", proj4, &zone, &str_to_int))
65  {
66  meridian = topo::utm_central_meridian_deg(zone);
67  }
68  else if (!find_value("+lon_0=", proj4, &meridian, &str_to_double))
69  {
70  logging::fatal("Can't find meridian for grid");
71  }
72  return meridian;
73  }
74  catch (...)
75  {
76  return logging::fatal<double>("Unable to parse meridian in string '%s'",
77  proj4.c_str());
78  }
79 }

110 void GridBase::createPrj(const string& dir, const string& base_name) const
111 {
112  ofstream out;
113  out.open(dir + base_name + ".prj");
114  logging::debug(proj4_.c_str());
115  // HACK: use what we know is true for the grids that were generated and
116  // hope it's correct otherwise
117  const auto proj = find_value("+proj=", proj4_);
118  if (!proj.empty() && proj != "tmerc" && proj != "utm")
119  {
120  logging::fatal("Cannot create projection for non-transverse grid");
121  }
122  out << "Projection TRANSVERSE\n";
123  out << "Datum AI_CSRS\n";
124  out << "Spheroid GRS80\n";
125  out << "Units METERS\n";
126  out << "Zunits NO\n";
127  out << "Xshift 0.0\n";
128  out << "Yshift 0.0\n";
129  out << "Parameters \n";
130  out << "0.9996 /* scale factor at central meridian\n";
131  // HACK: assume we're always on an exact degree
132  out << static_cast<int>(meridian_) << " 0 0.0 /* longitude of central meridian\n";
133  out << " 0 0 0.0 /* latitude of origin\n";
134  out << "500000.0 /* false easting (meters)\n";
135  out << "0.0 /* false northing (meters)\n";
136  out.close();

138 // Use pair instead of Location, so we can go above max columns & rows
139 unique_ptr<Coordinates> GridBase::findCoordinates(const topo::Point& point,
140  const bool flipped) const
141 {
142  auto x = 0.0;
143  auto y = 0.0;
144  lat_lon_to_utm(point, this->zone(), &x, &y);
145  logging::note("Coordinates (%f, %f) converted to (%0.1f, %f, %f)",
146  point.latitude(),
147  point.longitude(),
148  this->zone(),
149  x,
150  y);
151  logging::debug("Lower left is (%f, %f)", this->xllcorner_, this->yllcorner_);
152  // convert coordinates into cell position
153  // use INT64, so we can see if coordinates are out of bounds
154  const auto actual_x = (x - this->xllcorner_) / this->cell_size_;
155  // these are already flipped across the y-axis on reading, so it's the same as for x now
156  auto actual_y = (y - this->yllcorner_) / this->cell_size_;
157  if (!flipped)
158  {
159  actual_y = static_cast<double>(this->rows_) - actual_y;
160  }
161  const auto column = static_cast<int64_t>(actual_x);
162  const auto row = static_cast<int64_t>(round(actual_y - 0.5));
163  if (0 > column || column >= columns_ || 0 > row || row >= rows_)
164  {
165  logging::debug("Returning nullptr from findCoordinates() for (%f, %f) => (%d, %d)",
166  actual_x,
167  actual_y,
168  column,
169  row);
170  return nullptr;
171  }
172  const auto sub_x = static_cast<SubSize>((actual_x - column) * 1000);
173  const auto sub_y = static_cast<SubSize>((actual_y - row) * 1000);
174  return make_unique<Coordinates>(static_cast<Idx>(row),
175  static_cast<Idx>(column),
176  sub_x,
177  sub_y);
178 }
