222  template <class R>
223  void saveToAsciiFile(const string& dir,
224  const string& base_name,
225  std::function<R(V value)> convert) const
226  {
227  Idx min_row = this->rows();
228  int16 max_row = 0;
229  Idx min_column = this->columns();
230  Idx max_column = 0;
231  for (const auto& kv : this->data)
232  {
233  const Idx r = kv.first.row();
234  const Idx c = kv.first.column();
235  min_row = min(min_row, r);
236  max_row = max(max_row, r);
237  min_column = min(min_column, c);
238  max_column = max(max_column, c);
239  }
240  // do this so that we take the center point when there's no data since it should
241  // stay the same if the grid is centered on the fire
242  if (min_row > max_row)
243  {
244  min_row = max_row = this->rows() / 2;
245  }
246  if (min_column > max_column)
247  {
248  min_column = max_column = this->columns() / 2;
249  }
250  logging::verbose("Lower left corner is (%d, %d)", min_column, min_row);
251  logging::verbose("Upper right corner is (%d, %d)", max_column, max_row);
252  const double xll = this->xllcorner() + min_column * this->cellSize();
253  // offset is different for y since it's flipped
254  const double yll = this->yllcorner() + (min_row) * this->
255  cellSize();
256  logging::verbose("Lower left corner is (%f, %f)", xll, yll);
257  // HACK: make sure it's always at least 1
258  const auto num_rows = static_cast<double>(max_row) - min_row + 1;
259  const auto num_columns = static_cast<double>(max_column) - min_column + 1;
260  ofstream out;
261  out.open(dir + base_name + ".asc");
262  write_ascii_header(out,
263  num_columns,
264  num_rows,
265  xll,
266  yll,
267  this->cellSize(),
268  static_cast<double>(this->noData()));
269  for (Idx ro = 0; ro < num_rows; ++ro)
270  {
271  // HACK: do this so we always get at least one pixel in output
272  // need to output in reverse order since (0,0) is bottom left
273  const Idx r = static_cast<Idx>(max_row) - ro;
274  for (Idx co = 0; co < num_columns; ++co)
275  {
276  const Location idx(static_cast<Idx>(r), static_cast<Idx>(min_column + co));
277  // HACK: use + here so that it gets promoted to a printable number
278  // prevents char type being output as characters
279  out << +((this->data.find(idx) != this->data.end())
280  ? convert(this->data.at(idx))
281  : this->noData()) << " ";
282  }
283  out << "\n";
284  }
285  out.close();
286  this->createPrj(dir, base_name);
287  }

317  [[nodiscard]] list<Location> makeEdge() const
318  {
319  list<Location> edge{};
320  for (const auto& kv : this->data)
321  {
322  auto loc = kv.first;
323  auto on_edge = false;
324  for (Idx r = -1; !on_edge && r <= 1; ++r)
325  {
326  const Idx row_index = loc.row() + r;
327  if (!(row_index < 0 || row_index >= this->rows()))
328  {
329  for (Idx c = -1; !on_edge && c <= 1; ++c)
330  {
331  const Idx col_index = loc.column() + c;
332  if (!(col_index < 0 || col_index >= this->columns())
333  && this->data.find(Location(row_index, col_index)) == this->data.end())
334  {
335  on_edge = true;
336  }
337  }
338  }
339  }
340  if (on_edge)
341  {
342  edge.push_back(loc);
343  }
344  }
345  logging::info("Created edge for perimeter with length %lu m",
346  static_cast<size_t>(this->cellSize() * edge.size()));
347  return edge;
348  }
