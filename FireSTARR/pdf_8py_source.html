356 def makePDF(fire, days, dates, mxd_names, wxshield, risk_maps, sim_output, out_file, scores):
357  """!
358  Create summary pdfs and merge everything together before compressing
359  @param fire Fire name to use
360  @param days Array of days simulation created output for
361  @param dates Array of dates simulation created output for
362  @param mxd_names List of mxds to use for cover page thumbnails and full page exports
363  @param wxshield WeatherSHIELD PDF export
364  @param risk_maps List of mxds to use for RamPART page thumbnails and full page exports
365  @param sim_output Array of simulation output, by line
366  @param out_file File to save final output to
367  @param scores Array of scores for simulation output dates
368  @return None
369  """
370  cover = makeCover(fire, days, dates, mxd_names, sim_output, scores)
371  rampart = makeRampart(fire, days, dates, risk_maps, sim_output, scores)
372  logging.info("Merging...")
373  merged = os.path.join(os.path.dirname(mxd_names[0]), fire + "_merged.pdf")
374  merger = PdfFileMerger()
375  pdfs = map(lambda _: os.path.splitext(_)[0] + ".pdf", mxd_names)
376  pdfs = [x for x in pdfs if os.path.exists(x)]
377  merger.append(cover, 'rb')
378  for pdf in pdfs:
379  merger.append(pdf, 'rb')
380  if wxshield is not None and os.path.exists(wxshield):
381  merger.append(wxshield, 'rb')
382  merger.append(rampart, 'rb')
383  risk_pdfs = map(lambda _: os.path.splitext(_)[0] + ".pdf", risk_maps)
384  risk_pdfs = [x for x in risk_pdfs if os.path.exists(x)]
385  for pdf in risk_pdfs:
386  merger.append(pdf, 'rb')
387  with open(merged, 'wb') as fout:
388  merger.write(fout)
389  merger.close()
390  del merger
391  compressPdf(merged, out_file)
