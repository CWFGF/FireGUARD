66  r = SetNull(r == 0.0, r)

68  m = SetNull(m == 0.0, m)

73  fn = arcpy.RasterToNumPyArray(misses, nodata_to_value=0).sum()
74  fp = arcpy.RasterToNumPyArray(false_positives, nodata_to_value=0).sum()
