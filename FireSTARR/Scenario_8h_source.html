411  // use Mike's table
412  const auto mc = wx->mcDmcPct();
413  if (100 > mc

415  || (119 >= mc && 4 > time_at_location)

418  || (218 >= mc && 1 > time_at_location))
419  {
420  return true;
421  }
422  // we can look by fuel type because the entire landscape shares the weather
423  return extinctionThreshold(time) < weather_->survivalProbability(
