432  // use Mike's table

436  || 119 >= mc && 4 > time_at_location
437  || 131 >= mc && 3 > time_at_location
438  || 145 >= mc && 2 > time_at_location
439  || 218 >= mc && 1 > time_at_location)
440  {
441  return true;
442  }
443  // we can look by fuel type because the entire landscape shares the weather
444  return extinctionThreshold(time) < weather_->survivalProbability(

446  cell.fuelCode());
447  }
