24 template <class T>
25 static vector<T> parse_list(string str, T (*convert)(const string& s))
26 {
27  vector<int> result{};
28  // static const vector<int> OUTPUT_DATE_OFFSETS = { 1, 2, 3, 7, 14, 28 };
29  logging::check_fatal(str[0] != '{', "Expected list starting with '{'");
30  istringstream iss(str.substr(1));
31  while (getline(iss, str, ','))
32  {
33  // need to make sure this isn't an empty list
34  if (0 != strcmp("}", str.c_str()))
35  {
36  result.push_back(convert(str));
37  }
38  }
39  return result;
40 }

371 string get_value(unordered_map<string, string>& settings, const string& key)
372 {
373  const auto found = settings.find(key);
374  if (found != settings.end())
375  {
376  auto result = found->second;
377  settings.erase(found);
378  return result;
379  }
380  logging::fatal("Missing setting for %s", key.c_str());
381  // HACK: use return to avoid compiler warning
382  static const string Invalid = "INVALID";
383  return Invalid;
384 }

387 SettingsImplementation::SettingsImplementation(const char* filename) noexcept
388 {
389  try
390  {
391  unordered_map<string, string> settings{};
392  ifstream in;
393  in.open(filename);
394  if (in.is_open())
395  {
396  string str;
397  logging::info("Reading settings from '%s'", filename);
398  while (getline(in, str))
399  {
400  istringstream iss(str);
401  if (getline(iss, str, '#'))
402  {
403  iss = istringstream(str);
404  }
405  if (getline(iss, str, '='))
406  {
407  const auto key = util::trim_copy(str);
408  getline(iss, str, '\n');
409  const auto value = util::trim_copy(str);
410  settings.emplace(key, value);
411  }
412  }
413  in.close();
414  }
415  weather_file_ = get_value(settings, "WEATHER_FILE");
416  raster_root_ = get_value(settings, "RASTER_ROOT");
417  fuel_lookup_table_ = get_value(settings, "FUEL_LOOKUP_TABLE");
418  // HACK: run into fuel consumption being too low if we don't have a minimum ros
419  static const auto MinRos = 0.05;
420  // HACK: make sure this is always > 0 so that we don't have to check
421  // specifically for 0 to avoid div error
422  minimum_ros_ = max(stod(get_value(settings, "MINIMUM_ROS")), MinRos);
423  maximum_spread_distance_ = stod(get_value(settings, "MAX_SPREAD_DISTANCE"));
424  minimum_ffmc_ = stod(get_value(settings, "MINIMUM_FFMC"));
425  minimum_ffmc_at_night_ = stod(get_value(settings, "MINIMUM_FFMC_AT_NIGHT"));
426  offset_sunrise_ = stod(get_value(settings, "OFFSET_SUNRISE"));
427  offset_sunset_ = stod(get_value(settings, "OFFSET_SUNSET"));
428  max_grade_ = stod(get_value(settings, "MAX_GRADE"));
429  confidence_level_ = stod(get_value(settings, "CONFIDENCE_LEVEL"));
430  maximum_time_seconds_ = stol(get_value(settings, "MAXIMUM_TIME"));
431  threshold_scenario_weight_ = stod(get_value(settings, "THRESHOLD_SCENARIO_WEIGHT"));
432  threshold_daily_weight_ = stod(get_value(settings, "THRESHOLD_DAILY_WEIGHT"));
433  threshold_hourly_weight_ = stod(get_value(settings, "THRESHOLD_HOURLY_WEIGHT"));
434  output_date_offsets_ = parse_list<int>(get_value(settings, "OUTPUT_DATE_OFFSETS"),
435  [](const string& s) { return stoi(s); });
436  default_percent_conifer_ = stoi(get_value(settings, "DEFAULT_PERCENT_CONIFER"));
437  default_percent_dead_fir_ = stoi(get_value(settings, "DEFAULT_PERCENT_DEAD_FIR"));
438  intensity_max_low_ = stoi(get_value(settings, "INTENSITY_MAX_LOW"));
439  intensity_max_moderate_ = stoi(get_value(settings, "INTENSITY_MAX_MODERATE"));
440  simulation_recheck_interval_ = static_cast<size_t>(stoi(
441  get_value(settings, "SIMULATION_RECHECK_INTERVAL")));
442  concurrent_simulation_rounds_ = static_cast<size_t>(stoi(
443  get_value(settings, "CONCURRENT_SIMULATION_ROUNDS")));
444  minimum_simulation_rounds_ = static_cast<size_t>(stoi(
445  get_value(settings, "MINIMUM_SIMULATION_ROUNDS")));
446  max_cell_points_ = static_cast<size_t>(stoi(get_value(settings, "MAX_CELL_POINTS")));
447  max_date_offset_ = *std::max_element(output_date_offsets_.begin(),
448  output_date_offsets_.end());
449  if (!settings.empty())
450  {
451  logging::warning("Unused settings in settings file %s", filename);
452  for (const auto& kv : settings)
453  {
454  logging::warning("%s = %s", kv.first.c_str(), kv.second.c_str());
455  }
456  }
457  }
458  catch (...)
459  {
460  std::terminate();
461  }
462 }
