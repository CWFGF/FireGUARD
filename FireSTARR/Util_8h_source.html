337 template <typename T, typename V>
338 T binary_find(const T lower,
339  const T upper,
340  const double value,
341  const std::function<V(T)>& fct)
342 {
343  const auto mid = lower + (upper - lower) / 2;
344  if (lower == upper)
345  {
346  return lower;
347  }
348  if (fct(mid) < value)
349  {
350  return binary_find(lower, mid, value, fct);
351  }
352  return binary_find(mid + 1, upper, value, fct);
353 }
364 template <typename T, typename V>
365 T binary_find_checked(const T lower,

368  const std::function<V(T)>& fct)
369 {

371  {

373  }

375  {
376  return upper;

378  return binary_find(lower, upper, value, fct);
