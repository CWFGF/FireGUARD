98 void Model::readWeather(const string& filename,
99  const bool for_actuals,
100  const wx::FwiWeather& yesterday,
101  const double latitude)
102 {
103  map<size_t, map<Day, wx::FwiWeather>> wx{};
104  map<Day, struct tm> dates{};
105  auto min_date = numeric_limits<Day>::max();
106  ifstream in;
107  in.open(filename);
108  logging::check_fatal(!in.is_open(),
109  "Could not open input weather file %s",
110  filename.c_str());
111  if (in.is_open())
112  {
113  string str;
114  logging::info("Reading scenarios from '%s'", filename.c_str());
115  // read header line
116  getline(in, str);
117  // get rid of whitespace
118  str.erase(std::remove(str.begin(), str.end(), ' '), str.end());
119  str.erase(std::remove(str.begin(), str.end(), '\n'), str.end());
120  str.erase(std::remove(str.begin(), str.end(), '\r'), str.end());
121  constexpr auto expected_header =
122  "Scenario,Date,APCP,TMP,RH,WS,WD,FFMC,DMC,DC,ISI,BUI,FWI";
123  logging::check_fatal(0 != str.compare(expected_header),
124  "Input CSV must have columns in this order:\n'%s'\n but got:\n'%s'",
125  expected_header,
126  str.c_str());
127  while (getline(in, str))
128  {
129  istringstream iss(str);
130  if (getline(iss, str, ',') && !str.empty())
131  {
132  // HACK: ignore date and just worry about relative order??
133  // Scenario
134  logging::verbose("Scenario is %s", str.c_str());
135  auto cur = ACTUALS;
136  if (!(str == "actual"))
137  {
138  try
139  {
140  cur = static_cast<size_t>(-stoi(str));
141  }
142  catch (std::exception&)
143  {
144  // HACK: somehow stoi() is still getting empty strings
145  logging::fatal("Error reading weather file %s", filename.c_str());
146  }
147  }
148  if (for_actuals && cur != ACTUALS)
149  {
150  // HACK: rely on actuals always being first
151  break;
152  }
153  if (wx.find(cur) == wx.end())
154  {
155  logging::debug("Loading scenario %d...", cur);
156  wx.emplace(cur, map<Day, wx::FwiWeather>());
157  }
158  auto& s = wx.at(cur);
159  struct tm t
160  {
161  };
162  util::read_date(&iss, &str, &t);
163  const auto ticks = mktime(&t);
164  if (1 == cur)
165  {
166  logging::debug("Date '%s' is %ld and calculated jd is %d",
167  str.c_str(),
168  ticks,
169  t.tm_yday);
170  if (!s.empty() && t.tm_yday < min_date)
171  {
172  logging::fatal(
173  "Weather input file crosses year boundary or dates are not sequential");
174  }
175  }
176  min_date = min(min_date, static_cast<Day>(t.tm_yday));
177  logging::check_fatal(s.find(static_cast<Day>(t.tm_yday)) != s.end(),
178  "Day already exists");
179  s.emplace(static_cast<Day>(t.tm_yday), wx::FwiWeather(&iss, &str));
180  if (s.find(static_cast<Day>(t.tm_yday)) == s.end())
181  {
182  dates.emplace(static_cast<Day>(t.tm_yday), t);
183  }
184  }
185  }
186  in.close();
187  }
188  if (!for_actuals)
189  {
190  // HACK: it's added for everything right now, but alone if just doing actuals
191  const auto seek = wx.find(ACTUALS);
192  if (seek != wx.end())
193  {
194  wx.erase(ACTUALS);
195  }
196  }
197  // HACK: add yesterday into everything
198  for (auto& kv : wx)
199  {
200  auto& s = kv.second;
201  logging::check_fatal(s.find(static_cast<Day>(min_date - 1)) != s.end(),
202  "Day already exists");
203  // recalculate everything with startup indices
204  map<Day, wx::FwiWeather> new_wx{};
205  auto last_ffmc = yesterday.ffmc();
206  auto last_dmc = yesterday.dmc();
207  auto last_dc = yesterday.dc();
208  auto apcp = yesterday.apcp();
209  for (auto& kv2 : s)
210  {
211  auto& day = kv2.first;
212  auto& w = kv2.second;
213  const firestarr::wx::Ffmc ffmc(w.tmp(), w.rh(), w.wind().speed(), apcp, last_ffmc);
214  const auto t = dates[day];
215  const auto month = t.tm_mon + 1;
216  const firestarr::wx::Dmc dmc(w.tmp(), w.rh(), apcp, last_dmc, month, latitude);
217  const firestarr::wx::Dc dc(w.tmp(), apcp, last_dc, month, latitude);
218  const firestarr::wx::Isi isi(w.wind().speed(), ffmc);
219  const firestarr::wx::Bui bui(dmc, dc);
220  const firestarr::wx::Fwi fwi(isi, bui);
221  const firestarr::wx::FwiWeather n(w.tmp(),
222  w.rh(),
223  w.wind(),
224  apcp,
225  ffmc,
226  dmc,
227  dc,
228  isi,
229  bui,
230  fwi);
231  new_wx.emplace(static_cast<Day>(day), n);
232  }
233  new_wx.emplace(static_cast<Day>(min_date - 1), yesterday);
234  kv.second = new_wx;
235  }
236  const auto fuel_lookup = sim::Settings::fuelLookup();
237  // loop through and try to find duplicates
238  for (const auto& kv : wx)
239  {
240  const auto k = kv.first;
241  const auto s = kv.second;
242  if (wx_.find(k) == wx_.end())
243  {
244  const auto w = make_shared<wx::FireWeather>(fuel_lookup.usedFuels(), s);
245  wx_.emplace(k, w);
246  }
247  }

249 void Model::findStarts(const Location location)
250 {
251  logging::error("Trying to start a fire in non-fuel");
252  Idx range = 1;
253  // HACK: should always be centered in the grid
254  while (starts_.empty() && (range < (MAX_COLUMNS / 2)))
255  {
256  for (Idx x = -range; x <= range; ++x)
257  {
258  for (Idx y = -range; y <= range; ++y)
259  {
260  // make sure we only look at the outside of the box
261  if (1 == range || abs(x) == range || abs(y) == range)
262  {
263  // const auto loc = env_->cell(location.hash() + (y * MAX_COLUMNS) + x);
264  const auto loc = env_->cell(Location(location.row() + y, location.column() + x));
265  if (!fuel::is_null_fuel(loc))
266  {
267  starts_.push_back(make_shared<topo::Cell>(cell(loc)));
268  }
269  }
270  }
271  }
272  ++range;
273  }
274  logging::check_fatal(0 == starts_.size(), "Fuel grid is empty");
275  logging::info("Using %d start locations:", starts_.size());
276  for (const auto& s : starts_)
277  {
278  logging::info("\t%d, %d", s->row(), s->column());
279  }

284  const string& perim,
285  const size_t size)
286 {
287  const Location location(std::get<0>(coordinates), std::get<1>(coordinates));

296  perimeter_ = make_shared<topo::Perimeter>(

301  }
302  // figure out where the fire can exist
303  if (nullptr != perimeter_ && !perimeter_->burned().empty())

306  // we have a perimeter to start from

314  if (nullptr != perimeter_)
315  {
316  logging::check_fatal(!perimeter_->burned().empty(),
317  "Not using perimeter so it should be empty");
318  logging::note("Using fire perimeter results in empty fire - changing to use point");
319  perimeter_ = nullptr;
320  }
321  logging::note("Fire starting with size %0.1f ha", env_->cellSize() / 100.0);

323  if (0 == size && fuel::is_null_fuel(cell(location)))
324  {
325  findStarts(location);
326  }

328  {
329  starts_.push_back(make_shared<topo::Cell>(cell(location)));
330  }
331  }
332  logging::note("Creating %ld streams x %ld location%s = %ld scenarios",

338 Iteration Model::readScenarios(const topo::StartPoint& start_point,
339  const double start,
340  const bool save_intensity,
341  const Day start_day,
342  const Day last_date)
343 {
344  vector<Scenario*> result{};
345  auto saves = Settings::outputDateOffsets();
346  const auto setup_scenario = [&result, save_intensity, &saves](Scenario* scenario)
347  {
348  if (save_intensity)
349  {
350  scenario->registerObserver(new IntensityObserver(*scenario, ""));
351  scenario->registerObserver(new ArrivalObserver(*scenario));
352  scenario->registerObserver(new SourceObserver(*scenario));
353  }
354  // FIX: this should be relative to the start date, not the weather start date
355  for (const auto& i : saves)
356  {
357  scenario->addSaveByOffset(i);
358  }
359  result.push_back(scenario);
360  };
361  for (const auto& kv : wx_)
362  {
363  const auto id = kv.first;
364  const auto cur_wx = kv.second;
365  if (nullptr != perimeter_)
366  {
367  setup_scenario(new Scenario(this,
368  id,
369  cur_wx.get(),
370  start,
371  perimeter_,
372  start_point,
373  start_day,
374  last_date));
375  }
376  else
377  {
378  for (const auto& cur_start : starts_)
379  {
380  // should always have at least the day before the fire in the weather stream
381  setup_scenario(new Scenario(this,
382  id,
383  cur_wx.get(),
384  start,
385  cur_start,
386  start_point,
387  start_day,
388  last_date));
389  }
390  }
391  }
392  return Iteration(result);
393 }

465 bool add_statistics(const size_t i,
466  vector<double>* means,
467  vector<double>* pct,
468  const Model& model,
469  const util::SafeVector& v)
470 {
471  const auto sizes = v.getValues();
472  logging::check_fatal(sizes.empty(), "No sizes at end of simulation");
473  const util::Statistics s{sizes};
474  static_cast<void>(util::insert_sorted(pct, s.percentile(95)));
475  static_cast<void>(util::insert_sorted(means, s.mean()));
476  if (model.isOutOfTime())
477  {
478  logging::note(
479  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
480  i,
481  Settings::maximumTimeSeconds());
482  return false;
483  }
484  return true;
485 }
497 size_t runs_required(const size_t i,
498  const vector<double>* means,
499  const vector<double>* pct,
500  const Model& model)
501 {
502  if (model.isOutOfTime())
503  {
504  logging::note(
505  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
506  i,
507  Settings::maximumTimeSeconds());
508  return 0;
509  }
510  const auto for_means = util::Statistics{*means};
511  const auto for_pct = util::Statistics{*pct};
512  if (!(i < Settings::minimumSimulationRounds()
513  || !for_means.isConfident(Settings::confidenceLevel())
514  || !for_pct.isConfident(Settings::confidenceLevel())))
515  {
516  return 0;
517  }
518  const auto left = max(for_means.runsRequired(i, Settings::confidenceLevel()),
519  for_pct.runsRequired(i, Settings::confidenceLevel()));
520  return left;
521 }
522 map<double, ProbabilityMap*> Model::runIterations(const topo::StartPoint& start_point,
523  const double start,
524  const Day start_day,
525  const bool save_intensity,
526  const bool for_actuals)
527 {
528  auto last_date = start_day;
529  for (const auto i : Settings::outputDateOffsets())
530  {
531  last_date = max(static_cast<Day>(start_day + i), last_date);
532  }
533  // use independent seeds so that if we remove one threshold it doesn't affect the other
534  std::seed_seq seed_spread{0.0, start, start_point.latitude(), start_point.longitude()};
535  std::seed_seq seed_extinction{1.0, start, start_point.latitude(), start_point.longitude()};
536  mt19937 mt_spread(seed_spread);
537  mt19937 mt_extinction(seed_extinction);
538  vector<double> means{};
539  vector<double> pct{};
540  size_t i = 0;
541  auto iterations = readScenarios(start_point,
542  start,
543  save_intensity,
544  start_day,
545  last_date);
546  // put probability maps into map
547  const auto saves = iterations.savePoints();
548  const auto started = iterations.startTime();
549  auto probabilities = make_prob_map(*this,
550  saves,
551  for_actuals,
552  started,
553  0,
554  Settings::intensityMaxLow(),
555  Settings::intensityMaxModerate(),
556  numeric_limits<int>::max());
557  auto runs_left = Settings::minimumSimulationRounds();
558  const auto recheck_interval = Settings::simulationRecheckInterval();
559  // HACK: just do this here so that we know it happened
560  //iterations.reset(&mt_extinction, &mt_spread);
561  if (Settings::runAsync())
562  {
563  vector<Iteration> all_iterations{};
564  all_iterations.push_back(std::move(iterations));
565  size_t runs = 0;
566  const auto max_concurrent = Settings::concurrentSimulationRounds();
567  while (runs_left > 0)
568  {
569  // run what's left, up to min rounds at a time
570  const auto cur_runs = runs_left;
571  const auto total_runs = i + cur_runs;
572  while (min(cur_runs, max_concurrent) > all_iterations.size())
573  {
574  all_iterations.push_back(readScenarios(start_point,
575  start,
576  save_intensity,
577  start_day,
578  last_date));
579  }
580  auto threads = vector<std::thread>{};
581  auto all_scenarios = vector<Scenario*>{};
582  // need at least this many results
583  threads.reserve(max_concurrent);
584  for (size_t k = 0; k < min(cur_runs, all_iterations.size()); ++k)
585  {
586  logging::debug("Running iteration [%d of %d]", runs + 1, total_runs);
587  all_iterations.at(k).reset(&mt_extinction, &mt_spread);
588  auto& s = all_iterations.at(k).getScenarios();
589  all_scenarios.insert(all_scenarios.end(), s.begin(), s.end());
590  ++runs;
591  }
592  // // sort in run so that they still get the same extinction thresholds as when unsorted
593  // std::sort(all_scenarios.begin(),
594  // all_scenarios.end(),
595  // [](Scenario* lhs, Scenario* rhs) noexcept
596  // {
597  // // sort so that scenarios with highest DSRs are at the front
598  // return lhs->weightedDsr() > rhs->weightedDsr();
599  // });
600  for (auto s : all_scenarios)
601  {
602  threads.emplace_back(&Scenario::run,
603  s,
604  &probabilities);
605  }
606  for (auto& t : threads)
607  {
608  t.join();
609  }
610  for (size_t k = 0; k < min(cur_runs, all_iterations.size()); ++k)
611  {
612  auto& iteration = all_iterations.at(k);
613  auto final_sizes = iteration.finalSizes();
614  ++i;
615  //note("Adding %d", i);
616  if (!add_statistics(i, &means, &pct, *this, final_sizes))
617  {
618  // ran out of time
619  return probabilities;
620  }
621  }
622  runs_left = runs_required(i, &means, &pct, *this);
623  logging::note("Need another %d iterations", runs_left);
624  if (runs_left > recheck_interval)
625  {
626  runs_left = recheck_interval;
627  logging::note("Capping at %d iterations before checking again", runs_left);
628  }
629  }
630  // everything should be done when this section ends
631  }
632  else
633  {
634  logging::note("Running in synchronous mode");
635  while (runs_left > 0)
636  {
637  const auto cur_runs = runs_left;
638  size_t k = 0;
639  while (k < cur_runs)
640  {
641  logging::note("Running iteration %d", i + 1);
642  iterations.reset(&mt_extinction, &mt_spread);
643  for (auto s : iterations.getScenarios())
644  {
645  // s->run_fake(&probabilities);
646  // iterations.reset(&mt_extinction, &mt_spread);
647  s->run(&probabilities);
648  }
649  ++i;
650  if (!add_statistics(i, &means, &pct, *this, iterations.finalSizes()))
651  {
652  // ran out of time
653  return probabilities;
654  }
655  ++k;
656  }
657  runs_left = runs_required(i, &means, &pct, *this);
658  logging::note("Need another %d iterations", runs_left);
659  if (runs_left > recheck_interval)
660  {
661  runs_left = recheck_interval;
662  logging::note("Capping at %d iterations before checking again", runs_left);
663  }
664  }
665  }
666  return probabilities;
667 }

687  std::get<0>(*position),
688  std::get<1>(*position));
689  logging::info("Position is (%d, %d)", std::get<0>(*position), std::get<1>(*position));
690  const Location location{std::get<0>(*position), std::get<1>(*position)};
691  Model model(start_point, &env);
692  auto x = 0.0;
693  auto y = 0.0;
694  const auto zone = lat_lon_to_utm(start_point, &x, &y);
695  logging::note("UTM coordinates are: %d %d %d",
696  zone,
697  static_cast<int>(x),
698  static_cast<int>(y));
699  logging::note("Grid has size (%d, %d)", env.rows(), env.columns());
700  logging::note("Fire start position is cell (%d, %d)",
701  location.row(),
702  location.column());
703  model.readWeather(weather_input, for_actuals, yesterday, start_point.latitude());
704  if (model.wx_.empty())
705  {
706  logging::fatal("No weather provided");
707  }
708  const auto w = model.wx_.begin()->second;
709  logging::debug("Have weather from day %d to %d", w->minDate(), w->maxDate());
710  const auto numDays = (w->maxDate() - w->minDate() + 1);
711  const auto needDays = Settings::maxDateOffset();
712  if (numDays < needDays)
713  {
714  logging::fatal("Not enough weather to proceed - have %d days but looking for %d", numDays, needDays);
715  }
716  model.makeStarts(*position, start_point, yesterday, perimeter, size);
717  auto start_hour = ((start_time.tm_hour + (static_cast<double>(start_time.tm_min) / 60))
718  / DAY_HOURS);
719  // HACK: round to 2 digits so that we can keep test output the same
720  start_hour = static_cast<double>(static_cast<int>(start_hour * 100)) / 100;
721  const auto start = start_time.tm_yday + start_hour;
722  const auto start_day = static_cast<Day>(start);
723  auto probabilities =
724  model.runIterations(start_point, start, start_day, save_intensity, for_actuals);
725  logging::note("Ran %d simulations", Scenario::completed());
726  show_probabilities(probabilities);
727  auto final_time = numeric_limits<double>::min();
728  for (const auto by_time : probabilities)
729  {
730  const auto time = by_time.first;
731  final_time = max(final_time, time);
732  const auto prob = by_time.second;
733  prob->saveAll(model, start_time, for_actuals, time, start_day);
734  }
735  for (const auto& kv : probabilities)
736  {
737  delete kv.second;
738  }
739  return 0;
740 }
741 }
742 }
