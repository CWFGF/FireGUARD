182 //******************************************************************************************
183 // Function Name: DayLength
184 // Description: Calculates latitude/date dependent day length for DMC calculation
185 // Parameters:
186 // Latitude is latitude in decimal degrees of calculation location
187 // Month is integer (1..12) value of month of year for calculation
188 //
189 //******************************************************************************************
190 static constexpr double day_length(const double latitude, const int month) noexcept
191 {
192  //'''/* Day Length Arrays for four different latitude ranges '*/
193  //''/*
194  //' Use four ranges which respectively span:
195  //' - 90N to 33 N
196  //' - 33 N to 0
197  //' - 0 to -30
198  //' - -30 to -90
200  if ((latitude <= 90) && (latitude > 33.0))
201  {
202  return DAY_LENGTH46_N.at(static_cast<size_t>(month) - 1);
203  }
204  if ((latitude <= 33.0) && (latitude > 15.0))
205  {
206  return DAY_LENGTH20_N.at(static_cast<size_t>(month) - 1);
207  }
208  if ((latitude <= 15.0) && (latitude > -15.0))
209  {
210  return 9.0;
211  }
212  if ((latitude <= -15.0) && (latitude > -30.0))
213  {
214  return DAY_LENGTH20_S.at(static_cast<size_t>(month) - 1);
215  }
216  if ((latitude <= -30.0) && (latitude >= -90.0))
217  {
218  return DAY_LENGTH40_S.at(static_cast<size_t>(month) - 1);
219  }
220  return logging::fatal<double>("Unable to calculate DayLength");
221 }

223 // Function Name: FFMC
224 // Description: Calculates today's Fine Fuel Moisture Code
225 // Parameters:
226 // temperature is the 12:00 LST temperature in degrees celsius
227 // rh is the 12:00 LST relative humidity in %
228 // wind is the 12:00 LST wind speed in kph
229 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
230 // ffmc_previous is the previous day's FFMC
231 //******************************************************************************************
232 static double calculate_ffmc(const Temperature& temperature,
233  const RelativeHumidity& rh,
234  const Speed& wind,
235  const AccumulatedPrecipitation& rain,
236  const Ffmc& ffmc_previous) noexcept
237 {
238  //'''/* 1 '*/
239  auto mo = 147.2 * (101.0 - ffmc_previous.asDouble()) / (59.5 + ffmc_previous.asDouble());
240  if (rain.asDouble() > 0.5)
241  {
242  //'''/* 2 '*/
243  const auto rf = rain.asDouble() - 0.5;
244  //'''/* 3a '*/
245  auto mr = mo + 42.5 * rf * (exp(-100.0 / (251.0 - mo))) * (1 - exp(-6.93 / rf));
246  if (mo > 150.0)
247  {
248  //'''/* 3b '*/
249  mr += 0.0015 * util::pow_int<2>(mo - 150.0) * sqrt(rf);
250  }
251  if (mr > 250.0)
252  {
253  mr = 250.0;
254  }
255  mo = mr;
256  }
257  //'''/* 4 '*/
258  const auto find_m = [&temperature, &rh, &wind, &mo]() noexcept
259  {
260  const auto ed = 0.942 * pow(rh.asDouble(), 0.679) + 11.0 * exp((rh.asDouble() - 100.0) / 10.0) + 0.18 * (21.1 - temperature.asDouble()) * (1.0 - exp(-0.115 * rh.asDouble()));
261  if (mo > ed)
262  {
263  //'''/* 6a '*/
264  const auto ko = 0.424 * (1.0 - pow(rh.asDouble() / 100.0, 1.7)) + 0.0694 * sqrt(wind.asDouble()) * (1.0 - util::pow_int<8>(rh.asDouble() / 100.0));
265  //'''/* 6b '*/
266  const auto kd = ko * 0.581 * exp(0.0365 * temperature.asDouble());
267  //'''/* 8 '*/
268  return ed + (mo - ed) * pow(10.0, -kd);
269  }
270  //'''/* 5 '*/
271  const auto ew = 0.618 * pow(rh.asDouble(), 0.753) + 10.0 * exp((rh.asDouble() - 100.0) / 10.0) + 0.18 * (21.1 - temperature.asDouble()) * (1.0 - exp(-0.115 * rh.asDouble()));
272  if (mo < ew)
273  {
274  //'''/* 7a '*/
275  const auto kl = 0.424 * (1.0 - pow((100.0 - rh.asDouble()) / 100.0, 1.7)) + 0.0694 * sqrt(wind.asDouble()) * (1 - util::pow_int<8>((100.0 - rh.asDouble()) / 100.0));
276  //'''/* 7b '*/
277  const auto kw = kl * 0.581 * exp(0.0365 * temperature.asDouble());
278  //'''/* 9 '*/
279  return ew - (ew - mo) * pow(10.0, -kw);
280  }
281  return mo;
282  };
283  const auto m = find_m();
284  //'''/* 10 '*/
285  return (59.5 * (250.0 - m) / (147.2 + m));
286 }

295 //******************************************************************************************
296 // Function Name: DMC
297 // Description: Calculates today's Duff Moisture Code
298 // Parameters:
299 // temperature is the 12:00 LST temperature in degrees celsius
300 // rh is the 12:00 LST relative humidity in %
301 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
302 // dmc_previous is the previous day's DMC
303 // latitude is the latitude in decimal degrees of the location for calculation
304 // month is the month of Year (1..12) for the current day's calculations.
305 //******************************************************************************************
306 static double calculate_dmc(const Temperature& temperature,
307  const RelativeHumidity& rh,
308  const AccumulatedPrecipitation& rain,
309  const Dmc& dmc_previous,
310  const int month,
311  const double latitude) noexcept
312 {
313  auto previous = dmc_previous.asDouble();
314  if (rain.asDouble() > 1.5)
315  {
316  //'''/* 11 '*/
317  const auto re = 0.92 * rain.asDouble() - 1.27;
318  //'''/* 12 '*/
319  const auto mo = 20.0 + exp(5.6348 - previous / 43.43);
320  const auto b = (previous <= 33.0)
321  ? //'''/* 13a '*/
322  100.0 / (0.5 + 0.3 * previous)
323  : ((previous <= 65.0)
324  ? //'''/* 13b '*/
325  14.0 - 1.3 * (log(previous))
326  : //'''/* 13c '*/
327  6.2 * log(previous) - 17.2);
328  //'''/* 14 '*/
329  const auto mr = mo + 1000.0 * re / (48.77 + b * re);
330  //'''/* 15 '*/
331  const auto pr = 244.72 - 43.43 * log(mr - 20.0);
332  previous = max(pr, 0.0);
333  }
334  const auto k = (temperature.asDouble() > -1.1)
335  ? 1.894 * (temperature.asDouble() + 1.1) * (100.0 - rh.asDouble()) * day_length(latitude, month) * 0.000001
336  : 0.0;
337  //'''/* 17 '*/
338  return (previous + 100.0 * k);
339 }

349 //******************************************************************************************
350 // Function Name: DC
351 // Description: Calculates today's Drought Code
352 // Parameters:
353 // temperature is the 12:00 LST temperature in degrees celsius
354 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
355 // dc_previous is the previous day's DC
356 // latitude is the latitude in decimal degrees of the location for calculation
357 // month is the month of Year (1..12) for the current day's calculations.
358 //******************************************************************************************
359 static double calculate_dc(const Temperature& temperature,
360  const AccumulatedPrecipitation& rain,
361  const Dc& dc_previous,
362  const int month,
363  const double latitude) noexcept
364 {
365  auto previous = dc_previous.asDouble();
366  if (rain.asDouble() > 2.8)
367  {
368  //'/* 18 */
369  const auto rd = 0.83 * (rain.asDouble()) - 1.27;
370  //'/* 19 */
371  const auto qo = 800.0 * exp(-previous / 400.0);
372  //'/* 20 */
373  const auto qr = qo + 3.937 * rd;
374  //'/* 21 */
375  const auto dr = 400.0 * log(800.0 / qr);
376  previous = (dr > 0.0) ? dr : 0.0;
377  }
378  const auto lf = day_length_factor(latitude, month - 1);
379  //'/* 22 */
380  const auto v = max(0.0,
381  (temperature.asDouble() > -2.8)
382  ? 0.36 * (temperature.asDouble() + 2.8) + lf
383  : lf);
384  //'/* 23 */
385  const auto d = previous + 0.5 * v;
386  // HACK: don't allow negative values
387  return max(0.0, d);
388 }

404 static double calculate_isi(const Speed& wind, const Ffmc& ffmc) noexcept
405 {
406  //'''/* 24 '*/
407  const auto f_wind = exp(0.05039 * wind.asDouble());

447  if (dmc.asDouble() <= 0.4 * dc.asDouble())

449  // HACK: this isn't normally part of it, but it's division by 0 without this

551 FwiWeather read(istringstream* iss, string* str)
552 {
553  // APCP
554  util::getline(iss, str, ',');
555  logging::extensive("APCP is %s", str->c_str());
556  const AccumulatedPrecipitation apcp(stod(str));
557  // TMP
558  util::getline(iss, str, ',');
559  logging::extensive("TMP is %s", str->c_str());
560  const Temperature tmp(stod(str));
561  // RH
562  util::getline(iss, str, ',');
563  logging::extensive("RH is %s", str->c_str());
564  const RelativeHumidity rh(stod(str));
565  // WS
566  util::getline(iss, str, ',');
567  logging::extensive("WS is %s", str->c_str());
568  const Speed ws(stod(str));
569  // WD
570  util::getline(iss, str, ',');
571  logging::extensive("WD is %s", str->c_str());
572  const Direction wd(stod(str), false);
573  // FFMC
574  util::getline(iss, str, ',');
575  logging::extensive("FFMC is %s", str->c_str());
576  const Ffmc ffmc(stod(str));
577  // DMC
578  util::getline(iss, str, ',');
579  logging::extensive("DMC is %s", str->c_str());
580  const Dmc dmc(stod(str));
581  // DC
582  util::getline(iss, str, ',');
583  logging::extensive("DC is %s", str->c_str());
584  const Dc dc(stod(str));
585  // ISI
586  util::getline(iss, str, ',');
587  logging::extensive("ISI is %s", str->c_str());
588  const Isi isi(stod(str), ws, ffmc);
589  // BUI
590  util::getline(iss, str, ',');
591  logging::extensive("BUI is %s", str->c_str());
592  const Bui bui(stod(str), dmc, dc);
593  // FWI
594  util::getline(iss, str, ',');
595  logging::extensive("FWI is %s", str->c_str());
596  const Fwi fwi(stod(str), isi, bui);
597  const Wind wind(wd, ws);
598  return {tmp, rh, wind, apcp, ffmc, dmc, dc, isi, bui, fwi};
599 }

605 {
