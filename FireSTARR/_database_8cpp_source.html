76 map<wx::WeatherModel, map<int, vector<wx::Weather>>, wx::ModelCompare>
77 Database::readModel(wstring query)
78 {
79  SQLSMALLINT num_results;
80  logging::check_fatal(SQLExecDirect(h_stmt_, &query[0], SQL_NTS) != SQL_SUCCESS,
81  "Could not read from database");
82  SQLNumResultCols(h_stmt_, &num_results);
83  assert(num_results > 0);
84  map<wx::WeatherModel, map<int, vector<wx::Weather>>, wx::ModelCompare> results{};
85  auto ret_code = SQLFetch(h_stmt_);
86  while (ret_code != SQL_NO_DATA_FOUND)
87  {
88  cur_column_ = 1;
89  auto k = wx::WeatherModel(this);
90  const auto fk = results.find(k);
91  if (fk == results.end())
92  {
93  static const map<int, vector<wx::Weather>> EmptyMap{};
94  results.emplace(k, EmptyMap);
95  }
96  else
97  {
98  k = fk->first;
99  }
100  auto m = stoi(getString());
101  if (results.at(k).find(m) == results.at(k).end())
102  {
103  results.at(k).emplace(m, vector<wx::Weather>());
104  }
105  results.at(k).at(m).emplace_back(this);
106  ret_code = SQLFetch(h_stmt_);
107  }
108  SQLFreeStmt(h_stmt_, SQL_CLOSE);
109  return results;
110 }
