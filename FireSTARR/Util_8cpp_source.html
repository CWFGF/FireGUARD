27 void read_directory(const string& name, vector<string>* v, const string& match)
28 {
29  string full_match = ".*/" + match;
30  logging::verbose(("Matching '" + full_match + "'").c_str());
31  static const std::regex re(full_match, std::regex_constants::icase);
32  for (const auto& entry : fs::directory_iterator(name))
33  {
34  logging::verbose(("Checking if file: " + entry.path().string()).c_str());
35  if (fs::is_regular_file(entry))
36  {
37  logging::extensive(("Checking regex match: " + entry.path().string()).c_str());
38  if (std::regex_match(entry.path().string(), re))
39  {
40  v->push_back(entry.path());
41  }
42  }
43  }
44 }

94 void make_directory_recursive(const char* dir) noexcept
95 {
96  char tmp[256];
97  snprintf(tmp, sizeof tmp, "%s", dir);
98  const auto len = strlen(tmp);
99  if (tmp[len - 1] == '/')
100  tmp[len - 1] = 0;
101  for (auto p = tmp + 1; *p; ++p)
102  if (*p == '/')
103  {
104  *p = 0;
105  make_directory(tmp);
106  *p = '/';
107  }
108  make_directory(tmp);
