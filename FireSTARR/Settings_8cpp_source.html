22 static vector<T> parse_list(string str, T (*convert)(const string& s))
23 {
24  vector<int> result{};
25  // static const vector<int> OUTPUT_DATE_OFFSETS = { 1, 2, 3, 7, 14, 28 };
26  logging::check_fatal(str[0] != '{', "Expected list starting with '{'");
27  istringstream iss(str.substr(1));
28  while (getline(iss, str, ','))
29  {
30  // need to make sure this isn't an empty list
31  if (0 != strcmp("}", str.c_str()))
32  {
33  result.push_back(convert(str));
34  }
35  }
36  return result;
37 }

422 string get_value(unordered_map<string, string>& settings, const string& key)
423 {
424  const auto found = settings.find(key);
425  if (found != settings.end())
426  {
427  auto result = found->second;
428  settings.erase(found);
429  return result;
430  }
431  logging::fatal("Missing setting for %s", key.c_str());
432  // HACK: use return to avoid compiler warning
433  static const string Invalid = "INVALID";
434  return Invalid;
435 }
436 SettingsImplementation::SettingsImplementation(const char* filename) noexcept
437 {
438  try
439  {
440  unordered_map<string, string> settings{};
441  ifstream in;
442  in.open(filename);
443  if (in.is_open())
444  {
445  string str;
446  logging::info("Reading settings from '%s'", filename);
447  while (getline(in, str))
448  {
449  istringstream iss(str);
450  if (getline(iss, str, '#'))
451  {
452  iss = istringstream(str);
453  }
454  if (getline(iss, str, '='))
455  {
456  const auto key = util::trim_copy(str);
457  getline(iss, str, '\n');
458  const auto value = util::trim_copy(str);
459  settings.emplace(key, value);
460  }
461  }
462  in.close();
463  }
464  weather_file_ = get_value(settings, "WEATHER_FILE");
465  raster_root_ = get_value(settings, "RASTER_ROOT");
466  fuel_lookup_table_file_ = get_value(settings, "FUEL_LOOKUP_TABLE");
467  // HACK: run into fuel consumption being too low if we don't have a minimum ros
468  static const auto MinRos = 0.05;
469  // HACK: make sure this is always > 0 so that we don't have to check
470  // specifically for 0 to avoid div error
471  minimum_ros_ = max(stod(get_value(settings, "MINIMUM_ROS")), MinRos);
472  maximum_spread_distance_ = stod(get_value(settings, "MAX_SPREAD_DISTANCE"));
473  minimum_ffmc_ = stod(get_value(settings, "MINIMUM_FFMC"));
474  minimum_ffmc_at_night_ = stod(get_value(settings, "MINIMUM_FFMC_AT_NIGHT"));
475  offset_sunrise_ = stod(get_value(settings, "OFFSET_SUNRISE"));
476  offset_sunset_ = stod(get_value(settings, "OFFSET_SUNSET"));
477  max_grade_ = stod(get_value(settings, "MAX_GRADE"));
478  confidence_level_ = stod(get_value(settings, "CONFIDENCE_LEVEL"));
479  maximum_time_seconds_ = stol(get_value(settings, "MAXIMUM_TIME"));
480  threshold_scenario_weight_ = stod(get_value(settings, "THRESHOLD_SCENARIO_WEIGHT"));
481  threshold_daily_weight_ = stod(get_value(settings, "THRESHOLD_DAILY_WEIGHT"));
482  threshold_hourly_weight_ = stod(get_value(settings, "THRESHOLD_HOURLY_WEIGHT"));
483  output_date_offsets_ = parse_list<int>(get_value(settings, "OUTPUT_DATE_OFFSETS"),
484  [](const string& s)
485  {
486  return stoi(s);
487  });
488  default_percent_conifer_ = stoi(get_value(settings, "DEFAULT_PERCENT_CONIFER"));
489  default_percent_dead_fir_ = stoi(get_value(settings, "DEFAULT_PERCENT_DEAD_FIR"));
490  intensity_max_low_ = stoi(get_value(settings, "INTENSITY_MAX_LOW"));
491  intensity_max_moderate_ = stoi(get_value(settings, "INTENSITY_MAX_MODERATE"));
492  simulation_recheck_interval_ = static_cast<size_t>(stoi(
493  get_value(settings, "SIMULATION_RECHECK_INTERVAL")));
494  concurrent_simulation_rounds_ = static_cast<size_t>(stoi(
495  get_value(settings, "CONCURRENT_SIMULATION_ROUNDS")));
496  minimum_simulation_rounds_ = static_cast<size_t>(stoi(
497  get_value(settings, "MINIMUM_SIMULATION_ROUNDS")));
498  max_cell_points_ = static_cast<size_t>(stoi(get_value(settings, "MAX_CELL_POINTS")));
499  max_date_offset_ = *std::max_element(output_date_offsets_.begin(),
500  output_date_offsets_.end());
501  if (!settings.empty())
502  {
503  logging::warning("Unused settings in settings file %s", filename);
504  for (const auto& kv : settings)
505  {
506  logging::warning("%s = %s", kv.first.c_str(), kv.second.c_str());
507  }
508  }
509  }
510  catch (...)
511  {
512  std::terminate();
513  }
