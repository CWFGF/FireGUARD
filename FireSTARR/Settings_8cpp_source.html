23 static vector<T> parse_list(string str, T (*convert)(const string& s))
24 {
25  vector<int> result{};
26  // static const vector<int> OUTPUT_DATE_OFFSETS = { 1, 2, 3, 7, 14, 28 };
27  logging::check_fatal(str[0] != '{', "Expected list starting with '{'");
28  istringstream iss(str.substr(1));
29  while (getline(iss, str, ','))
30  {
31  // need to make sure this isn't an empty list
32  if (0 != strcmp("}", str.c_str()))
33  {
34  result.push_back(convert(str));
35  }
36  }
37  return result;
38 }

423 string get_value(unordered_map<string, string>& settings, const string& key)
424 {
425  const auto found = settings.find(key);
426  if (found != settings.end())
427  {
428  auto result = found->second;
429  settings.erase(found);
430  return result;
431  }
432  logging::fatal("Missing setting for %s", key.c_str());
433  // HACK: use return to avoid compiler warning
434  static const string Invalid = "INVALID";
435  return Invalid;
436 }
437 SettingsImplementation::SettingsImplementation(const char* filename) noexcept
438 {
439  try
440  {
441  unordered_map<string, string> settings{};
442  ifstream in;
443  in.open(filename);
444  if (in.is_open())
445  {
446  string str;
447  logging::info("Reading settings from '%s'", filename);
448  while (getline(in, str))
449  {
450  istringstream iss(str);
451  if (getline(iss, str, '#'))
452  {
453  iss = istringstream(str);
454  }
455  if (getline(iss, str, '='))
456  {
457  const auto key = util::trim_copy(str);
458  getline(iss, str, '\n');
459  const auto value = util::trim_copy(str);
460  settings.emplace(key, value);
461  }
462  }
463  in.close();
464  }
465  weather_file_ = get_value(settings, "WEATHER_FILE");
466  raster_root_ = get_value(settings, "RASTER_ROOT");
467  fuel_lookup_table_file_ = get_value(settings, "FUEL_LOOKUP_TABLE");
468  // HACK: run into fuel consumption being too low if we don't have a minimum ros
469  static const auto MinRos = 0.05;
470  // HACK: make sure this is always > 0 so that we don't have to check
471  // specifically for 0 to avoid div error
472  minimum_ros_ = max(stod(get_value(settings, "MINIMUM_ROS")), MinRos);
473  maximum_spread_distance_ = stod(get_value(settings, "MAX_SPREAD_DISTANCE"));
474  minimum_ffmc_ = stod(get_value(settings, "MINIMUM_FFMC"));
475  minimum_ffmc_at_night_ = stod(get_value(settings, "MINIMUM_FFMC_AT_NIGHT"));
476  offset_sunrise_ = stod(get_value(settings, "OFFSET_SUNRISE"));
477  offset_sunset_ = stod(get_value(settings, "OFFSET_SUNSET"));
478  max_grade_ = stod(get_value(settings, "MAX_GRADE"));
479  confidence_level_ = stod(get_value(settings, "CONFIDENCE_LEVEL"));
480  maximum_time_seconds_ = stol(get_value(settings, "MAXIMUM_TIME"));
481  threshold_scenario_weight_ = stod(get_value(settings, "THRESHOLD_SCENARIO_WEIGHT"));
482  threshold_daily_weight_ = stod(get_value(settings, "THRESHOLD_DAILY_WEIGHT"));
483  threshold_hourly_weight_ = stod(get_value(settings, "THRESHOLD_HOURLY_WEIGHT"));
484  output_date_offsets_ = parse_list<int>(get_value(settings, "OUTPUT_DATE_OFFSETS"),
485  [](const string& s)
486  {
487  return stoi(s);
488  });
489  default_percent_conifer_ = stoi(get_value(settings, "DEFAULT_PERCENT_CONIFER"));
490  default_percent_dead_fir_ = stoi(get_value(settings, "DEFAULT_PERCENT_DEAD_FIR"));
491  intensity_max_low_ = stoi(get_value(settings, "INTENSITY_MAX_LOW"));
492  intensity_max_moderate_ = stoi(get_value(settings, "INTENSITY_MAX_MODERATE"));
493  simulation_recheck_interval_ = static_cast<size_t>(stoi(
494  get_value(settings, "SIMULATION_RECHECK_INTERVAL")));
495  concurrent_simulation_rounds_ = static_cast<size_t>(stoi(
496  get_value(settings, "CONCURRENT_SIMULATION_ROUNDS")));
497  minimum_simulation_rounds_ = static_cast<size_t>(stoi(
498  get_value(settings, "MINIMUM_SIMULATION_ROUNDS")));
499  max_cell_points_ = static_cast<size_t>(stoi(get_value(settings, "MAX_CELL_POINTS")));
500  max_date_offset_ = *std::max_element(output_date_offsets_.begin(),
501  output_date_offsets_.end());
502  if (!settings.empty())
503  {
504  logging::warning("Unused settings in settings file %s", filename);
505  for (const auto& kv : settings)
506  {
507  logging::warning("%s = %s", kv.first.c_str(), kv.second.c_str());
508  }
509  }
510  }
511  catch (...)
512  {
513  std::terminate();
514  }
515 }
