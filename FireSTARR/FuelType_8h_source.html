244 class FuelBase
245  : public FuelType
246 {
247 public:
248  ~FuelBase() = default;
257  constexpr FuelBase(const FuelCodeSize& code,
258  const char* name,
259  const bool can_crown,
260  const Duff* duff_ffmc,
261  const Duff* duff_dmc)
262  : FuelType(code, name, can_crown),
263  duff_ffmc_(duff_ffmc),
264  duff_dmc_(duff_dmc)
265  {
266  }
267  FuelBase(FuelBase&& rhs) noexcept = delete;
268  FuelBase(const FuelBase& rhs) = delete;
269  FuelBase& operator=(FuelBase&& rhs) noexcept = delete;
270  FuelBase& operator=(const FuelBase& rhs) = delete;
277  [[nodiscard]] double crownFractionBurned(const double rss,
278  const double rso) const noexcept override
279  {
280  return max(0.0, 1.0 - exp(-0.230 * (rss - rso)));
281  }
287  [[nodiscard]] double probabilityPeat(const double mc_fraction) const noexcept override
288  {
289  // Anderson table 1
290  constexpr auto pb = bulkDensity();
291  // Anderson table 1
292  constexpr auto fi = inorganicPercent();
293  constexpr auto pi = fi * pb;
294  // Inorganic ratio
295  constexpr auto ri = fi / (1 - fi);
296  constexpr auto const_part = -19.329 + 1.7170 * ri + 23.059 * pi;
297  // Anderson eq 1
298  return 1 / (1 + exp(17.047 * mc_fraction / (1 - fi) + const_part));
299  }
305  [[nodiscard]] double survivalProbability(const wx::FwiWeather& wx) const noexcept
306  override
307  {
308  // divide by 100 since we need moisture ratio
309  // IFERROR(((1 / (1 + EXP($G$43 + $I$43 *
310  // (Q$44 * $O$43 + $N$43)))) -
311  // (1 / (1 + EXP($G$43 + $I$43 * (2.5 * $O$43 + $N$43)))))
312  // / (1 / (1 + EXP($G$43 + $I$43 * $N$43))), 0)
313  // HACK: use same constants for all fuels because they seem to work nicer than
314  // using the ratios, but they change anyway because of the other fuel attributes
315  static const auto WFfmc = 0.25;
316  static const auto WDmc = 1.0;
317  static const auto RatioHartford = 0.5;
318  static const auto RatioFrandsen = 1.0 - RatioHartford;
319  static const auto RatioAspen = 0.5;
320  static const auto RatioFuel = 1.0 - RatioAspen;
321  const auto mc_ffmc = wx.mcFfmc() * WFfmc + WDmc;
322  static const auto McFfmcSaturated = 2.5 * WFfmc + WDmc;
323  static const auto McDmc = WDmc;
324  const auto prob_ffmc_peat = probabilityPeat(mc_ffmc);
325  const auto prob_ffmc_peat_saturated = probabilityPeat(McFfmcSaturated);
326  const auto prob_ffmc_peat_zero = probabilityPeat(McDmc);
327  const auto prob_ffmc_peat_weighted = (prob_ffmc_peat - prob_ffmc_peat_saturated) / prob_ffmc_peat_zero;
328  const auto prob_ffmc = duffFfmcType()->probabilityOfSurvival(mc_ffmc * 100);
329  const auto prob_ffmc_saturated = duffFfmcType()->probabilityOfSurvival(
330  McFfmcSaturated * 100);
331  const auto prob_ffmc_zero = duffFfmcType()->probabilityOfSurvival(McDmc);
332  const auto prob_ffmc_weighted = (prob_ffmc - prob_ffmc_saturated) / prob_ffmc_zero;
333  const auto term_otway = exp(-3.11 + 0.12 * wx.dmc().asDouble());
334  const auto prob_otway = term_otway / (1 + term_otway);
335  const auto mc_pct = wx.mcDmcPct() * dmcRatio() + wx.mcFfmcPct() * ffmcRatio();
336  const auto prob_weight_ffmc = duffFfmcType()->probabilityOfSurvival(mc_pct);
337  const auto prob_weight_ffmc_peat = probabilityPeat(mc_pct / 100);
338  const auto prob_weight_dmc = duffDmcType()->probabilityOfSurvival(wx.mcDmcPct());
339  const auto prob_weight_dmc_peat = probabilityPeat(wx.mcDmc());
340  // chance of survival is 1 - chance of it not surviving in every fuel
341  const auto tot_prob = 1 - (1 - prob_ffmc_peat_weighted) * (1 - prob_ffmc_weighted) * ((1 - prob_otway) * RatioAspen + ((1 - prob_weight_ffmc_peat) * RatioHartford + (1 - prob_weight_ffmc) * RatioFrandsen) * ((1 - prob_weight_dmc_peat) * RatioHartford + (1 - prob_weight_dmc) * RatioFrandsen) * RatioFuel);
342  return tot_prob;
343  }
348  [[nodiscard]] static constexpr double bulkDensity()
349  {
350  return BulkDensity / 1000.0;
351  }
356  [[nodiscard]] static constexpr double inorganicPercent()
357  {
358  return InorganicPercent / 100.0;
359  }
364  [[nodiscard]] static constexpr double duffDepth()
365  {
366  return DuffDepth / 10.0;
367  }
372  [[nodiscard]] constexpr const Duff* duffDmcType() const
373  {
374  return duff_dmc_;
375  }
380  [[nodiscard]] constexpr const Duff* duffFfmcType() const
381  {
382  return duff_ffmc_;
383  }
388  [[nodiscard]] static constexpr double ffmcRatio()
389  {
390  return 1 - dmcRatio();
391  }
396  [[nodiscard]] static constexpr double dmcRatio()
397  {
398  return (duffDepth() - DUFF_FFMC_DEPTH) / duffDepth();
399  }
400 private:
404  const Duff* duff_ffmc_;
408  const Duff* duff_dmc_;
409 };
