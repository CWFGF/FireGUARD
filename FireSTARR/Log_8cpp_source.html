51 void output(const int log_level, const char* format, va_list* args) noexcept
52 {
53  if (Log::getLogLevel() > log_level)
54  {
55  return;
56  }
57  try
58  {
59  // NOTE: create string first so that entire line writes
60  // (otherwise threads might mix lines)
61  const string tmp;
62  stringstream iss(tmp);
63 #ifdef NDEBUG
64  const time_t now = time(nullptr);
65  auto buf = localtime(&now);
66  iss << put_time(buf, "[%F %T] ");
67 #endif
68  // try to make output consistent if in debug mode
69  iss << LOG_LABELS[log_level];
70  static char buffer[1024]{0};
71  vsprintf(buffer, format, *args);
72  iss << buffer << "\n";
73  {
74  lock_guard<mutex> lock(mutex_);
75  cout << iss.str();
76  out_stream_ << iss.str();
77  }
78  }
79  catch (...)
80  {
81  std::terminate();
82  }
83 }
