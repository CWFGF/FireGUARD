63 void output(const int log_level, const char* format, va_list* args) noexcept
64 {
65  if (Log::getLogLevel() > log_level)
66  {
67  return;
68  }
69  try
70  {
71  // NOTE: create string first so that entire line writes
72  // (otherwise threads might mix lines)
73  const string tmp;
74  stringstream iss(tmp);
75 #ifdef NDEBUG
76  const time_t now = time(nullptr);
77  auto buf = localtime(&now);
78  iss << put_time(buf, "[%F %T] ");
79 #endif
80  // try to make output consistent if in debug mode
81  iss << LOG_LABELS[log_level];
82  static char buffer[1024]{0};
83  vsprintf(buffer, format, *args);
84  iss << buffer << "\n";
85  {
86  lock_guard<mutex> lock(mutex_);
87  cout << iss.str();
88  out_stream_ << iss.str();
89  }
90  }
91  catch (...)
92  {
93  std::terminate();
94  }
95 }
