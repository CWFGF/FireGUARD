39 void Log::output(const char* name, const char* format, va_list* args) noexcept
40 {
41  try
42  {
43  const time_t now = time(nullptr);
44  auto buf = localtime(&now);
45  // NOTE: create string first so that entire line writes
46  // (otherwise threads might mix lines)
47  const string tmp;
48  stringstream iss(tmp);
49  iss << put_time(buf, "[%F %T] ") << name;
50  static char buffer[1024]{0};
51  vsprintf(buffer, format, *args);
52  iss << buffer << "\n";
53  cout << iss.str();
54  }
55  catch (...)
56  {
57  std::terminate();
58  }
59 }
