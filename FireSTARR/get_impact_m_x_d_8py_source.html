49  copyMXD = getImpactMXDName(fire_prefix, run_output, fire, extent, perim)
50  shutil.copyfile(os.path.join(Settings.HOME_DIR, "mxds", "impact.mxd"), copyMXD)
51  theMXD = arcpy.mapping.MapDocument(copyMXD)
52  impact =Settings.RAMPART_MASK.format('high', findSuffix(findFuelRaster(sim_output)))
53  setDataSource(theMXD, "*Potential Impact*", impact)
54  txtAssumptions = ("Potential Impact is based on staff assessment of impact on a 1 - 10 scale,\n"
55  "representing the loss from the damage or disruption from various types of RA data\n"
56  "burned with a 4000+ kW/m head fire intensity. Each cell is coloured based on the\n"
57  "amount and type of RA within it. Potential Impact is the 'worst case'\n"
58  "and the risk calculation on subsequent maps is a downscaling of this (e.g., burned\n"
59  "with lower fire intensity and/or lower probability). Warning: use with caution and\n"
60  "awareness of the assumptions and simplifications. Impact does not include indirect\n"
61  "(e.g. smoke) or long-term effects. A hectare must burn in the model to show any risk;\n"
62  "limitations of FireSTARR apply. RA may be missing from the GIS database, affecting\n"
63  "results.")
64  setCommon(theMXD,
65  fire,
66  txtAssumptions,
67  " ",
68  run_output,
69  fire,
70  perim,
71  extent,
72  impact)
73  arcpy.RefreshActiveView()
74  theMXD.save()
75  return copyMXD, theMXD
