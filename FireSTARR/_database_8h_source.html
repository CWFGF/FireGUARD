142  template <class T>
143  [[nodiscard]] vector<T> readList(wstring query)
144  {
145  SQLSMALLINT num_results;
146  logging::check_fatal(SQLExecDirect(h_stmt_, &query[0], SQL_NTS) != SQL_SUCCESS,
147  "Could not read from database");
148  SQLNumResultCols(h_stmt_, &num_results);
149  assert(num_results > 0);
150  vector<T> results{};
151  results.reserve(static_cast<size_t>(num_results));
152  auto ret_code = SQLFetch(h_stmt_);
153  while (ret_code != SQL_NO_DATA_FOUND)
154  {
155  cur_column_ = 1;
156  results.emplace_back(this);
157  ret_code = SQLFetch(h_stmt_);
158  }
159  SQLFreeStmt(h_stmt_, SQL_CLOSE);
160  return results;
161  }
