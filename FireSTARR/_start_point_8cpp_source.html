27 static T fix_range(T value, T min_value, T max_value) noexcept
28 {
29  while (value < min_value)
30  {
31  value += max_value;
32  }
33  while (value >= max_value)
34  {
35  value -= max_value;
36  }
37  return value;
38 }

49 static double sunrise_sunset(const int jd,
50  const double latitude,
51  const double longitude,
52  const bool for_sunrise) noexcept
53 {
54  static const auto Zenith = util::to_radians(96);
55  static const auto LocalOffset = -5;
56  const auto t_hour = for_sunrise ? 6 : 18;
57  // http://edwilliams.org/sunrise_sunset_algorithm.htm
58  const auto lng_hour = longitude / 15;
59  const auto t = jd + (t_hour - lng_hour) / 24;
60  const auto m = 0.9856 * t - 3.289;
61  const auto l = fix_degrees(
62  m + 1.916 * sin(util::to_radians(m)) + 0.020 * sin(
63  util::to_radians(2 * m)) + 282.634);
64  auto ra = fix_degrees(util::to_degrees(atan(0.91764 * tan(util::to_radians(l)))));
65  const auto l_quadrant = floor(l / 90) * 90;
66  const auto ra_quadrant = floor(ra / 90) * 90;
67  ra += l_quadrant - ra_quadrant;
68  ra /= 15;
69  const auto sin_dec = 0.39782 * sin(util::to_radians(l));
70  const auto cos_dec = cos(asin(sin_dec));
71  const auto cos_h = (cos(Zenith) - sin_dec * sin(util::to_radians(latitude))) / (cos_dec
72  *
73  cos(util::to_radians(latitude)));
74  if (cos_h > 1)
75  {
76  // sun never rises
77  return for_sunrise ? -1 : 25;
78  }
79  if (cos_h < -1)
80  {
81  // sun never sets
82  return for_sunrise ? 25 : -1;
83  }
84  auto h = util::to_degrees(acos(cos_h));
85  if (for_sunrise)
86  {
87  h = 360 - h;
88  }
89  h /= 15;
90  const auto mean_t = h + ra - 0.06571 * t - 6.622;
91  const auto ut = mean_t - lng_hour;
92  return fix_hours(ut + LocalOffset);
93 }

104 static array<tuple<double, double>, MAX_DAYS> make_days(
105  const double latitude,
106  const double longitude) noexcept
107 {
108  array<tuple<double, double>, MAX_DAYS> days{};
109  array<double, MAX_DAYS> day_length_hours{};
110  for (size_t i = 0; i < day_length_hours.size(); ++i)
111  {
112  days[i] = make_tuple(
113  fix_hours(
114  sunrise(static_cast<int>(i), latitude, longitude) + sim::Settings::
115  offsetSunrise()),
116  fix_hours(
117  sunset(static_cast<int>(i),
118  latitude,
119  longitude) - sim::Settings::offsetSunset()));
120  day_length_hours[i] = get<1>(days[i]) - get<0>(days[i]);
121  }
122  return days;
123 }
