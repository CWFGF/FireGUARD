66 bool ModelCompare::operator()(const WeatherModel& x, const WeatherModel& y) const noexcept
67 {
68  const auto cs = x.name().compare(y.name());
69  if (0 == cs)
70  {
71  if (x.generated() == y.generated())
72  {
73  // HACK: these should always be equal, so just check that they are
74  assert(x.point().latitude() == y.point().latitude()
75  && x.point().longitude() == y.point().longitude()
76  && x.distanceFrom() == y.distanceFrom());
77  if (x.point().latitude() == y.point().latitude())
78  {
79  if (x.point().longitude() == y.point().longitude())
80  {
81  return x.distanceFrom() < y.distanceFrom();
82  }
83  return x.point().longitude() < y.point().longitude();
84  }
85  return x.point().latitude() < y.point().latitude();
86  }
87  return x.generated() < y.generated();
88  }
89  return -1 == cs;
90 }
