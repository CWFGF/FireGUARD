159  logging::info("Reading file %s", filename.c_str());
160 #ifndef NDEBUG
161  auto min_value = std::numeric_limits<int16>::max();
162  auto max_value = std::numeric_limits<int16>::min();
163 #endif
164  const GridBase grid_info = read_header<T>(tif, gtif);
165  int tile_width;
166  int tile_length;
167  TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
168  TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
169  auto actual_rows = grid_info.calculateRows();
170  auto actual_columns = grid_info.calculateColumns();
171  const auto coordinates = grid_info.findFullCoordinates(point, true);
172  auto min_column = max(static_cast<FullIdx>(0),
173  static_cast<FullIdx>(std::get<1>(*coordinates) - static_cast<FullIdx>(MAX_COLUMNS) / static_cast<FullIdx>(2)));
174  if (min_column + MAX_COLUMNS >= actual_columns)
175  {
176  min_column = max(static_cast<FullIdx>(0), actual_columns - MAX_COLUMNS);
177  }
178  // make sure we're at the start of a tile
179  const auto tile_column = tile_width * static_cast<FullIdx>(min_column / tile_width);
180  const auto max_column = static_cast<FullIdx>(min(min_column + MAX_COLUMNS - 1, actual_columns));
181  logging::check_fatal(min_column < 0, "Column can't be less than 0");
182  logging::check_fatal(max_column - min_column > MAX_COLUMNS, "Can't have more than %d columns", MAX_COLUMNS);
183  logging::check_fatal(max_column > actual_columns, "Can't have more than actual %d columns", actual_columns);
184  auto min_row = max(static_cast<FullIdx>(0),
185  static_cast<FullIdx>(std::get<0>(*coordinates) - static_cast<FullIdx>(MAX_ROWS) / static_cast<FullIdx>(2)));
186  if (min_row + MAX_COLUMNS >= actual_rows)
187  {
188  min_row = max(static_cast<FullIdx>(0), actual_rows - MAX_ROWS);
189  }
190  const auto tile_row = tile_width * static_cast<FullIdx>(min_row / tile_width);
191  const auto max_row = static_cast<FullIdx>(min(min_row + MAX_ROWS - 1, actual_rows));
192  logging::check_fatal(min_row < 0, "Row can't be less than 0 but is %d", min_row);
193  logging::check_fatal(max_row - min_row > MAX_ROWS, "Can't have more than %d rows but have %d", MAX_ROWS, max_row - min_row);
194  logging::check_fatal(max_row > actual_rows, "Can't have more than actual %d rows", actual_rows);
195  T no_data = convert(grid_info.nodata(), grid_info.nodata());
196  vector<T> values(static_cast<size_t>(MAX_ROWS) * MAX_COLUMNS, no_data);
197  logging::warning("%s: malloc start", filename.c_str());
198  const auto buf = _TIFFmalloc(TIFFTileSize(tif));
199  logging::warning("%s: read start", filename.c_str());
200  const tsample_t smp{};
201  logging::debug("Want to clip grid to (%d, %d) => (%d, %d)", min_row, min_column, max_row, max_column);
202  for (auto h = tile_row; h <= max_row; h += tile_length)
203  {
204  // const auto y_min = static_cast<FullIdx>(max(static_cast<FullIdx>(0), min_row - h));
205  // const auto y_limit = min(static_cast<FullIdx>(tile_length),
206  // min(static_cast<FullIdx>(actual_rows), max_row) - h);
207  for (auto w = tile_column; w <= max_column; w += tile_width)
208  {
209  TIFFReadTile(tif, buf, static_cast<uint32>(w), static_cast<uint32>(h), 0, smp);
210  // const auto x_min = static_cast<FullIdx>(max(static_cast<FullIdx>(0), min_column - w));
211  // const auto x_limit = min(static_cast<FullIdx>(tile_width),
212  // min(actual_columns,
213  // max_column)
214  // - w);
215  // for (auto y = y_min; y < y_limit; ++y)
216  for (auto y = 0; (y < tile_length) && (y + h <= max_row); ++y)
217  {
218  // read in so that (0, 0) has a hash of 0
219  // const FullIdx cur_row = (static_cast<FullIdx>(h) + y);
220  // const FullIdx i = static_cast<FullIdx>(MAX_ROWS) - (cur_row - min_row + 1);
221  // for (auto x = x_min; x < x_limit; ++x)
222  const auto y_row = static_cast<HashSize>((h - min_row) + y);
223  const auto actual_row = (max_row - min_row) - y_row;
224  if (actual_row >= 0 && actual_row < MAX_ROWS)
225  {
226  for (auto x = 0; (x < tile_width) && (x + w <= max_column); ++x)
227  {
228  // const auto cur_hash = static_cast<HashSize>(i) * MAX_COLUMNS + (w - min_column + 1) + x;
229  // const auto offset = (y - (static_cast<HashSize>(y / tile_length) * tile_length)) + x;
230  const auto offset = y * tile_width + x;
231  const auto actual_column = ((w - min_column) + x);
232  if (actual_column >= 0 && actual_column < MAX_ROWS)
233  {
234  const auto cur_hash = actual_row * MAX_COLUMNS + actual_column;
235  auto cur = *(static_cast<int16*>(buf) + offset);
236 #ifndef NDEBUG
237  min_value = min(cur, min_value);
238  max_value = max(cur, max_value);
239 #endif
240  values.at(cur_hash) = convert(cur, grid_info.nodata());
241  }
242  }
243  }
244  }
245  }
246  }
247  logging::warning("%s: read end", filename.c_str());
248  _TIFFfree(buf);
249  logging::warning("%s: free end", filename.c_str());
250  const auto new_xll = grid_info.xllcorner() + (min_column * grid_info.cellSize());
251  const auto new_yll = grid_info.yllcorner()
252  + (static_cast<double>(actual_rows) - max_row)
253  * grid_info.cellSize();
254  logging::check_fatal(new_yll < grid_info.yllcorner(),
255  "New yllcorner is outside original grid");
256  logging::verbose("Translated lower left is (%f, %f) from (%f, %f)",
257  new_xll,
258  new_yll,
259  grid_info.xllcorner(),
260  grid_info.yllcorner());
261  const auto num_rows = max_row - min_row + 1;
262  const auto num_columns = max_column - min_column + 1;
263  auto result = new ConstantGrid<T, V>(grid_info.cellSize(),
264  num_rows,
265  num_columns,
266  no_data,
267  grid_info.nodata(),
268  new_xll,
269  new_yll,
270  new_xll + (num_columns + 1) * grid_info.cellSize(),
271  new_yll + (num_rows + 1) * grid_info.cellSize(),
272  string(grid_info.proj4()),
273  std::move(values));
274  auto new_location = result->findCoordinates(point, true);
275  logging::check_fatal(nullptr == new_location, "Invalid location after reading");
276  logging::note("Coordinates are (%d, %d => %f, %f)",
277  std::get<0>(*new_location),
278  std::get<1>(*new_location),
279  std::get<0>(*new_location) + std::get<2>(*new_location) / 1000.0,
280  std::get<1>(*new_location) + std::get<3>(*new_location) / 1000.0);
281 #ifndef NDEBUG
282  logging::note("Values for %s range from %d to %d",
283  filename.c_str(),
284  min_value,
285  max_value);
286 #endif
287  return result;
288  }

339  void saveToAsciiFile(const string& dir,
340  const string& base_name,
341  std::function<R(T value)> convert) const
342  {
343  Idx min_row = 0;
344  Idx num_rows = this->rows();
345  Idx min_column = 0;
346  Idx num_columns = this->columns();
347  const double xll = this->xllcorner() + min_column * this->cellSize();
348  // offset is different for y since it's flipped
349  const double yll = this->yllcorner() + (min_row) * this->cellSize();
350  logging::extensive("Lower left corner is (%f, %f)", xll, yll);
351  ofstream out;
352  out.open(dir + base_name + ".asc");
353  write_ascii_header(out,
354  num_columns,
355  num_rows,
356  xll,
357  yll,
358  this->cellSize(),
359  static_cast<double>(this->noDataInt()));
360  // need to output in reverse order since (0,0) is bottom left
361  for (Idx r = num_rows - 1; r >= 0; --r)
362  {
363  for (Idx c = 0; c < num_columns; ++c)
364  {
365  const Location idx(static_cast<Idx>(r), static_cast<Idx>(c));
366  // HACK: use + here so that it gets promoted to a printable number
367  // prevents char type being output as characters
368  out << +(convert(this->at(idx)))
369  << " ";
370  }
371  out << "\n";
372  }
373  out.close();
374  this->createPrj(dir, base_name);
375  }
