149  [[nodiscard]] static ConstantGrid<T, V>* readTiff(const string& filename,
150  TIFF* tif,
151  GTIF* gtif,
152  const topo::Point& point,
153  std::function<T(int, int)> convert)
154  {
155  logging::info("Reading file %s", filename.c_str());
156 #ifndef NDEBUG
157  auto min_value = std::numeric_limits<int16>::max();
158  auto max_value = std::numeric_limits<int16>::min();
159 #endif
160  const GridBase grid_info = read_header<T>(tif, gtif);
161  const auto coordinates = grid_info.findFullCoordinates(point, false);
162  auto min_column = max(static_cast<FullIdx>(0),
163  static_cast<FullIdx>(std::get<1>(*coordinates) - static_cast<FullIdx>(MAX_COLUMNS) / static_cast<FullIdx>(2)));
164  if (min_column + MAX_COLUMNS >= grid_info.calculateColumns())
165  {
166  min_column = grid_info.calculateColumns() - MAX_COLUMNS;
167  }
168  const auto max_column = static_cast<FullIdx>(min_column + MAX_COLUMNS);
169  auto min_row = max(static_cast<FullIdx>(0),
170  static_cast<FullIdx>(std::get<0>(*coordinates) - static_cast<FullIdx>(MAX_ROWS) / static_cast<FullIdx>(2)));
171  if (min_row + MAX_COLUMNS >= grid_info.calculateRows())
172  {
173  min_row = grid_info.calculateRows() - MAX_COLUMNS;
174  }
175  const auto max_row = static_cast<FullIdx>(min_row + MAX_COLUMNS);
176  const FullIdx offset_x = -min_column;
177  const FullIdx offset_y = -min_row;
178  T no_data = convert(grid_info.nodata(), grid_info.nodata());
179  vector<T> values(MAX_COLUMNS * MAX_COLUMNS, no_data);
180  int tile_width;
181  int tile_length;
182  TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
183  TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
184  logging::warning("%s: malloc start", filename.c_str());
185  const auto buf = _TIFFmalloc(TIFFTileSize(tif));
186  logging::warning("%s: read start", filename.c_str());
187  const tsample_t smp{};
188  for (auto h = 0; h < grid_info.calculateRows(); h += tile_length)
189  {
190  const auto y_min = static_cast<FullIdx>(max(static_cast<FullIdx>(0), min_row - h));
191  const auto y_limit = min(static_cast<FullIdx>(tile_length),
192  min(static_cast<FullIdx>(grid_info.calculateRows()), max_row) - h);
193  for (auto w = 0; w < grid_info.calculateColumns(); w += tile_width)
194  {
195  TIFFReadTile(tif, buf, static_cast<uint32>(w), static_cast<uint32>(h), 0, smp);
196  const auto x_min = static_cast<FullIdx>(max(static_cast<FullIdx>(0), min_column - w));
197  const auto x_limit = min(static_cast<FullIdx>(tile_width),
198  min(grid_info.calculateColumns(),
199  max_column)
200  - w);
201  for (auto y = y_min; y < y_limit; ++y)
202  {
203  // read in so that (0, 0) has a hash of 0
204  const FullIdx i = static_cast<FullIdx>(MAX_COLUMNS) - (static_cast<FullIdx>(h) + y + offset_y + 1);
205  for (auto x = x_min; x < x_limit; ++x)
206  {
207  const auto cur_hash = static_cast<HashSize>(i) * MAX_COLUMNS + w + x + offset_x;
208  const auto offset = y * tile_length + x;
209  auto cur = *(static_cast<int16*>(buf) + offset);
210 #ifndef NDEBUG
211  min_value = min(cur, min_value);
212  max_value = max(cur, max_value);
213 #endif
214  values.at(cur_hash) = convert(cur, grid_info.nodata());
215  }
216  }
217  }
218  }
219  logging::warning("%s: read end", filename.c_str());
220  _TIFFfree(buf);
221  logging::warning("%s: free end", filename.c_str());
222  const auto new_xll = grid_info.xllcorner() - offset_x * grid_info.cellSize();
223  const auto new_yll = grid_info.yllcorner()
224  + (static_cast<double>(grid_info.calculateRows()) - max_row)
225  * grid_info.cellSize();
226  logging::check_fatal(new_yll < grid_info.yllcorner(),
227  "New yllcorner is outside original grid");
228  logging::note("Translated lower left is (%f, %f) from (%f, %f)",
229  new_xll,
230  new_yll,
231  grid_info.xllcorner(),
232  grid_info.yllcorner());
233  auto result = new ConstantGrid<T, V>(grid_info.cellSize(),
234  MAX_ROWS,
235  MAX_COLUMNS,
236  no_data,
237  grid_info.nodata(),
238  new_xll,
239  new_yll,
240  new_xll + MAX_COLUMNS * grid_info.cellSize(),
241  new_yll + MAX_ROWS * grid_info.cellSize(),
242  string(grid_info.proj4()),
243  std::move(values));
244  auto new_location = result->findCoordinates(point, true);
245  logging::check_fatal(nullptr == new_location, "Invalid location after reading");
246  logging::note("Coordinates are (%d, %d => %f, %f)",
247  std::get<0>(*new_location),
248  std::get<1>(*new_location),
249  std::get<0>(*new_location) + std::get<3>(*new_location) / 1000.0,
250  std::get<1>(*new_location) + std::get<2>(*new_location) / 1000.0);
251 #ifndef NDEBUG
252  logging::note("Values for %s range from %d to %d",
253  filename.c_str(),
254  min_value,
255  max_value);
256 #endif
257  return result;
258  }

308  template <class R>
309  void saveToAsciiFile(const string& dir,
310  const string& base_name,
311  std::function<R(T value)> convert) const
312  {
313  Idx min_row = 0;
314  Idx num_rows = MAX_ROWS;
315  Idx min_column = 0;
316  Idx num_columns = MAX_COLUMNS;
317  const double xll = this->xllcorner() + min_column * this->cellSize();
318  // offset is different for y since it's flipped
319  const double yll = this->yllcorner() + (min_row) * this->cellSize();
320  logging::verbose("Lower left corner is (%f, %f)", xll, yll);
321  ofstream out;
322  out.open(dir + base_name + ".asc");
323  write_ascii_header(out,
324  num_columns,
325  num_rows,
326  xll,
327  yll,
328  this->cellSize(),
329  static_cast<double>(this->noDataInt()));
330  for (Idx ro = 0; ro < num_rows; ++ro)
331  {
332  // HACK: do this so that we always get at least one pixel in output
333  // need to output in reverse order since (0,0) is bottom left
334  const Idx r = num_rows - 1 - ro;
335  for (Idx co = 0; co < num_columns; ++co)
336  {
337  const Location idx(static_cast<Idx>(r), static_cast<Idx>(min_column + co));
338  // HACK: use + here so that it gets promoted to a printable number
339  // prevents char type being output as characters
340  out << +(convert(this->at(idx)))
341  << " ";
342  }
343  out << "\n";
344  }
345  out.close();
346  this->createPrj(dir, base_name);
347  }
