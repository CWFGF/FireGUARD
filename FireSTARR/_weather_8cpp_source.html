35 TIMESTAMP_STRUCT read_timestamp(util::Database* db) noexcept
36 {
37  // HACK: do it this way so that we know database calls are in this order
38  const auto db_utc = db->getTimestamp();
39  tm utc{};
40  util::to_tm_gm(db_utc, &utc);
41  const auto utc_t = _mkgmtime(&utc);
42  tm local{};
43  localtime_s(&local, &utc_t);
44  TIMESTAMP_STRUCT result;
45  util::to_ts(local, &result);
46  if (result.hour != 13)
47  {
48  // HACK: not sure what to do with time zones yet, so just set it as 13:00
49  if (!WARNED_ABOUT_BAD_TIME_ZONE)
50  {
51  logging::warning(
52  "Overriding hour to be 13:00 even though it's being converted to %02d:00",
53  result.hour);
54  WARNED_ABOUT_BAD_TIME_ZONE = true;
55  }
56  result.hour = 13;
57  }
58  return result;
59 }
