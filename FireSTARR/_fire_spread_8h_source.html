74  [[nodiscard]] static constexpr double calculateRosFromThreshold(const double threshold)
75  {
76  // for some reason it returns -nan instead of nan if it's 1, so return this instead
77  if (1.0 == threshold)
78  {
79  return std::numeric_limits<double>::infinity();
80  }
81  if (0.0 == threshold)
82  {
83  return 0.0;
84  }
85  // NOTE: based off spread event probability from wotton
86  // should be the inverse of calculateSpreadProbability()
87  return 25.0 / 4.0 * log(-(exp(41.0 / 25.0) * threshold) / (threshold - 1));
88  }
