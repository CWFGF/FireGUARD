420 template <typename T>
421 [[nodiscard]] GridBase read_header(TIFF* tif, GTIF* gtif)
422 {
423  GTIFDefn definition;
424  if (GTIFGetDefn(gtif, &definition))
425  {
426  void* data;
427  uint32 count;
428  int columns;
429  int rows;
430  TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &columns);
431  logging::check_fatal(columns > numeric_limits<Idx>::max(),
432  "Cannot use grids with more than %d columns",
433  numeric_limits<Idx>::max());
434  TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &rows);
435  logging::check_fatal(rows > numeric_limits<Idx>::max(),
436  "Cannot use grids with more than %d rows",
437  numeric_limits<Idx>::max());
438  TIFFGetField(tif, GDAL_NODATA, &count, &data);
439  const auto nodata = stoi(string(static_cast<char*>(data)));
440  double x = 0.0;
441  double y = rows;
442  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
443  "Unable to translate image to PCS coordinates.");
444  const auto yllcorner = y;
445  const auto xllcorner = x;
446  logging::debug("Lower left for header is (%f, %f)", xllcorner, yllcorner);
447  double adf_coefficient[6] = {0};
448  x = 0.5;
449  y = 0.5;
450  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
451  "Unable to translate image to PCS coordinates.");
452  adf_coefficient[4] = x;
453  adf_coefficient[5] = y;
454  x = 1.5;
455  y = 0.5;
456  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
457  "Unable to translate image to PCS coordinates.");
458  const auto cell_width = x - adf_coefficient[4];
459  x = 0.5;
460  y = 1.5;
461  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
462  "Unable to translate image to PCS coordinates.");
463  const auto cell_height = y - adf_coefficient[5];
464  logging::check_fatal(cell_width != -cell_height,
465  "Can only use grids with square pixels");
466  const auto proj4_char = GTIFGetProj4Defn(&definition);
467  auto proj4 = string(proj4_char);
468  delete proj4_char;
469  const auto zone_pos = proj4.find("+zone=");
470  if (string::npos != zone_pos && string::npos != proj4.find("+proj=utm"))
471  {
472  // convert from utm zone to tmerc
473  const auto zone_str = proj4.substr(zone_pos + 6);
474  const auto zone = stoi(zone_str);
475  // zone 15 is -93 and other zones are 6 degrees difference
476  const auto degrees = static_cast<int>(6.0 * (zone - 15.0) - 93);
477  // HACK: assume utm zone is at start
478  proj4 = string(
479  "+proj=tmerc +lat_0=0.000000000 +lon_0=" + to_string(degrees) + ".000000000 +k=0.999600 +x_0=500000.000 +y_0=0.000");
480  }
481  return GridBase(cell_width,
482  static_cast<Idx>(rows),
483  static_cast<Idx>(columns),
484  nodata,
485  xllcorner,
486  yllcorner,
487  string(proj4));
488  }
489  throw runtime_error("Cannot read TIFF header");
490 }
491 template <class R>
492 [[nodiscard]] R with_tiff(const string& filename, function<R(TIFF*, GTIF*)> fct)
493 {
494  logging::debug("Reading file %s", filename.c_str());
495  // suppress warnings about geotiff tags that aren't found
496  TIFFSetWarningHandler(nullptr);
497  auto tif = XTIFFOpen(filename.c_str(), "r");
498  logging::check_fatal(!tif, "Cannot open file %s as a TIF", filename.c_str());
499  auto gtif = GTIFNew(tif);
500  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
501  try
502  {
503  R result = fct(tif, gtif);
504  if (tif)
505  XTIFFClose(tif);
506  if (gtif)
507  GTIFFree(gtif);
508  GTIFDeaccessCSV();
509  return result;
510  }
511  catch (std::exception&)
512  {
513  return logging::fatal<R>("Unable to process file %s", filename.c_str());
514  }
515 }
