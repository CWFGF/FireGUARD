392 template <typename T>
393 [[nodiscard]] GridBase read_header(TIFF* tif, GTIF* gtif)
394 {
395  GTIFDefn definition;
396  if (GTIFGetDefn(gtif, &definition))
397  {
398  void* data;
399  uint32 count;
400  int columns;
401  int rows;
402  TIFFGetField(tif, TIFFTAG_IMAGEWIDTH, &columns);
403  logging::check_fatal(columns > numeric_limits<Idx>::max(),
404  "Cannot use grids with more than %d columns",
405  numeric_limits<Idx>::max());
406  TIFFGetField(tif, TIFFTAG_IMAGELENGTH, &rows);
407  logging::check_fatal(rows > numeric_limits<Idx>::max(),
408  "Cannot use grids with more than %d rows",
409  numeric_limits<Idx>::max());
410  TIFFGetField(tif, GDAL_NODATA, &count, &data);
411  const auto nodata = stoi(string(static_cast<char*>(data)));
412  double x = 0.0;
413  double y = rows;
414  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
415  "Unable to translate image to PCS coordinates.");
416  const auto yllcorner = y;
417  const auto xllcorner = x;
418  logging::debug("Lower left for header is (%f, %f)", xllcorner, yllcorner);
419  double adf_coefficient[6] = {0};
420  x = 0.5;
421  y = 0.5;
422  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
423  "Unable to translate image to PCS coordinates.");
424  adf_coefficient[4] = x;
425  adf_coefficient[5] = y;
426  x = 1.5;
427  y = 0.5;
428  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
429  "Unable to translate image to PCS coordinates.");
430  const auto cell_width = x - adf_coefficient[4];
431  x = 0.5;
432  y = 1.5;
433  logging::check_fatal(!GTIFImageToPCS(gtif, &x, &y),
434  "Unable to translate image to PCS coordinates.");
435  const auto cell_height = y - adf_coefficient[5];
436  logging::check_fatal(cell_width != -cell_height,
437  "Can only use grids with square pixels");
438  const auto proj4_char = GTIFGetProj4Defn(&definition);
439  auto proj4 = string(proj4_char);
440  delete proj4_char;
441  const auto zone_pos = proj4.find("+zone=");
442  if (string::npos != zone_pos && string::npos != proj4.find("+proj=utm"))
443  {
444  // convert from utm zone to tmerc
445  const auto zone_str = proj4.substr(zone_pos + 6);
446  const auto zone = stoi(zone_str);
447  // zone 15 is -93 and other zones are 6 degrees difference
448  const auto degrees = static_cast<int>(6.0 * (zone - 15.0) - 93);
449  // HACK: assume utm zone is at start
450  proj4 = string(
451  "+proj=tmerc +lat_0=0.000000000 +lon_0=" + to_string(degrees) +
452  ".000000000 +k=0.999600 +x_0=500000.000 +y_0=0.000");
453  }
454  return GridBase(cell_width,
455  static_cast<Idx>(rows),
456  static_cast<Idx>(columns),
457  nodata,
458  xllcorner,
459  yllcorner,
460  string(proj4));
461  }
462  throw runtime_error("Cannot read TIFF header");
463 }
464 template <class R>
465 [[nodiscard]] R with_tiff(const string& filename, function<R(TIFF*, GTIF*)> fct)
466 {
467  logging::info("Reading file %s", filename.c_str());
468  // suppress warnings about geotiff tags that aren't found
469  TIFFSetWarningHandler(nullptr);
470  auto tif = XTIFFOpen(filename.c_str(), "r");
471  logging::check_fatal(!tif, "Cannot open file %s as a TIF", filename.c_str());
472  auto gtif = GTIFNew(tif);
473  logging::check_fatal(!gtif, "Cannot open file %s as a GEOTIFF", filename.c_str());
474  try
475  {
476  R result = fct(tif, gtif);
477  if (tif) XTIFFClose(tif);
478  if (gtif) GTIFFree(gtif);
479  GTIFDeaccessCSV();
480  return result;
481  }
482  catch (std::exception&)
483  {
484  return logging::fatal<R>("Unable to process file %s", filename.c_str());
485  }
486 }
