229  [[nodiscard]] static data::ConstantGrid<Cell>* makeCells(
230  const FuelGrid& fuel,
231  const SlopeGrid& slope,
232  const AspectGrid& aspect)
233  {
234  logging::check_fatal(fuel.yllcorner() != slope.yllcorner(),
235  "Expected yllcorner %f but got %f",
236  fuel.yllcorner(),
237  slope.yllcorner());
238  logging::check_fatal(fuel.yllcorner() != aspect.yllcorner(),
239  "Expected yllcorner %f but got %f",
240  fuel.yllcorner(),
241  aspect.yllcorner());
242  static Cell nodata{};
243  auto values = vector<Cell>{fuel.data.size()};
244  //fuel.saveToAsciiFile("C:/FireGUARD/", "fuel");
245  vector<HashSize> hashes{};
246  for (HashSize h = 0; h < static_cast<size_t>(fuel.rows()) * fuel.columns(); ++h)
247  {
248  hashes.emplace_back(h);
249  }
250  std::for_each(
251  std::execution::par_unseq,
252  hashes.begin(),
253  hashes.end(),
254  [&fuel, &slope, &aspect, &values](auto&& h)
255  {
256  const auto s = min(static_cast<SlopeSize>(MAX_SLOPE_FOR_DISTANCE),
257  slope.at(h));
258  const auto a = 0 == s ? static_cast<AspectSize>(0) : aspect.at(h);
259  const auto f = fuel::FuelType::safeCode(fuel.at(h));
260  const auto cell = Cell(h, s, a, f);
261  values.at(h) = cell;
262 #ifdef _DEBUG
263 #ifndef VLD_RPTHOOK_INSTALL
264  const topo::Location loc{ h };
265  const auto r = loc.row();
266  const auto c = loc.column();
267  logging::check_fatal(cell.row() != r, "Cell row %d not %d", cell.row(), r);
268  logging::check_fatal(cell.column() != c, "Cell column %d not %d", cell.column(), c);
269  logging::check_fatal(cell.slope() != s, "Cell slope %d not %d", cell.slope(), s);
270  logging::check_fatal(cell.aspect() != a, "Cell aspect %d not %d", cell.aspect(), a);
271  logging::check_fatal(cell.fuelCode() != f, "Cell fuel %d not %d", cell.fuelCode(), f);
272  const auto v = values.at(h);
273  logging::check_fatal(v.row() != r, "Row %d not %d", v.row(), r);
274  logging::check_fatal(v.column() != c, "Column %d not %d", v.column(), c);
275  logging::check_fatal(v.slope() != s, "Slope %d not %d", v.slope(), s);
276  if (0 != s)
277  {
278  logging::check_fatal(v.aspect() != a, "Aspect %d not %d", v.aspect(), a);
279  }
280  else
281  {
282  logging::check_fatal(v.aspect() != 0, "Aspect %d not %d", v.aspect(), 0);
283  }
284  logging::check_fatal(v.fuelCode() != f, "Fuel %d not %d", v.fuelCode(), f);
285 #endif
286 #endif
287  });
288  return new data::ConstantGrid<Cell>(fuel.cellSize(),
289  fuel.rows(),
290  fuel.columns(),
291  nodata,
292  -1,
293  fuel.xllcorner(),
294  fuel.yllcorner(),
295  string(fuel.proj4()),
296  std::move(values));
297  }
