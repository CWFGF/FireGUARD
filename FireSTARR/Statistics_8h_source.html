251  // values should already be sorted
252  // std::sort(values.begin(), values.end());
253  n_ = values.size();
254  min_ = values[0];
255  max_ = values[n_ - 1];
256  median_ = values[n_ / 2];
257  const auto total_sum = std::accumulate(values.begin(),
258  values.end(),
259  0.0,
260  [](const double t, const double x)
261  {
262  return t + x;
263  });
264  mean_ = total_sum / n_;
265  for (size_t i = 0; i < percentiles_.size(); ++i)
266  {
267  const auto pos = std::min(n_ - 1,
268  static_cast<size_t>(truncl(
269  (static_cast<double>(i) / (percentiles_.size() - 1)) * n_)));
270  // note("For %d values %dth percentile is at %d", n_, i, pos);
271  percentiles_[i] = values[pos];
272  }
273  const auto total = std::accumulate(values.begin(),
274  values.end(),
275  0.0,
276  [this](const double t, const double x)
277  {
278  return t + pow_int<2>(x - mean_);
279  });
280  standard_deviation_ = sqrt(total / n_);
