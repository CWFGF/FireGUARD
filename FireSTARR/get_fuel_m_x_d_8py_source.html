49  theMXD = arcpy.mapping.MapDocument(copyMXD)
50  txtAssumptions = ("FBP Fuels are derived from eFRI data and the CFS national fuel grid.\n" +

52  "Fuels substituted in the following manner:\n" +
53  '\n'.join(find_lines(sim_output, 'is treated like', 'Fuel')))
54  fbp_raster = findFuelRaster(sim_output)
55  fbp = setDataSource(theMXD, "*FBP*", fbp_raster)
56  setCommon(theMXD,
57  fire,
58  txtAssumptions,
59  " ",
60  run_output,
61  fire,
62  perim,
63  extent,
64  fbp_raster)
