41 [[nodiscard]] constexpr int calculate_grass_curing(const int nd)
42 {
43  return (nd < START_GREENING)
44  ? // we're before foliar moisture dip has started
45  100
46  : (nd >= 50)
47  ? 0 // foliar moisture is at 120.0, so grass should be totally uncured
48  // HACK: invent a formula that has 50% curing at the bottom of the foliar
49  // moisture dip foliar moisture above ranges between 120 and 85, with 85
50  // being at the point where we want 50% cured Curing:
51  // -43 => 100, 0 => 50, 50 => 0 least-squares best fit:
52  : static_cast<int>(52.5042 - 1.07324 * nd);
53 }
