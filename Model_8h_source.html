51 void notify()
52 {
53 std::unique_lock<std::mutex> l(mutex_);
54 ++count_;
55 cv_.notify_one();
56 }
60 void wait()
61 {
62 std::unique_lock<std::mutex> l(mutex_);
63 cv_.wait(l, [this] { return count_ != 0; });
64 --count_;
65 }
66private:
