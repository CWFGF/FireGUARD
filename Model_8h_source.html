60  void notify()
61  {
62  std::unique_lock<std::mutex> l(mutex_);
63  ++count_;
64  cv_.notify_one();
65  }
69  void wait()
70  {
71  std::unique_lock<std::mutex> l(mutex_);
72  cv_.wait(l, [this] { return count_ != 0; });
