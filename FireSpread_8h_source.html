64  [[nodiscard]] static constexpr double calculateRosFromThreshold(const double threshold)
65  {
66  // for some reason it returns -nan instead of nan if it's 1, so return this instead
67  if (1.0 == threshold)
68  {
69  return std::numeric_limits<double>::infinity();
70  }
71  if (0.0 == threshold)
72  {
73  return 0.0;
74  }
75  // NOTE: based off spread event probability from wotton
76  // should be the inverse of calculateSpreadProbability()
77  return 25.0 / 4.0 * log(-(exp(41.0 / 25.0) * threshold) / (threshold - 1));
78  }
