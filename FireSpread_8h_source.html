61 [[nodiscard]] static constexpr MathSize calculateRosFromThreshold(const ThresholdSize threshold)
62 {
63 // for some reason it returns -nan instead of nan if it's 1, so return this instead
64 if (1.0 == threshold)
65 {
66 return std::numeric_limits<ThresholdSize>::infinity();
67 }
68 if (0.0 == threshold)
69 {
70 return 0.0;
71 }
72 // NOTE: based off spread event probability from wotton
73 // should be the inverse of calculateSpreadProbability()
74 return 25.0 / 4.0 * log(-(exp(41.0 / 25.0) * threshold) / (threshold - 1));
75 }
