85 def rh_from_sh(shum, tmp, pressure = 1013.25):
86  """!
87  Convert specific humidity to relative humidity
88  @param shum Specific humidity (dimensionless, e.g. kg/kg)
89  @param tmp Temperature (Celcius)
90  @param pressure Pressure (mb)
91  @return Relative Humidity (%)
92  """
93  es = 6.112 * math.exp((17.67 * tmp) / (tmp + 243.5))
94  e = shum * pressure / (0.378 * shum + 0.622)
95  rh = e / es
96  return rh * 100
