178  fixLocation(id);
179  var theZone = document.getElementById("txtzone").value;
180  var basemap = document.getElementById("txtbasemap").value;
181  if (0 >= theZone.length || 5 != basemap.length)
182  {
183  // zone and basemap not valid
184  return;
185  }
186  var block = 55;//document.getElementById("txtblock").value;
187 
188  if(IsNumeric(theZone) & IsNumeric(basemap) & IsNumeric(block)){
189  var easting = parseFloat(basemap.toString().substring(0,2)+block.toString().substring(0,1)+"000");
190  var northing = parseFloat(basemap.toString().substring(2,5)+block.toString().substring(1,2)+"000");
191  var zone = parseInt(theZone);
192  var southern = false;//parseFloat(document.getElementById("utmHemi").selectedIndex) == 1;
193 
194  if (isNaN(easting) || isNaN(northing)) {
195  return;
196  }
197 
198  if (isNaN(zone)) {
199  return;
200  }
201 
202 
203  if (northing < 0 || northing > 10000000) {
204  return;
205  }
206 
207  utmconv.setDatum(1);
208  var latlon = utmconv.utmToLatLng(easting, northing, zone, southern); // get lat/lon for this set of coordinates
209 
210 
211  //
212  // 0 is N or W hemisphere, 1 is S or E
213  //
214 
215  setDecimalDegrees(latlon.lat, latlon.lng);
216  }
217  else if(theZone==="" || basemap==="" )
218  return;
219  $('#txtzone, #txtbasemap, #txtLATDeg, #txtLONGDeg').valid();
220  }

223  function changeLATLONGtoUTM(id){
224  fixLocation(id);
225  var lat = document.getElementById("txtLATDeg").value;
226  var lon = document.getElementById("txtLONGDeg").value;
227 
228  if(IsNumeric(lat) & IsNumeric(lon)){
229 
230  lat = parseFloat(document.getElementById("txtLATDeg").value);
231 
232  lon = parseFloat(document.getElementById("txtLONGDeg").value);
233 
234  if (BOUNDS.latitude.min <= parseFloat(lat) && parseFloat(lat) <= BOUNDS.latitude.max){
235  if(BOUNDS.longitude.min <= parseFloat(lon) && parseFloat(lon)<= BOUNDS.longitude.max){
236  var mapDatum = 1;
237  utmconv.setDatum(mapDatum);
238  var coords = utmconv.latLngToUtm(lat, lon);
239  document.getElementById("txtLATDeg").value = parseFloat(document.getElementById("txtLATDeg").value).toFixed(5);
240  document.getElementById("txtLONGDeg").value = parseFloat(document.getElementById("txtLONGDeg").value).toFixed(5);
241  setUTM(coords.global.easting, coords.global.northing, coords.global.zone);//, coords.global.southern);
242  }
243  }
244  }
245  else if(lat==="" || lon==="" )
246  return;
247  $('#txtzone, #txtbasemap, #txtLATDeg, #txtLONGDeg').valid();
248  }

307  function loadSelect(id, xmlhttp)
308  {
309  optionValues[id] = null;
310  if (xmlhttp.readyState == 4 && xmlhttp.status == 200) {
311  var elem = $('#' + id);
312  var placeholder = elem.attr('placeholder');
313  try
314  {
315  values = JSON.parse(xmlhttp.responseText);
316  for (var v in values)
317  {
318  var opt = $('<option>', { value: v, text: v });
319  opt.css('color', '#000000');
320  elem.append(opt);
321  }
322  $('#' + id + ' option[value=""]').text(placeholder);
323  optionValues[id] = values;
324  var hidValue = $('#' + id + 'Hidden').val();
325  if (hidValue)
326  {
327  elem.val(hidValue);
328  }
329  }
330  catch(e)
331  {
332  $('#' + id + ' option[value=""]').text("No " + placeholder + "s");
333  $('#' + id).attr("disabled", true);
334  }
335  }
336  }
