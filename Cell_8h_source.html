140 protected:
141  /*
142  * Field Natural Range Used Range Bits Bit Range
143  * Row 0 - 2047 0 - 2047 11 0 - 2047
144  * Column 0 - 2047 0 - 2047 11 0 - 2047
145  * PADDING 10
146  * Fuel 0 - 56 0 - 56 6 0 - 63
147  * Aspect 0 - 359 0 - 359 9 0 - 511
148  * Slope 0 - infinity 0 - 127 7 0 - 127
149  * Extra 10
150  *
151  * Rows and Columns are restricted to 2048 since that's what gets clipped out of
152  * the GIS outputs.
153  *
154  * Fuel is tied to how many variations of percent conifer/dead fir we want to use, and
155  * if we want to allow M1/M2/M3/M4 on their own, or just use the ones that are
156  * automatically tied to the green-up.
157  *
158  * Aspect is calculated to be in degrees, so 0 - 359.
159  *
160  * Slope is truncated to 0 - 60 because that's the range that affects effective wind
161  * speed for slopes, but there's an issue with this when it tries to calculate the
162  * horizontal rate of spread since then the slope has been truncated and the distance
163  * calculated will be wrong.
164  */
168  static constexpr uint32_t FuelShift = 32;
169  // Need to make sure that fuel, slope & aspect aren't in first 32 bits
170  static_assert(32 <= FuelShift);
174  static constexpr uint32_t FuelBits = std::bit_width<uint32_t>(NUMBER_OF_FUELS);
178  static constexpr Topo FuelBitMask = util::bit_mask<FuelBits, Topo>();
179  static_assert(FuelBitMask == 0x3F);
180  static_assert(FuelBitMask >= NUMBER_OF_FUELS);
184  static constexpr Topo FuelMask = FuelBitMask << FuelShift;
188  static constexpr uint32_t AspectShift = FuelBits + FuelShift;
192  static constexpr uint32_t AspectBits = std::bit_width<uint32_t>(MAX_ASPECT);
196  static constexpr Topo AspectBitMask = util::bit_mask<AspectBits, Topo>();
197  static_assert(AspectBitMask == 0x1FF);
201  static constexpr Topo AspectMask = AspectBitMask << AspectShift;
205  static constexpr uint32_t SlopeShift = AspectBits + AspectShift;
209  static constexpr uint32_t SlopeBits = std::bit_width<uint32_t>(MAX_SLOPE_FOR_DISTANCE);
210  static_assert(SlopeBits == 7);
214  static constexpr Topo SlopeBitMask = util::bit_mask<SlopeBits, Topo>();
215  static_assert(SlopeBitMask == 0x7F);
216  static_assert(SlopeBitMask >= MAX_SLOPE_FOR_FACTOR);
220  static constexpr Topo SlopeMask = SlopeBitMask << SlopeShift;
224  static constexpr Topo CellMask = HashMask | FuelMask | AspectMask | SlopeMask;
225  static_assert(static_cast<size_t>(std::bit_width(std::numeric_limits<Topo>::max())) >= SlopeBits + SlopeShift);
226 };
