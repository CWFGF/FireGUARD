212 /*
213 * Field Natural Range Used Range Bits Bit Range
214 * Row 0 - 4095 0 - 4095 12 0 - 4095
215 * Column 0 - 4095 0 - 4095 12 0 - 4095
216 * PADDING 10
217 * Fuel 0 - 56 0 - 56 6 0 - 63
218 * Aspect 0 - 359 0 - 359 9 0 - 511
219 * Slope 0 - infinity 0 - 127 7 0 - 127
220 * Extra 10
221 *
222 * Rows and Columns are restricted to 4096 since that's what gets clipped out of
223 * the GIS outputs.
224 *
225 * Fuel is tied to how many variations of percent conifer/dead fir we want to use, and
226 * if we want to allow M1/M2/M3/M4 on their own, or just use the ones that are
227 * automatically tied to the green-up.
228 *
229 * Aspect is calculated to be in degrees, so 0 - 359.
230 *
231 * Slope is truncated to 0 - 60 because that's the range that affects effective wind
232 * speed for slopes, but there's an issue with this when it tries to calculate the
233 * horizontal rate of spread since then the slope has been truncated and the distance
234 * calculated will be wrong.
235 */
239 static constexpr uint32_t FuelShift = 32;
240 // Need to make sure that fuel, slope & aspect aren't in first 32 bits
241 static_assert(32 <= FuelShift);
245 static constexpr uint32_t FuelBits = std::bit_width<uint32_t>(NUMBER_OF_FUELS);
249 static constexpr Topo FuelBitMask = util::bit_mask<FuelBits, Topo>();
250 static_assert(FuelBitMask == 0x3F);
251 static_assert(FuelBitMask >= NUMBER_OF_FUELS);
255 static constexpr Topo FuelMask = FuelBitMask << FuelShift;
259 static constexpr uint32_t AspectShift = FuelBits + FuelShift;
263 static constexpr uint32_t AspectBits = std::bit_width<uint32_t>(MAX_ASPECT);
267 static constexpr Topo AspectBitMask = util::bit_mask<AspectBits, Topo>();
268 static_assert(AspectBitMask == 0x1FF);
272 static constexpr Topo AspectMask = AspectBitMask << AspectShift;
276 static constexpr uint32_t SlopeShift = AspectBits + AspectShift;
280 static constexpr uint32_t SlopeBits = std::bit_width<uint32_t>(MAX_SLOPE_FOR_DISTANCE);
281 static_assert(SlopeBits == 7);
285 static constexpr Topo SlopeBitMask = util::bit_mask<SlopeBits, Topo>();
286 static_assert(SlopeBitMask == 0x7F);
287 static_assert(SlopeBitMask >= MAX_SLOPE_FOR_FACTOR);
291 static constexpr Topo SlopeMask = SlopeBitMask << SlopeShift;
295 static constexpr Topo CellMask = HashMask | FuelMask | AspectMask | SlopeMask;
296 static_assert(static_cast<size_t>(std::bit_width(std::numeric_limits<Topo>::max())) >= SlopeBits + SlopeShift);
297};
