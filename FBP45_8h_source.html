39 [[nodiscard]] constexpr int calculate_grass_curing(const int nd)
40 {
41  return (nd < START_GREENING)
42  ? // we're before foliar moisture dip has started
43  100
44  : (nd >= 50)
45  ? 0 // foliar moisture is at 120.0, so grass should be totally uncured
46  // HACK: invent a formula that has 50% curing at the bottom of the foliar
47  // moisture dip foliar moisture above ranges between 120 and 85, with 85
48  // being at the point where we want 50% cured Curing:
49  // -43 => 100, 0 => 50, 50 => 0 least-squares best fit:
50  : static_cast<int>(52.5042 - 1.07324 * nd);
51 }
