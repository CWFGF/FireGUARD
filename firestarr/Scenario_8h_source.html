435  const double time_at_location) const
436  {
437  try
438  {
439  const auto fire_wx = weather_;
440  // // NOTE: Does using daily makes sense if we're looking at moisture?
441  // // HACK: use daily with diurnal curves to be consistent with pre-hourly wx version
442  // const auto fire_wx = weather_daily_;
443  const auto wx = fire_wx->at(time);
444  // use Mike's table
445  const auto mc = wx->mcDmcPct();
446  if (100 > mc
447  || (109 >= mc && 5 > time_at_location)
448  || (119 >= mc && 4 > time_at_location)
449  || (131 >= mc && 3 > time_at_location)
450  || (145 >= mc && 2 > time_at_location)
451  || (218 >= mc && 1 > time_at_location))
452  {
453  return true;
454  }
455  // we can look by fuel type because the entire landscape shares the weather
456  return extinctionThreshold(time) < fire_wx->survivalProbability(
457  time,
458  cell.fuelCode());
459  }
460  catch (const std::out_of_range& e)
461  {
462  // FIX: just ignore for now
463  // std::cerr << e.what() << '\n';
464  // logging::warning("Survival is checking for weather that doesn't exist at %f", time);
465  // no weather, so don't survive
466  return false;
467  }
