251  explicit Statistics(vector<double> values)
252  {
253  // values should already be sorted
254  // std::sort(values.begin(), values.end());
255  n_ = values.size();
256  min_ = values[0];
257  max_ = values[n_ - 1];
258  median_ = values[n_ / 2];
259  const auto total_sum = std::accumulate(values.begin(),
260  values.end(),
261  0.0,
262  [](const double t, const double x) { return t + x; });
263  mean_ = total_sum / n_;
264  for (size_t i = 0; i < percentiles_.size(); ++i)
265  {
266  const auto pos = std::min(n_ - 1,
267  static_cast<size_t>(truncl(
268  (static_cast<double>(i) / (percentiles_.size() - 1)) * n_)));
269  // note("For %d values %dth percentile is at %d", n_, i, pos);
270  percentiles_[i] = values[pos];
271  }
272  const auto total = std::accumulate(values.begin(),
273  values.end(),
274  0.0,
275  [this](const double t, const double x) { return t + pow_int<2>(x - mean_); });
276  standard_deviation_ = sqrt(total / n_);
277  sample_variance_ = total / (n_ - 1);
278 #ifndef NDEBUG
279  logging::check_fatal(min_ != percentiles_[0],
